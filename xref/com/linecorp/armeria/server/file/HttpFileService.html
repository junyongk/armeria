101     /**
102      * An LRU cache map that releases the buffer that contains the cached content.
103      */
104     private final Map<String, CachedEntry> cache;

106     HttpFileService(HttpFileServiceConfig config) {
107         this.config = requireNonNull(config, "config");
108 
109         if (config.maxCacheEntries() != 0) {
110             cache = Collections.synchronizedMap(new LruMap<String, CachedEntry>(config.maxCacheEntries()));
111         } else {
112             cache = null;
113         }
114     }

123     @Override
124     protected void doGet(ServiceRequestContext ctx, HttpRequest req, HttpResponseWriter res) {
125         final Entry entry = getEntry(ctx, req);
126         final long lastModifiedMillis = entry.lastModifiedMillis();
127 
128         if (lastModifiedMillis == 0) {
129             res.respond(HttpStatus.NOT_FOUND);
130             return;
131         }
132 
133         long ifModifiedSinceMillis = Long.MIN_VALUE;
134         try {
135             ifModifiedSinceMillis =
136                     req.headers().getTimeMillis(HttpHeaderNames.IF_MODIFIED_SINCE, Long.MIN_VALUE);
137         } catch (Exception e) {
138             // Ignore the ParseException, which is raised on malformed date.
139             //noinspection ConstantConditions
140             if (!(e instanceof ParseException)) {
141                 throw e;
142             }
143         }
144 
145         // HTTP-date does not have subsecond-precision; add 999ms to it.
146         if (ifModifiedSinceMillis > Long.MAX_VALUE - 999) {
147             ifModifiedSinceMillis = Long.MAX_VALUE;
148         } else {
149             ifModifiedSinceMillis += 999;
150         }
151 
152         if (lastModifiedMillis < ifModifiedSinceMillis) {
153             res.write(HttpHeaders.of(HttpStatus.NOT_MODIFIED)
154                                  .setTimeMillis(HttpHeaderNames.DATE, config().clock().millis())
155                                  .setTimeMillis(HttpHeaderNames.LAST_MODIFIED, lastModifiedMillis));
156             res.close();
157             return;
158         }
159 
160         final HttpData data;
161         try {
162             data = entry.readContent();
163         } catch (FileNotFoundException ignored) {
164             res.respond(HttpStatus.NOT_FOUND);
165             return;
166         } catch (Exception e) {
167             logger.warn("{} Unexpected exception reading a file:", ctx, e);
168             res.respond(HttpStatus.INTERNAL_SERVER_ERROR);
169             return;
170         }
171 
172         HttpHeaders headers = HttpHeaders.of(HttpStatus.OK)
173                                          .setInt(HttpHeaderNames.CONTENT_LENGTH, data.length())
174                                          .setTimeMillis(HttpHeaderNames.DATE, config().clock().millis())
175                                          .setTimeMillis(HttpHeaderNames.LAST_MODIFIED, lastModifiedMillis);
176         if (entry.mediaType() != null) {
177             headers.set(HttpHeaderNames.CONTENT_TYPE, entry.mediaType().toString());
178         }
179         if (entry.contentEncoding() != null) {
180             headers.set(HttpHeaderNames.CONTENT_ENCODING, entry.contentEncoding());
181         }
182         res.write(headers);
183         res.write(data);
184         res.close();
185     }

187     private Entry getEntry(ServiceRequestContext ctx, HttpRequest req) {
188         final String mappedPath = ctx.mappedPath();
189 
190         EnumSet<FileServiceContentEncoding> supportedEncodings =
191                 EnumSet.noneOf(FileServiceContentEncoding.class);
192 
193         if (config.serveCompressedFiles()) {
194             // We do a simple parse of the accept-encoding header, without worrying about star values
195             // or priorities.
196             String acceptEncoding = req.headers().get(HttpHeaderNames.ACCEPT_ENCODING);
197             if (acceptEncoding != null) {
198                 for (String encoding : COMMA_SPLITTER.split(acceptEncoding)) {
199                     for (FileServiceContentEncoding possibleEncoding : FileServiceContentEncoding.values()) {
200                         if (encoding.contains(possibleEncoding.headerValue)) {
201                             supportedEncodings.add(possibleEncoding);
202                         }
203                     }
204                 }
205             }
206         }
207 
208         final Entry entry = getEntryWithSupportedEncodings(mappedPath, supportedEncodings);
209 
210         if (entry.lastModifiedMillis() == 0) {
211             if (mappedPath.charAt(mappedPath.length() - 1) == '/') {
212                 // Try index.html if it was a directory access.
213                 final Entry indexEntry = getEntryWithSupportedEncodings(
214                         mappedPath + "index.html", supportedEncodings);
215                 if (indexEntry.lastModifiedMillis() != 0) {
216                     return indexEntry;
217                 }
218             }
219         }
220 
221         return entry;
222     }
223 
224     private Entry getEntry(String path, @Nullable String contentEncoding) {
225         assert path != null;
226 
227         if (cache == null) {
228             return config.vfs().get(path, contentEncoding);
229         }
230 
231         CachedEntry e = cache.get(path);
232         if (e != null) {
233             return e;
234         }
235 
236         e = new CachedEntry(config.vfs().get(path, contentEncoding), config.maxCacheEntrySizeBytes());
237         cache.put(path, e);
238         return e;
239     }
240 
241     private Entry getEntryWithSupportedEncodings(String path,
242                                                  EnumSet<FileServiceContentEncoding> supportedEncodings) {
243         for (FileServiceContentEncoding encoding : supportedEncodings) {
244             final Entry entry = getEntry(path + encoding.extension, encoding.headerValue);
245             if (entry.lastModifiedMillis() != 0) {
246                 return entry;
247             }
248         }
249         return getEntry(path, null);
250     }

252     private static final class CachedEntry implements Entry {
253 
254         private final Entry entry;
255         private final int maxCacheEntrySizeBytes;
256         private HttpData cachedContent;
257         private volatile long cachedLastModifiedMillis;
258 
259         CachedEntry(Entry entry, int maxCacheEntrySizeBytes) {
260             this.entry = entry;
261             this.maxCacheEntrySizeBytes = maxCacheEntrySizeBytes;
262             cachedLastModifiedMillis = entry.lastModifiedMillis();
263         }
264 
265         @Override
266         public MediaType mediaType() {
267             return entry.mediaType();
268         }
269 
270         @Nullable
271         @Override
272         public String contentEncoding() {
273             return entry.contentEncoding();
274         }
275 
276         @Override
277         public long lastModifiedMillis() {
278             final long newLastModifiedMillis = entry.lastModifiedMillis();
279             if (newLastModifiedMillis != cachedLastModifiedMillis) {
280                 cachedLastModifiedMillis = newLastModifiedMillis;
281                 destroyContent();
282             }
283 
284             return newLastModifiedMillis;
285         }
286 
287         @Override
288         public synchronized HttpData readContent() throws IOException {
289             if (cachedContent == null) {
290                 final HttpData newContent = entry.readContent();
291                 if (newContent.length() > maxCacheEntrySizeBytes) {
292                     // Do not cache if the content is too large.
293                     return newContent;
294                 }
295                 cachedContent = newContent;
296             }
297 
298             return cachedContent;
299         }
300 
301         synchronized void destroyContent() {
302             if (cachedContent != null) {
303                 cachedContent = null;
304             }
305         }
306 
307         @Override
308         public String toString() {
309             return entry.toString();
310         }
311     }

314     /**
315      * Creates a new {@link HttpService} that tries this {@link HttpFileService} first and then the specified
316      * {@link HttpService} when this {@link HttpFileService} does not have a requested resource.
317      *
318      * @param nextService the {@link HttpService} to try secondly
319      */
320     public HttpService orElse(Service<HttpRequest, HttpResponse> nextService) {
321         requireNonNull(nextService, "nextService");
322         return new OrElseHttpService(this, nextService);
323     }
324 
325     private static final class OrElseHttpService extends AbstractHttpService {
326 
327         private final HttpFileService first;
328         private final Service<HttpRequest, HttpResponse> second;
329 
330         OrElseHttpService(HttpFileService first, Service<HttpRequest, HttpResponse> second) {
331             this.first = first;
332             this.second = second;
333         }
334 
335         @Override
336         public void serviceAdded(ServiceConfig cfg) throws Exception {
337             first.serviceAdded(cfg);
338             second.serviceAdded(cfg);
339         }
340 
341         @Override
342         public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
343             final Entry firstEntry = first.getEntry(ctx, req);
344             if (firstEntry.lastModifiedMillis() != 0) {
345                 return first.serve(ctx, req);
346             } else {
347                 return second.serve(ctx, req);
348             }
349         }
350     }

352     /**
353      * Content encodings supported by {@link HttpFileService}. Will generally support more formats than
354      * {@link HttpEncodingService} because new formats can be added as soon as browsers and build tools
355      * support them, without having to implement on-the-fly compression.
356      */
357     private enum FileServiceContentEncoding {
358         // Order matters, we use the enum ordinal as the priority to pick an encoding in. Encodings should
359         // be ordered by priority.
360         BROTLI(".br", "br"),
361         GZIP(".gz", "gzip");
362 
363         private final String extension;
364         private final String headerValue;
365 
366         FileServiceContentEncoding(String extension, String headerValue) {
367             this.extension = extension;
368             this.headerValue = headerValue;
369         }
370     }
