40  /**
41   * A <a href="https://en.wikipedia.org/wiki/Virtual_hosting#Name-based">name-based virtual host</a>.
42   * A {@link VirtualHost} contains the following information:
43   * <ul>
44   *   <li>the hostname pattern, as defined in
45   *       <a href="http://tools.ietf.org/html/rfc2818#section-3.1">the section 3.1 of RFC2818</a></li>
46   *   <li>{@link SslContext} if TLS is enabled</li>
47   *   <li>the list of available {@link Service}s and their {@link PathMapping}s</li>
48   * </ul>
49   *
50   * @see VirtualHostBuilder
51   */
52  public final class VirtualHost {
53  
54      private static final Pattern HOSTNAME_PATTERN = Pattern.compile(
55              "^(?:[-_a-zA-Z0-9]|[-_a-zA-Z0-9][-_.a-zA-Z0-9]*[-_a-zA-Z0-9])$");
56  
57      /**
58       * Initialized later by {@link ServerConfig} via {@link #setServerConfig(ServerConfig)}.
59       */
60      private ServerConfig serverConfig;
61  
62      private final String defaultHostname;
63      private final String hostnamePattern;
64      private final SslContext sslContext;
65      private final List<ServiceConfig> services;
66      private final PathMappings<ServiceConfig> serviceMapping = new PathMappings<>();
67  
68      private String strVal;
69  
70      VirtualHost(String defaultHostname, String hostnamePattern,
71                  SslContext sslContext, Iterable<ServiceConfig> serviceConfigs) {
72  
73          defaultHostname = normalizeDefaultHostname(defaultHostname);
74          hostnamePattern = normalizeHostnamePattern(hostnamePattern);
75          ensureHostnamePatternMatchesDefaultHostname(hostnamePattern, defaultHostname);
76  
77          this.defaultHostname = defaultHostname;
78          this.hostnamePattern = hostnamePattern;
79          this.sslContext = validateSslContext(sslContext);
80  
81          requireNonNull(serviceConfigs, "serviceConfigs");
82  
83          final List<ServiceConfig> servicesCopy = new ArrayList<>();
84  
85          for (ServiceConfig c : serviceConfigs) {
86              c = c.build(this);
87              servicesCopy.add(c);
88              serviceMapping.add(c.pathMapping(), c);
89          }
90  
91          services = Collections.unmodifiableList(servicesCopy);
92          serviceMapping.freeze();
93      }
94  
95      /**
96       * IDNA ASCII conversion, case normalization and validation.
97       */
98      static String normalizeDefaultHostname(String defaultHostname) {
99          requireNonNull(defaultHostname, "defaultHostname");
100         if (needsNormalization(defaultHostname)) {
101             defaultHostname = IDN.toASCII(defaultHostname, IDN.ALLOW_UNASSIGNED);
102         }
103 
104         if (!HOSTNAME_PATTERN.matcher(defaultHostname).matches()) {
105             throw new IllegalArgumentException("defaultHostname: " + defaultHostname);
106         }
107 
108         return Ascii.toLowerCase(defaultHostname);
109     }
110 
111     /**
112      * IDNA ASCII conversion, case normalization and validation.
113      */
114     static String normalizeHostnamePattern(String hostnamePattern) {
115         requireNonNull(hostnamePattern, "hostnamePattern");
116         if (needsNormalization(hostnamePattern)) {
117             hostnamePattern = IDN.toASCII(hostnamePattern, IDN.ALLOW_UNASSIGNED);
118         }
119 
120         if (!"*".equals(hostnamePattern) &&
121             !HOSTNAME_PATTERN.matcher(hostnamePattern.startsWith("*.") ? hostnamePattern.substring(2)
122                                                                        : hostnamePattern).matches()) {
123             throw new IllegalArgumentException("hostnamePattern: " + hostnamePattern);
124         }
125 
126         return Ascii.toLowerCase(hostnamePattern);
127     }
128 
129     /**
130      * Ensure that 'hostnamePattern' matches 'defaultHostname'.
131      */
132     static void ensureHostnamePatternMatchesDefaultHostname(String hostnamePattern, String defaultHostname) {
133         if ("*".equals(hostnamePattern)) {
134             return;
135         }
136 
137         // Pretty convoluted way to validate but it's done only once and
138         // we don't need to duplicate the pattern matching logic.
139         final DomainNameMapping<Boolean> mapping =
140                 new DomainNameMappingBuilder<>(Boolean.FALSE).add(hostnamePattern, Boolean.TRUE).build();
141 
142         if (!mapping.map(defaultHostname)) {
143             throw new IllegalArgumentException(
144                     "defaultHostname: " + defaultHostname +
145                     " (must be matched by hostnamePattern: " + hostnamePattern + ')');
146         }
147     }
148 
149     private static boolean needsNormalization(String hostnamePattern) {
150         final int length = hostnamePattern.length();
151         for (int i = 0; i < length; i++) {
152             int c = hostnamePattern.charAt(i);
153             if (c > 0x7F) {
154                 return true;
155             }
156         }
157         return false;
158     }
159 
160     static SslContext validateSslContext(SslContext sslContext) {
161         if (sslContext != null && !sslContext.isServer()) {
162             throw new IllegalArgumentException("sslContext: " + sslContext + " (expected: server context)");
163         }
164         return sslContext;
165     }
166 
167     /**
168      * Returns the {@link Server} where this {@link VirtualHost} belongs to.
169      */
170     public Server server() {
171         if (serverConfig == null) {
172             throw new IllegalStateException("server is not configured yet.");
173         }
174         return serverConfig.server();
175     }
176 
177     void setServerConfig(ServerConfig serverConfig) {
178         if (this.serverConfig != null) {
179             throw new IllegalStateException("VirtualHost cannot be added to more than one Server.");
180         }
181 
182         this.serverConfig = requireNonNull(serverConfig, "serverConfig");
183     }
184 
185     /**
186      * Returns the default hostname of this virtual host.
187      */
188     public String defaultHostname() {
189         return defaultHostname;
190     }
191 
192     /**
193      * Returns the hostname pattern of this virtual host, as defined in
194      * <a href="http://tools.ietf.org/html/rfc2818#section-3.1">the section 3.1 of RFC2818</a>
195      */
196     public String hostnamePattern() {
197         return hostnamePattern;
198     }
199 
200     /**
201      * Returns the {@link SslContext} of this virtual host.
202      */
203     public SslContext sslContext() {
204         return sslContext;
205     }
206 
207     /**
208      * Returns the information about the {@link Service}s bound to this virtual host.
209      */
210     public List<ServiceConfig> serviceConfigs() {
211         return services;
212     }
213 
214     /**
215      * Finds the {@link Service} whose {@link PathMapping} matches the {@code path}.
216      *
217      * @param path an absolute path, as defined in <a href="https://tools.ietf.org/html/rfc3986">RFC3986</a>
218      * @param query a query, as defined in <a href="https://tools.ietf.org/html/rfc3986">RFC3986</a>.
219      *              {@code null} if query does not exist.
220      *
221      * @return the {@link ServiceConfig} wrapped by a {@link PathMapped} if there's a match.
222      *         {@link PathMapped#empty()} if there's no match.
223      */
224     public PathMapped<ServiceConfig> findServiceConfig(String path, @Nullable String query) {
225         requireNonNull(path, "path");
226         return serviceMapping.apply(path, query);
227     }
228 
229     VirtualHost decorate(@Nullable Function<Service<HttpRequest, HttpResponse>,
230                                             Service<HttpRequest, HttpResponse>> decorator) {
231         if (decorator == null) {
232             return this;
233         }
234 
235         final List<ServiceConfig> services =
236                 this.services.stream().map(cfg -> {
237                     final PathMapping pathMapping = cfg.pathMapping();
238                     final Service<HttpRequest, HttpResponse> service = decorator.apply(cfg.service());
239                     final String loggerName = cfg.loggerName().orElse(null);
240                     return new ServiceConfig(pathMapping, service, loggerName);
241                 }).collect(Collectors.toList());
242 
243         return new VirtualHost(defaultHostname(), hostnamePattern(), sslContext(), services);
244     }
245 
246     @Override
247     public String toString() {
248         String strVal = this.strVal;
249         if (strVal == null) {
250             this.strVal = strVal = toString(
251                     getClass(), defaultHostname(), hostnamePattern(), sslContext(), serviceConfigs());
252         }
253 
254         return strVal;
255     }
256 
257     static String toString(Class<?> type, String defaultHostname, String hostnamePattern,
258                            SslContext sslContext, List<?> services) {
259 
260         StringBuilder buf = new StringBuilder();
261         if (type != null) {
262             buf.append(type.getSimpleName());
263         }
264 
265         buf.append('(');
266         buf.append(defaultHostname);
267         buf.append('/');
268         buf.append(hostnamePattern);
269         buf.append(", ssl: ");
270         buf.append(sslContext != null);
271         buf.append(", services: ");
272         buf.append(services);
273         buf.append(')');
274 
275         return buf.toString();
276     }
