89      Server(ServerConfig config) {
90          this.config = requireNonNull(config, "config");
91          config.setServer(this);
92  
93          // Pre-populate the domain name mapping for later matching.
94          SslContext lastSslContext = null;
95          for (VirtualHost h: config.virtualHosts()) {
96              lastSslContext = h.sslContext();
97          }
98  
99          if (lastSslContext == null) {
100             sslContexts = null;
101             for (ServerPort p: config.ports()) {
102                 if (p.protocol().isTls()) {
103                     throw new IllegalArgumentException("no SSL context specified");
104                 }
105             }
106         } else {
107             final DomainNameMappingBuilder<SslContext>
108                     mappingBuilder = new DomainNameMappingBuilder<>(lastSslContext);
109             for (VirtualHost h : config.virtualHosts()) {
110                 final SslContext sslCtx = h.sslContext();
111                 if (sslCtx != null) {
112                     mappingBuilder.add(h.hostnamePattern(), sslCtx);
113                 }
114             }
115             sslContexts = mappingBuilder.build();
116         }
117 
118         // Invoke the serviceAdded() method in Service so that it can keep the reference to this Server or
119         // add a listener to it.
120         config.serviceConfigs().forEach(cfg -> ServiceCallbackInvoker.invokeServiceAdded(cfg, cfg.service()));
121 
122         connectionLimitingHandler = new ConnectionLimitingHandler(config.maxNumConnections());
123     }

197     /**
198      * Starts this {@link Server} to listen to the {@link ServerPort}s specified in the {@link ServerConfig}.
199      * Note that the startup procedure is asynchronous and thus this method returns immediately. To wait until
200      * this {@link Server} is fully started up, wait for the returned {@link CompletableFuture}:
201      * <pre>{@code
202      * ServerBuilder builder = new ServerBuilder();
203      * ...
204      * Server server = builder.build();
205      * server.start().get();
206      * }</pre>
207      */
208     public CompletableFuture<Void> start() {
209         final CompletableFuture<Void> future = new CompletableFuture<>();
210         start(future);
211         return future;
212     }

214     private void start(CompletableFuture<Void> future) {
215         final State state = stateManager.state();
216         switch (state.type) {
217         case STOPPING:
218             // A user called start() to restart the server, but the server is not stopped completely.
219             // Try again after stopping.
220             state.future.handle(voidFunction((ret, cause) -> start(future)))
221                         .exceptionally(CompletionActions::log);
222             return;
223         }
224 
225         if (!stateManager.enterStarting(future, this::stop0)) {
226             assert future.isCompletedExceptionally();
227             return;
228         }
229 
230         try {
231             // Initialize the event loop groups.
232             final TransportType transportType = TransportType.detectTransportType();
233             bossGroup = transportType.newEventLoopGroup(
234                     config.numBosses(),
235                     type -> new DefaultThreadFactory("armeria-server-boss-" + type.lowerCasedName(), false));
236             workerGroup = transportType.newEventLoopGroup(
237                     config.numWorkers(),
238                     type -> new DefaultThreadFactory("armeria-server-" + type.lowerCasedName(), false));
239 
240             // Initialize the server sockets asynchronously.
241             final List<ServerPort> ports = config().ports();
242             final AtomicInteger remainingPorts = new AtomicInteger(ports.size());
243             if (config().gracefulShutdownQuietPeriod().isZero()) {
244                 gracefulShutdownSupport = GracefulShutdownSupport.disabled();
245             } else {
246                 gracefulShutdownSupport =
247                         GracefulShutdownSupport.create(config().gracefulShutdownQuietPeriod(),
248                                                        config().blockingTaskExecutor());
249             }
250 
251             for (ServerPort p: ports) {
252                 start(p).addListener(new ServerPortStartListener(remainingPorts, future, p));
253             }
254         } catch (Throwable t) {
255             completeFutureExceptionally(future, t);
256         }
257     }
258 
259     private ChannelFuture start(ServerPort port) {
260         ServerBootstrap b = new ServerBootstrap();
261 
262         b.group(bossGroup, workerGroup);
263         b.channel(TransportType.detectTransportType().serverChannelClass());
264         b.handler(connectionLimitingHandler);
265         b.childHandler(new HttpServerPipelineConfigurator(config, port, sslContexts, gracefulShutdownSupport));
266 
267         return b.bind(port.localAddress());
268     }

270     /**
271      * Stops this {@link Server} to close all active {@link ServerPort}s. Note that the shutdown procedure is
272      * asynchronous and thus this method returns immediately. To wait until this {@link Server} is fully
273      * shut down, wait for the returned {@link CompletableFuture}:
274      * <pre>{@code
275      * Server server = ...;
276      * server.stop().get();
277      * }</pre>
278      */
279     public CompletableFuture<Void> stop() {
280         final CompletableFuture<Void> future = new CompletableFuture<>();
281         stop(future);
282         return future;
283     }

285     private void stop(CompletableFuture<Void> future) {
286         for (;;) {
287             final State state = stateManager.state();
288             switch (state.type) {
289             case STOPPED:
290                 completeFuture(future);
291                 return;
292             case STOPPING:
293                 state.future.handle(voidFunction((ret, cause) -> {
294                     if (cause == null) {
295                         completeFuture(future);
296                     } else {
297                         completeFutureExceptionally(future, cause);
298                     }
299                 })).exceptionally(CompletionActions::log);
300                 return;
301             case STARTED:
302                 if (!stateManager.enterStopping(state, future)) {
303                     // Someone else changed the state meanwhile; try again.
304                     continue;
305                 }
306 
307                 stop0(future);
308                 return;
309             case STARTING:
310                 // Wait until the start process is finished, and then try again.
311                 state.future.handle(voidFunction((ret, cause) -> stop(future)))
312                             .exceptionally(CompletionActions::log);
313                 return;
314             }
315         }
316     }

318     /**
319      * Returns a {@link EventLoop} from the worker group. This can be used for, e.g., scheduling background
320      * tasks for the lifetime of the {@link Server} using
321      * {@link EventLoop#scheduleAtFixedRate(Runnable, long, long, TimeUnit)}. It is very important that these
322      * tasks do not block as this would block all requests in the server on that {@link EventLoop}.
323      */
324     public EventLoop nextEventLoop() {
325         if (workerGroup == null) {
326             throw new IllegalStateException("nextEventLoop must be called after the server is started.");
327         }
328         return workerGroup.next();
329     }

331     private void stop0(CompletableFuture<Void> future) {
332         assert future != null;
333 
334         final EventLoopGroup bossGroup = this.bossGroup;
335         final GracefulShutdownSupport gracefulShutdownSupport = this.gracefulShutdownSupport;
336 
337         if (gracefulShutdownSupport == null) {
338             stop1(future, bossGroup);
339             return;
340         }
341 
342         // Check every 100 ms for the server to have completed processing requests.
343         final ScheduledFuture<?> quietPeriodFuture = bossGroup.scheduleAtFixedRate(() -> {
344             if (gracefulShutdownSupport.completedQuietPeriod()) {
345                 stop1(future, bossGroup);
346             }
347         }, 0, 100, TimeUnit.MILLISECONDS);
348 
349         // Make sure the event loop stops after the timeout, regardless of what
350         // the GracefulShutdownSupport says.
351         bossGroup.schedule(() -> {
352             quietPeriodFuture.cancel(false);
353             stop1(future, bossGroup);
354         }, config.gracefulShutdownTimeout().toMillis(), TimeUnit.MILLISECONDS);
355     }
356 
357     private void stop1(CompletableFuture<Void> future, EventLoopGroup bossGroup) {
358         // FIXME(trustin): Shutdown and terminate the blockingTaskExecutor.
359         //                 Could be fixed while fixing https://github.com/line/armeria/issues/46
360 
361         final Future<?> bossShutdownFuture;
362         if (bossGroup != null) {
363             bossShutdownFuture = bossGroup.shutdownGracefully();
364             this.bossGroup = null;
365         } else {
366             bossShutdownFuture = ImmediateEventExecutor.INSTANCE.newSucceededFuture(null);
367         }
368 
369         bossShutdownFuture.addListener(f1 -> {
370             // All server ports have been unbound.
371             primaryActivePort = null;
372             activePorts.clear();
373 
374             // Shut down the workers.
375             final EventLoopGroup workerGroup = this.workerGroup;
376             final Future<?> workerShutdownFuture;
377             if (workerGroup != null) {
378                 workerShutdownFuture = workerGroup.shutdownGracefully();
379                 this.workerGroup = null;
380             } else {
381                 workerShutdownFuture = ImmediateEventExecutor.INSTANCE.newSucceededFuture(null);
382             }
383 
384             workerShutdownFuture.addListener(f2 -> {
385                 stateManager.enter(State.STOPPED);
386                 completeFuture(future);
387             });
388         });
389     }

391     /**
392      * A shortcut to {@link #stop() stop().get()}.
393      */
394     @Override
395     public void close() {
396         final CompletableFuture<Void> f = stop();
397         boolean interrupted = false;
398         for (;;) {
399             try {
400                 f.get();
401                 break;
402             } catch (InterruptedException ignored) {
403                 interrupted = true;
404             } catch (ExecutionException e) {
405                 logger.warn("Failed to stop a server", e);
406                 break;
407             }
408         }
409 
410         if (interrupted) {
411             Thread.currentThread().interrupt();
412         }
413     }

422     enum StateType {
423         STARTING,
424         STARTED,
425         STOPPING,
426         STOPPED
427     }
428 
429     static final class State {
430         static final State STARTED = new State(StateType.STARTED, null);
431         static final State STOPPED = new State(StateType.STOPPED, null);
432 
433         final StateType type;
434         final CompletableFuture<Void> future;
435 
436         State(StateType type, CompletableFuture<Void> future) {
437             this.type = type;
438             this.future = future;
439         }
440 
441         @Override
442         public String toString() {
443             return "(" + type + ", " + future + ')';
444         }
445     }
446 
447     private final class StateManager {
448 
449         private final AtomicReference<State> ref = new AtomicReference<>(State.STOPPED);
450 
451         State state() {
452             return ref.get();
453         }
454 
455         void enter(State state) {
456             assert state.type != StateType.STARTING; // must be set via setStarting()
457             assert state.type != StateType.STOPPING; // must be set via setStopping()
458 
459             ref.set(state);
460             notifyState(state);
461         }
462 
463         boolean enterStarting(CompletableFuture<Void> future, Consumer<CompletableFuture<Void>> rollbackTask) {
464             final State startingState = new State(StateType.STARTING, future);
465             if (!ref.compareAndSet(State.STOPPED, startingState)) {
466                 completeFutureExceptionally(
467                         future,
468                         new IllegalStateException("must be stopped to start: " + this));
469                 return false;
470             }
471 
472             // Note that we added the listener only ..
473             // - after the compareAndSet() above so that the listener is invoked only when state transition
474             //   actually occurred.
475             // - before notifyState() below so that the listener is invoked when listener notification fails.
476 
477             future.handle(voidFunction((ret, cause) -> {
478                 if (cause == null) {
479                     enter(State.STARTED);
480                 } else {
481                     rollbackTask.accept(stateManager.enterStopping(new CompletableFuture<>()));
482                 }
483             })).exceptionally(CompletionActions::log);
484 
485             if (!notifyState(startingState)) {
486                 completeFutureExceptionally(
487                         future,
488                         new IllegalStateException("failed to notify all server listeners"));
489                 return false;
490             }
491 
492             return true;
493         }
494 
495         CompletableFuture<Void> enterStopping(CompletableFuture<Void> future) {
496             final State update = new State(StateType.STOPPING, future);
497             ref.set(update);
498 
499             notifyState(update);
500             return future;
501         }
502 
503         boolean enterStopping(State expect, CompletableFuture<Void> future) {
504             final State update = new State(StateType.STOPPING, future);
505             if (expect != null) {
506                 if (!ref.compareAndSet(expect, update)) {
507                     return false;
508                 }
509             } else {
510                 ref.set(update);
511             }
512 
513             notifyState(update);
514             return true;
515         }
516 
517         private boolean notifyState(State state) {
518             final AtomicBoolean success = new AtomicBoolean(true);
519             listeners.forEach(l -> {
520                 try {
521                     switch (state.type) {
522                     case STARTING:
523                         l.serverStarting(Server.this);
524                         break;
525                     case STARTED:
526                         l.serverStarted(Server.this);
527                         break;
528                     case STOPPING:
529                         l.serverStopping(Server.this);
530                         break;
531                     case STOPPED:
532                         l.serverStopped(Server.this);
533                         break;
534                     default:
535                         throw new Error("unknown state type " + state.type);
536                     }
537                 } catch (Throwable t) {
538                     success.set(false);
539                     logger.warn("Failed to notify a server listener: {}", l, t);
540                 }
541             });
542 
543             return success.get();
544         }
545 
546         @Override
547         public String toString() {
548             return ref.toString();
549         }
550     }

552     private final class ServerPortStartListener implements ChannelFutureListener {
553 
554         private final AtomicInteger remainingPorts;
555         private final CompletableFuture<Void> startFuture;
556         private final ServerPort port;
557 
558         ServerPortStartListener(
559                 AtomicInteger remainingPorts, CompletableFuture<Void> startFuture, ServerPort port) {
560 
561             this.remainingPorts = requireNonNull(remainingPorts, "remainingPorts");
562             this.startFuture = requireNonNull(startFuture, "startFuture");
563             this.port = requireNonNull(port, "port");
564         }
565 
566         @Override
567         public void operationComplete(ChannelFuture f) throws Exception {
568             if (startFuture.isDone()) {
569                 return;
570             }
571 
572             if (f.isSuccess()) {
573                 InetSocketAddress localAddress = (InetSocketAddress) f.channel().localAddress();
574                 ServerPort actualPort = new ServerPort(localAddress, port.protocol());
575 
576                 activePorts.put(localAddress, actualPort);
577 
578                 // The port that has been activated first becomes the primary port.
579                 if (primaryActivePort == null) {
580                     primaryActivePort = actualPort;
581                 }
582 
583                 if (remainingPorts.decrementAndGet() == 0) {
584                     completeFuture(startFuture);
585                 }
586             } else {
587                 completeFutureExceptionally(startFuture, f.cause());
588             }
589         }
590     }
591 
592     private static void completeFuture(CompletableFuture<Void> future) {
593         if (GlobalEventExecutor.INSTANCE.inEventLoop()) {
594             future.complete(null);
595         } else {
596             GlobalEventExecutor.INSTANCE.execute(() -> future.complete(null));
597         }
598     }
599 
600     private static void completeFutureExceptionally(CompletableFuture<Void> future, Throwable cause) {
601         if (GlobalEventExecutor.INSTANCE.inEventLoop()) {
602             future.completeExceptionally(cause);
603         } else {
604             GlobalEventExecutor.INSTANCE.execute(() -> future.completeExceptionally(cause));
605         }
606     }
