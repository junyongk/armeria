17  package com.linecorp.armeria.server;
18  
19  import com.linecorp.armeria.common.SessionProtocol;
20  
21  import io.netty.channel.Channel;
22  import io.netty.channel.ChannelHandler;
23  import io.netty.channel.ChannelHandlerContext;
24  import io.netty.channel.ChannelPipeline;
25  
26  interface HttpServer {
27  
28      static HttpServer get(Channel channel) {
29          final ChannelPipeline p = channel.pipeline();
30          final ChannelHandler lastHandler = p.last();
31          if (lastHandler instanceof HttpServer) {
32              return (HttpServer) lastHandler;
33          }
34  
35          for (ChannelHandler h : p.toMap().values()) {
36              if (h instanceof HttpServer) {
37                  return (HttpServer) h;
38              }
39          }
40  
41          return null;
42      }
43  
44      static HttpServer get(ChannelHandlerContext ctx) {
45          return get(ctx.channel());
46      }
47  
48      SessionProtocol protocol();
49  
50      int unfinishedRequests();
51  }
