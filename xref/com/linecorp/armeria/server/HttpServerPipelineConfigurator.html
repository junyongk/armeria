59  /**
60   * Configures Netty {@link ChannelPipeline} to serve HTTP/1 and 2 requests.
61   */
62  final class HttpServerPipelineConfigurator extends ChannelInitializer<Channel> {
63  
64      private static final AsciiString SCHEME_HTTP = AsciiString.of("http");
65      private static final AsciiString SCHEME_HTTPS = AsciiString.of("https");
66  
67      private static final int UPGRADE_REQUEST_MAX_LENGTH = 16384;
68  
69      private final ServerConfig config;
70      private final ServerPort port;
71      private final DomainNameMapping<SslContext> sslContexts;
72      private final GracefulShutdownSupport gracefulShutdownSupport;
73  
74      /**
75       * Creates a new instance.
76       */
77      HttpServerPipelineConfigurator(
78              ServerConfig config, ServerPort port,
79              DomainNameMapping<SslContext> sslContexts,
80              GracefulShutdownSupport gracefulShutdownSupport) {
81  
82          this.config = requireNonNull(config, "config");
83          this.port = requireNonNull(port, "port");
84          this.sslContexts = sslContexts;
85          this.gracefulShutdownSupport = requireNonNull(gracefulShutdownSupport, "gracefulShutdownSupport");
86      }
87  
88      @Override
89      protected void initChannel(Channel ch) throws Exception {
90          final ChannelPipeline p = ch.pipeline();
91          p.addLast(new FlushConsolidationHandler());
92          p.addLast(ReadSuppressingHandler.INSTANCE);
93  
94          if (port.protocol().isTls()) {
95              p.addLast(new SniHandler(sslContexts));
96              p.addLast(TrafficLoggingHandler.SERVER);
97              configureHttps(p);
98          } else {
99              p.addLast(TrafficLoggingHandler.SERVER);
100             configureHttp(p);
101         }
102     }
103 
104     private void configureHttp(ChannelPipeline p) {
105         p.addLast(new Http2PrefaceOrHttpHandler());
106         configureRequestCountingHandlers(p);
107         p.addLast(new HttpServerHandler(config, gracefulShutdownSupport, SessionProtocol.H1C));
108     }
109 
110     private void configureRequestCountingHandlers(ChannelPipeline p) {
111         if (config.idleTimeoutMillis() > 0) {
112             p.addFirst(new HttpServerIdleTimeoutHandler(config.idleTimeoutMillis()));
113         }
114     }
115 
116     private void configureHttps(ChannelPipeline p) {
117         p.addLast(new Http2OrHttpHandler());
118     }
119 
120     private Http2ConnectionHandler newHttp2ConnectionHandler(ChannelPipeline pipeline) {
121 
122         final Http2Connection conn = new DefaultHttp2Connection(true);
123         conn.addListener(new Http2GoAwayListener(pipeline.channel()));
124 
125         Http2FrameReader reader = new DefaultHttp2FrameReader(true);
126         Http2FrameWriter writer = new DefaultHttp2FrameWriter();
127 
128         Http2ConnectionEncoder encoder = new DefaultHttp2ConnectionEncoder(conn, writer);
129         Http2ConnectionDecoder decoder = new DefaultHttp2ConnectionDecoder(conn, encoder, reader);
130 
131         final Http2ConnectionHandler handler =
132                 new Http2ServerConnectionHandler(decoder, encoder, new Http2Settings());
133 
134         // Setup post build options
135         final Http2RequestDecoder listener =
136                 new Http2RequestDecoder(config, pipeline.channel(), handler.encoder());
137 
138         handler.connection().addListener(listener);
139         handler.decoder().frameListener(listener);
140         handler.gracefulShutdownTimeoutMillis(config.idleTimeoutMillis());
141 
142         return handler;
143     }
144 
145     private final class Http2OrHttpHandler extends ApplicationProtocolNegotiationHandler {
146 
147         Http2OrHttpHandler() {
148             super(ApplicationProtocolNames.HTTP_1_1);
149         }
150 
151         @Override
152         protected void configurePipeline(ChannelHandlerContext ctx, String protocol) throws Exception {
153             if (ApplicationProtocolNames.HTTP_2.equals(protocol)) {
154                 addHttp2Handlers(ctx);
155                 return;
156             }
157 
158             if (ApplicationProtocolNames.HTTP_1_1.equals(protocol)) {
159                 addHttpHandlers(ctx);
160                 return;
161             }
162 
163             throw new IllegalStateException("unknown protocol: " + protocol);
164         }
165 
166 
167         private void addHttp2Handlers(ChannelHandlerContext ctx) {
168             final ChannelPipeline p = ctx.pipeline();
169             p.addLast(newHttp2ConnectionHandler(p));
170             configureRequestCountingHandlers(p);
171             p.addLast(new HttpServerHandler(config, gracefulShutdownSupport, SessionProtocol.H2));
172         }
173 
174         private void addHttpHandlers(ChannelHandlerContext ctx) {
175             final ChannelPipeline p = ctx.pipeline();
176             p.addLast(new HttpServerCodec());
177             p.addLast(new Http1RequestDecoder(config, ctx.channel(), SCHEME_HTTPS));
178             configureRequestCountingHandlers(p);
179             p.addLast(new HttpServerHandler(config, gracefulShutdownSupport, SessionProtocol.H1));
180         }
181     }
182 
183     private final class Http2PrefaceOrHttpHandler extends ByteToMessageDecoder {
184 
185         private String name;
186 
187         @Override
188         public void handlerAdded(ChannelHandlerContext ctx) throws Exception {
189             super.handlerAdded(ctx);
190             name = ctx.name();
191         }
192 
193         @Override
194         protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
195             if (in.readableBytes() < 4) {
196                 return;
197             }
198 
199             if (in.getInt(in.readerIndex()) == 0x50524920) { // If starts with 'PRI '
200                 // Probably HTTP/2; received the HTTP/2 preface string.
201                 configureHttp2(ctx);
202             } else {
203                 // Probably HTTP/1; the client can still upgrade using the traditional HTTP/1 upgrade request.
204                 configureHttp1WithUpgrade(ctx);
205             }
206 
207             ctx.pipeline().remove(this);
208         }
209 
210         private void configureHttp1WithUpgrade(ChannelHandlerContext ctx) {
211             final ChannelPipeline p = ctx.pipeline();
212             final HttpServerCodec http1codec = new HttpServerCodec();
213 
214             String baseName = name;
215             baseName = addAfter(p, baseName, http1codec);
216             baseName = addAfter(p, baseName, new HttpServerUpgradeHandler(
217                     http1codec,
218                     protocol -> {
219                         if (!AsciiString.contentEquals(Http2CodecUtil.HTTP_UPGRADE_PROTOCOL_NAME, protocol)) {
220                             return null;
221                         }
222 
223                         return new Http2ServerUpgradeCodec(
224                                 newHttp2ConnectionHandler(p));
225                     },
226                     UPGRADE_REQUEST_MAX_LENGTH));
227 
228             addAfter(p, baseName, new Http1RequestDecoder(config, ctx.channel(), SCHEME_HTTP));
229         }
230 
231         private void configureHttp2(ChannelHandlerContext ctx) {
232             final ChannelPipeline p = ctx.pipeline();
233             addAfter(p, name, newHttp2ConnectionHandler(p));
234         }
235 
236         private String addAfter(ChannelPipeline p, String baseName, ChannelHandler handler) {
237             p.addAfter(baseName, null, handler);
238             return p.context(handler).name();
239         }
240     }
