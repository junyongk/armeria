86      static final TomcatHandler TOMCAT_HANDLER;
87  
88      static {
89          final String prefix = TomcatService.class.getPackage().getName() + '.';
90          final ClassLoader classLoader = TomcatService.class.getClassLoader();
91          final Class<?> handlerClass;
92          try {
93              if (TomcatVersion.major() < 8 || TomcatVersion.major() == 8 && TomcatVersion.minor() < 5) {
94                  handlerClass = Class.forName(prefix + "Tomcat80Handler", true, classLoader);
95              } else {
96                  handlerClass = Class.forName(prefix + "Tomcat85Handler", true, classLoader);
97              }
98              TOMCAT_HANDLER = (TomcatHandler) handlerClass.getDeclaredConstructor().newInstance();
99          } catch (ReflectiveOperationException e) {
100             throw new IllegalStateException(
101                     "could not find the matching classes for Tomcat version " + ServerInfo.getServerNumber() +
102                     "; using a wrong armeria-tomcat JAR?", e);
103         }
104     }

156     /**
157      * Creates a new {@link TomcatService} from an existing {@link Tomcat} instance.
158      * If the specified {@link Tomcat} instance is not configured properly, the returned {@link TomcatService}
159      * may respond with '503 Service Not Available' error.
160      */
161     public static TomcatService forTomcat(Tomcat tomcat) {
162         requireNonNull(tomcat, "tomcat");
163 
164         final String hostname = tomcat.getEngine().getDefaultHost();
165         if (hostname == null) {
166             throw new IllegalArgumentException("default hostname not configured: " + tomcat);
167         }
168 
169         final Connector connector = tomcat.getConnector();
170         if (connector == null) {
171             throw new IllegalArgumentException("connector not configured: " + tomcat);
172         }
173 
174         return forConnector(hostname, connector);
175     }

199     static TomcatService forConfig(TomcatServiceConfig config) {
200         final Consumer<Connector> postStopTask = connector -> {
201             final org.apache.catalina.Server server = connector.getService().getServer();
202             if (server.getState() == LifecycleState.STOPPED) {
203                 try {
204                     logger.info("Destroying an embedded Tomcat: {}", toString(server));
205                     server.destroy();
206                 } catch (Exception e) {
207                     logger.warn("Failed to destroy an embedded Tomcat: {}", toString(server), e);
208                 }
209             }
210         };
211 
212         return new TomcatService(null, new ManagedConnectorFactory(config), postStopTask);
213     }

215     static String toString(org.apache.catalina.Server server) {
216         requireNonNull(server, "server");
217 
218         final Service[] services = server.findServices();
219         final String serviceName;
220         if (services.length == 0) {
221             serviceName = "<unknown>";
222         } else {
223             serviceName = services[0].getName();
224         }
225 
226         final StringBuilder buf = new StringBuilder(128);
227 
228         buf.append("(serviceName: ");
229         buf.append(serviceName);
230         if (TomcatVersion.major() >= 8) {
231             buf.append(", catalinaBase: " + server.getCatalinaBase());
232         }
233         buf.append(')');
234 
235         return buf.toString();
236     }

262     @Override
263     public void serviceAdded(ServiceConfig cfg) throws Exception {
264         if (hostname == null) {
265             hostname = cfg.server().defaultHostname();
266         }
267 
268         if (armeriaServer != null) {
269             if (armeriaServer != cfg.server()) {
270                 throw new IllegalStateException("cannot be added to more than one server");
271             } else {
272                 return;
273             }
274         }
275 
276         armeriaServer = cfg.server();
277         armeriaServer.addListener(configurator);
278     }

292     void start() throws Exception {
293         started = false;
294         connector = connectorFactory.apply(hostname);
295         final Service service = connector.getService();
296         if (service == null) {
297             return;
298         }
299 
300         final Engine engine = TomcatUtil.engine(service);
301         if (engine == null) {
302             return;
303         }
304 
305         final String engineName = engine.getName();
306         if (engineName == null) {
307             return;
308         }
309 
310         if (activeEngines.contains(engineName)) {
311             throw new TomcatServiceException("duplicate engine name: " + engineName);
312         }
313 
314         server = service.getServer();
315 
316         if (!TOMCAT_START_STATES.contains(server.getState())) {
317             logger.info("Starting an embedded Tomcat: {}", toString(server));
318             server.start();
319             started = true;
320         }
321 
322         activeEngines.add(engineName);
323         this.engineName = engineName;
324     }
325 
326     void stop() throws Exception {
327         final org.apache.catalina.Server server = this.server;
328         final Connector connector = this.connector;
329         this.server = null;
330         this.connector = null;
331 
332         if (engineName != null) {
333             activeEngines.remove(engineName);
334             engineName = null;
335         }
336 
337         if (server == null || !started) {
338             return;
339         }
340 
341         try {
342             logger.info("Stopping an embedded Tomcat: {}", toString(server));
343             server.stop();
344         } catch (Exception e) {
345             logger.warn("Failed to stop an embedded Tomcat: {}", toString(server), e);
346         }
347 
348         postStopTask.accept(connector);
349     }

351     @Override
352     public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
353         final Adapter coyoteAdapter = connector().getProtocolHandler().getAdapter();
354         if (coyoteAdapter == null) {
355             // Tomcat is not configured / stopped.
356             throw ServiceUnavailableException.get();
357         }
358 
359         final DefaultHttpResponse res = new DefaultHttpResponse();
360         req.aggregate().handle(voidFunction((aReq, cause) -> {
361             if (cause != null) {
362                 logger.warn("{} Failed to aggregate a request:", ctx, cause);
363                 res.respond(HttpStatus.INTERNAL_SERVER_ERROR);
364                 return;
365             }
366 
367             try {
368                 final Request coyoteReq = convertRequest(ctx, aReq);
369                 if (coyoteReq == null) {
370                     res.respond(HttpStatus.BAD_REQUEST);
371                     return;
372                 }
373                 final Response coyoteRes = new Response();
374                 coyoteReq.setResponse(coyoteRes);
375                 coyoteRes.setRequest(coyoteReq);
376 
377                 final Queue<HttpData> data = new ArrayDeque<>();
378                 coyoteRes.setOutputBuffer(TOMCAT_HANDLER.outputBuffer(data));
379 
380                 ctx.blockingTaskExecutor().execute(() -> {
381                     if (!res.isOpen()) {
382                         return;
383                     }
384 
385                     try {
386                         coyoteAdapter.service(coyoteReq, coyoteRes);
387                         final HttpHeaders headers = convertResponse(coyoteRes);
388                         res.write(headers);
389                         for (;;) {
390                             final HttpData d = data.poll();
391                             if (d == null) {
392                                 break;
393                             }
394 
395                             res.write(d);
396                         }
397                         res.close();
398                     } catch (Throwable t) {
399                         logger.warn("{} Failed to produce a response:", ctx, t);
400                         res.close();
401                     }
402                 });
403             } catch (Throwable t) {
404                 logger.warn("{} Failed to invoke Tomcat:", ctx, t);
405                 res.close();
406             }
407         })).exceptionally(CompletionActions::log);
408 
409         return res;
410     }

412     @Nullable
413     private Request convertRequest(ServiceRequestContext ctx, AggregatedHttpMessage req) {
414         final String mappedPath = ctx.mappedPath();
415         final Request coyoteReq = new Request();
416 
417         coyoteReq.scheme().setString(req.scheme());
418 
419         // Set the remote host/address.
420         final InetSocketAddress remoteAddr = ctx.remoteAddress();
421         coyoteReq.remoteAddr().setString(remoteAddr.getAddress().getHostAddress());
422         coyoteReq.remoteHost().setString(remoteAddr.getHostString());
423         coyoteReq.setRemotePort(remoteAddr.getPort());
424 
425         // Set the local host/address.
426         final InetSocketAddress localAddr = ctx.localAddress();
427         coyoteReq.localAddr().setString(localAddr.getAddress().getHostAddress());
428         coyoteReq.localName().setString(hostname);
429         coyoteReq.setLocalPort(localAddr.getPort());
430 
431         final String hostHeader = req.headers().authority();
432         int colonPos = hostHeader.indexOf(':');
433         if (colonPos < 0) {
434             coyoteReq.serverName().setString(hostHeader);
435         } else {
436             coyoteReq.serverName().setString(hostHeader.substring(0, colonPos));
437             try {
438                 int port = Integer.parseInt(hostHeader.substring(colonPos + 1));
439                 coyoteReq.setServerPort(port);
440             } catch (NumberFormatException e) {
441                 // Invalid port number
442                 return null;
443             }
444         }
445 
446         // Set the method.
447         final HttpMethod method = req.method();
448         coyoteReq.method().setString(method.name());
449 
450         // Set the request URI.
451         final byte[] uriBytes = mappedPath.getBytes(StandardCharsets.US_ASCII);
452         coyoteReq.requestURI().setBytes(uriBytes, 0, uriBytes.length);
453 
454         // Set the query string if any.
455         if (ctx.query() != null) {
456             coyoteReq.queryString().setString(ctx.query());
457         }
458 
459         // Set the headers.
460         final MimeHeaders cHeaders = coyoteReq.getMimeHeaders();
461         convertHeaders(req.headers(), cHeaders);
462         convertHeaders(req.trailingHeaders(), cHeaders);
463 
464         // Set the content.
465         final HttpData content = req.content();
466         coyoteReq.setInputBuffer(TOMCAT_HANDLER.inputBuffer(content));
467 
468         return coyoteReq;
469     }

471     private static void convertHeaders(HttpHeaders headers, MimeHeaders cHeaders) {
472         if (headers.isEmpty()) {
473             return;
474         }
475 
476         for (Entry<AsciiString, String> e : headers) {
477             final AsciiString k = e.getKey();
478             final String v = e.getValue();
479 
480             if (k.isEmpty() || k.byteAt(0) == ':') {
481                 continue;
482             }
483 
484             final MessageBytes cValue = cHeaders.addValue(k.array(), k.arrayOffset(), k.length());
485             final byte[] valueBytes = v.getBytes(StandardCharsets.US_ASCII);
486             cValue.setBytes(valueBytes, 0, valueBytes.length);
487         }
488     }

490     private static HttpHeaders convertResponse(Response coyoteRes) {
491         final HttpHeaders headers = HttpHeaders.of(HttpStatus.valueOf(coyoteRes.getStatus()));
492 
493         final String contentType = coyoteRes.getContentType();
494         if (contentType != null && !contentType.isEmpty()) {
495             headers.set(HttpHeaderNames.CONTENT_TYPE, contentType);
496         }
497 
498         final MimeHeaders cHeaders = coyoteRes.getMimeHeaders();
499         final int numHeaders = cHeaders.size();
500         for (int i = 0; i < numHeaders; i++) {
501             final AsciiString name = toHeaderName(cHeaders.getName(i));
502             if (name == null) {
503                 continue;
504             }
505 
506             final String value = toHeaderValue(cHeaders.getValue(i));
507             if (value == null) {
508                 continue;
509             }
510 
511             headers.add(name.toLowerCase(), value);
512         }
513 
514         return headers;
515     }

517     private static AsciiString toHeaderName(MessageBytes value) {
518         switch (value.getType()) {
519             case MessageBytes.T_BYTES: {
520                 final ByteChunk chunk = value.getByteChunk();
521                 return new AsciiString(chunk.getBuffer(), chunk.getOffset(), chunk.getLength(), true);
522             }
523             case MessageBytes.T_CHARS: {
524                 final CharChunk chunk = value.getCharChunk();
525                 return new AsciiString(chunk.getBuffer(), chunk.getOffset(), chunk.getLength());
526             }
527             case MessageBytes.T_STR: {
528                 return new AsciiString(value.getString());
529             }
530         }
531         return null;
532     }
533 
534     private static String toHeaderValue(MessageBytes value) {
535         switch (value.getType()) {
536             case MessageBytes.T_BYTES: {
537                 final ByteChunk chunk = value.getByteChunk();
538                 return new String(chunk.getBuffer(), chunk.getOffset(), chunk.getLength(),
539                                   StandardCharsets.US_ASCII);
540             }
541             case MessageBytes.T_CHARS: {
542                 final CharChunk chunk = value.getCharChunk();
543                 return new String(chunk.getBuffer(), chunk.getOffset(), chunk.getLength());
544             }
545             case MessageBytes.T_STR: {
546                 return value.getString();
547             }
548         }
549         return null;
550     }
