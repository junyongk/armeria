17  package com.linecorp.armeria.server.thrift;
18  
19  import static com.google.common.collect.ImmutableMap.toImmutableMap;
20  import static java.util.Objects.requireNonNull;
21  
22  import java.util.List;
23  import java.util.Map;
24  
25  import org.apache.thrift.AsyncProcessFunction;
26  import org.apache.thrift.ProcessFunction;
27  import org.apache.thrift.TApplicationException;
28  import org.apache.thrift.TBase;
29  import org.apache.thrift.TException;
30  import org.apache.thrift.async.AsyncMethodCallback;
31  
32  import com.google.common.collect.ImmutableMap;
33  
34  import com.linecorp.armeria.common.DefaultRpcResponse;
35  import com.linecorp.armeria.common.RpcRequest;
36  import com.linecorp.armeria.common.RpcResponse;
37  import com.linecorp.armeria.internal.thrift.ThriftFunction;
38  import com.linecorp.armeria.server.Service;
39  import com.linecorp.armeria.server.ServiceRequestContext;
40  
41  /**
42   * A {@link Service} that handles a Thrift {@link RpcRequest}.
43   *
44   * @see THttpService
45   */
46  public final class ThriftCallService implements Service<RpcRequest, RpcResponse> {
47  
48      /**
49       * Creates a new {@link ThriftCallService} with the specified service implementation.
50       *
51       * @param implementation an implementation of {@code *.Iface} or {@code *.AsyncIface} service interface
52       *                       generated by the Apache Thrift compiler
53       */
54      public static ThriftCallService of(Object implementation) {
55          return new ThriftCallService(ImmutableMap.of("", implementation));
56      }
57  
58      /**
59       * Creates a new multiplexed {@link ThriftCallService} with the specified service implementations.
60       *
61       * @param implementations a {@link Map} whose key is service name and value is the implementation of
62       *                        {@code *.Iface} or {@code *.AsyncIface} service interface generated by
63       *                        the Apache Thrift compiler
64       */
65      public static ThriftCallService of(Map<String, ?> implementations) {
66          return new ThriftCallService(implementations);
67      }
68  
69      private final Map<String, ThriftServiceEntry> entries;
70  
71      private ThriftCallService(Map<String, ?> implementations) {
72          requireNonNull(implementations, "implementations");
73          if (implementations.isEmpty()) {
74              throw new IllegalArgumentException("empty implementations");
75          }
76  
77          entries = implementations.entrySet().stream().collect(
78                  toImmutableMap(Map.Entry::getKey, ThriftServiceEntry::new));
79      }
80  
81      /**
82       * Returns the information about the Thrift services being served.
83       *
84       * @return a {@link Map} whose key is a service name, which could be an empty string if this service
85       *         is not multiplexed
86       */
87      public Map<String, ThriftServiceEntry> entries() {
88          return entries;
89      }
90  
91      @Override
92      public RpcResponse serve(ServiceRequestContext ctx, RpcRequest call) throws Exception {
93          final int colonPos = call.method().indexOf(':');
94          final String method;
95          final String serviceName;
96          if (colonPos < 0) {
97              serviceName = "";
98              method = call.method();
99          } else {
100             serviceName = call.method().substring(0, colonPos);
101             method = call.method().substring(colonPos + 1);
102         }
103 
104         // Ensure that such a service exists.
105         final ThriftServiceEntry e = entries.get(serviceName);
106         if (e != null) {
107             // Ensure that such a method exists.
108             final ThriftFunction f = e.metadata.function(method);
109             if (f != null) {
110                 final DefaultRpcResponse reply = new DefaultRpcResponse();
111                 invoke(ctx, e.implementation, f, call.params(), reply);
112                 return reply;
113             }
114         }
115 
116         return new DefaultRpcResponse(new TApplicationException(
117                 TApplicationException.UNKNOWN_METHOD, "unknown method: " + call.method()));
118     }
119 
120     private static void invoke(
121             ServiceRequestContext ctx,
122             Object impl, ThriftFunction func, List<Object> args, DefaultRpcResponse reply) {
123 
124         try {
125             final TBase<?, ?> tArgs = func.newArgs(args);
126             if (func.isAsync()) {
127                 invokeAsynchronously(impl, func, tArgs, reply);
128             } else {
129                 invokeSynchronously(ctx, impl, func, tArgs, reply);
130             }
131         } catch (Throwable t) {
132             reply.completeExceptionally(t);
133         }
134     }
135 
136     private static void invokeAsynchronously(
137             Object impl, ThriftFunction func, TBase<?, ?> args, DefaultRpcResponse reply) throws TException {
138 
139         final AsyncProcessFunction<Object, TBase<?, ?>, Object> f = func.asyncFunc();
140         f.start(impl, args, new AsyncMethodCallback<Object>() {
141             @Override
142             public void onComplete(Object response) {
143                 if (func.isOneWay()) {
144                     reply.complete(null);
145                 } else {
146                     reply.complete(response);
147                 }
148             }
149 
150             @Override
151             public void onError(Exception e) {
152                 reply.completeExceptionally(e);
153             }
154         });
155     }
156 
157     private static void invokeSynchronously(
158             ServiceRequestContext ctx, Object impl,
159             ThriftFunction func, TBase<?, ?> args, DefaultRpcResponse reply) {
160 
161         final ProcessFunction<Object, TBase<?, ?>> f = func.syncFunc();
162         ctx.blockingTaskExecutor().execute(() -> {
163             if (reply.isDone()) {
164                 // Closed already most likely due to timeout.
165                 return;
166             }
167 
168             try {
169                 final TBase<?, ?> result = f.getResult(impl, args);
170                 if (func.isOneWay()) {
171                     reply.complete(null);
172                 } else {
173                     reply.complete(func.getResult(result));
174                 }
175             } catch (Throwable t) {
176                 reply.completeExceptionally(t);
177             }
178         });
179     }
180 }
