100     @Override
101     public ServiceSpecification generateSpecification(Set<ServiceConfig> serviceConfigs) {
102 
103         final Map<Class<?>, EntryBuilder> map = new LinkedHashMap<>();
104 
105         for (ServiceConfig c : serviceConfigs) {
106             final THttpService service = c.service().as(THttpService.class).get();
107             service.entries().forEach((serviceName, entry) -> {
108                 for (Class<?> iface : entry.interfaces()) {
109                     final Class<?> serviceClass = iface.getEnclosingClass();
110                     final EntryBuilder builder =
111                             map.computeIfAbsent(serviceClass, cls -> new EntryBuilder(serviceClass));
112 
113                     // Add all available endpoints.
114                     c.pathMapping().exactPath().ifPresent(
115                             p -> builder.endpoint(new EndpointInfo(
116                                     c.virtualHost().hostnamePattern(),
117                                     p, serviceName,
118                                     service.defaultSerializationFormat(),
119                                     service.allowedSerializationFormats())));
120                 }
121             });
122         }
123 
124         final List<Entry> entries = map.values().stream()
125                                        .map(EntryBuilder::build)
126                                        .collect(Collectors.toList());
127         return generate(entries);
128     }

130     @VisibleForTesting
131     static ServiceSpecification generate(List<Entry> entries) {
132         final List<ServiceInfo> services = entries.stream()
133                                                   .map(e -> newServiceInfo(e.serviceType, e.endpointInfos))
134                                                   .collect(toImmutableList());
135 
136         return ServiceSpecification.generate(services, ThriftDocServicePlugin::newNamedTypeInfo);
137     }

139     @VisibleForTesting
140     static ServiceInfo newServiceInfo(Class<?> serviceClass, Iterable<EndpointInfo> endpoints) {
141         requireNonNull(serviceClass, "serviceClass");
142 
143         final String name = serviceClass.getName();
144         final ClassLoader serviceClassLoader = serviceClass.getClassLoader();
145         final String interfaceClassName = name + "$Iface";
146         final Class<?> interfaceClass;
147         try {
148             interfaceClass = Class.forName(interfaceClassName, false, serviceClassLoader);
149         } catch (ClassNotFoundException e) {
150             throw new IllegalStateException("failed to find a class: " + interfaceClassName, e);
151         }
152         final Method[] methods = interfaceClass.getDeclaredMethods();
153 
154         return new ServiceInfo(name, Arrays.stream(methods).map(m -> newMethodInfo(m, endpoints))::iterator);
155     }

157     private static MethodInfo newMethodInfo(Method method, Iterable<EndpointInfo> endpoints) {
158         requireNonNull(method, "method");
159 
160         final String methodName = method.getName();
161 
162         final Class<?> serviceClass = method.getDeclaringClass().getDeclaringClass();
163         final String serviceName = serviceClass.getName();
164         final ClassLoader classLoader = serviceClass.getClassLoader();
165 
166         final String argsClassName = serviceName + '$' + methodName + "_args";
167         final Class<? extends TBase<?, ?>> argsClass;
168         try {
169             @SuppressWarnings("unchecked")
170             final Class<? extends TBase<?, ?>> argsClass0 =
171                     (Class<? extends TBase<?, ?>>) Class.forName(argsClassName, false, classLoader);
172             argsClass = argsClass0;
173         } catch (ClassNotFoundException e) {
174             throw new IllegalStateException("failed to find a class: " + argsClassName, e);
175         }
176 
177         Class<?> resultClass;
178         try {
179             resultClass =  Class.forName(serviceName + '$' + methodName + "_result", false, classLoader);
180         } catch (ClassNotFoundException ignored) {
181             // Oneway function does not have a result type.
182             resultClass = null;
183         }
184 
185         @SuppressWarnings("unchecked")
186         final MethodInfo methodInfo =
187                 newMethodInfo(methodName, argsClass,
188                               (Class<? extends TBase<?, ?>>) resultClass,
189                               (Class<? extends TException>[]) method.getExceptionTypes(),
190                               endpoints);
191         return methodInfo;
192     }

195     private static MethodInfo newMethodInfo(String name,
196                                             Class<? extends TBase<?, ?>> argsClass,
197                                             @Nullable Class<? extends TBase<?, ?>> resultClass,
198                                             Class<? extends TException>[] exceptionClasses,
199                                             Iterable<EndpointInfo> endpoints) {
200         requireNonNull(name, "name");
201         requireNonNull(argsClass, "argsClass");
202         requireNonNull(exceptionClasses, "exceptionClasses");
203         requireNonNull(endpoints, "endpoints");
204 
205         final List<FieldInfo> parameters =
206                 FieldMetaData.getStructMetaDataMap(argsClass).values().stream()
207                              .map(fieldMetaData -> newFieldInfo(argsClass, fieldMetaData))
208                              .collect(toImmutableList());
209 
210         // Find the 'success' field.
211         FieldInfo fieldInfo = null;
212         if (resultClass != null) { // Function isn't "oneway" function
213             final Map<? extends TFieldIdEnum, FieldMetaData> resultMetaData =
214                     FieldMetaData.getStructMetaDataMap(resultClass);
215 
216             for (FieldMetaData fieldMetaData : resultMetaData.values()) {
217                 if ("success".equals(fieldMetaData.fieldName)) {
218                     fieldInfo = newFieldInfo(resultClass, fieldMetaData);
219                     break;
220                 }
221             }
222         }
223 
224         final TypeSignature returnTypeSignature;
225         if (fieldInfo == null) {
226             returnTypeSignature = VOID;
227         } else {
228             returnTypeSignature = fieldInfo.typeSignature();
229         }
230 
231         final List<TypeSignature> exceptionTypeSignatures =
232                 Arrays.stream(exceptionClasses)
233                       .filter(e -> e != TException.class)
234                       .map(TypeSignature::ofNamed)
235                       .collect(toImmutableList());
236 
237         return new MethodInfo(name, returnTypeSignature, parameters, exceptionTypeSignatures, endpoints);
238     }

240     private static NamedTypeInfo newNamedTypeInfo(TypeSignature typeSignature) {
241         Class<?> type = (Class<?>) typeSignature.namedTypeDescriptor().get();
242         if (type.isEnum()) {
243             return newEnumInfo(type);
244         }
245 
246         if (TException.class.isAssignableFrom(type)) {
247             @SuppressWarnings("unchecked")
248             final Class<? extends TException> castType = (Class<? extends TException>) type;
249             return newExceptionInfo(castType);
250         }
251 
252         assert TBase.class.isAssignableFrom(type);
253         @SuppressWarnings("unchecked")
254         final Class<? extends TBase<?, ?>> castType = (Class<? extends TBase<?, ?>>) type;
255         return newStructInfo(castType);
256     }

258     @VisibleForTesting
259     static EnumInfo newEnumInfo(Class<?> enumClass) {
260         requireNonNull(enumClass, "enumClass");
261 
262         final List<EnumValueInfo> values = new ArrayList<>();
263         final Field[] fields = enumClass.getDeclaredFields();
264         for (Field field : fields) {
265             if (field.isEnumConstant()) {
266                 try {
267                     values.add(new EnumValueInfo(String.valueOf(field.get(null))));
268                 } catch (IllegalAccessException ignored) {
269                     // Skip inaccessible fields.
270                 }
271             }
272         }
273 
274         final String name = enumClass.getName();
275         return new EnumInfo(name, values);
276     }

278     @VisibleForTesting
279     static StructInfo newStructInfo(Class<? extends TBase<?, ?>> structClass) {
280         final String name = structClass.getName();
281 
282         final Map<?, FieldMetaData> metaDataMap = FieldMetaData.getStructMetaDataMap(structClass);
283         final List<FieldInfo> fields =
284                 metaDataMap.values().stream()
285                            .map(fieldMetaData -> newFieldInfo(structClass, fieldMetaData))
286                            .collect(Collectors.toList());
287 
288         return new StructInfo(name, fields);
289     }

291     @VisibleForTesting
292     static ExceptionInfo newExceptionInfo(Class<? extends TException> exceptionClass) {
293         requireNonNull(exceptionClass, "exceptionClass");
294         final String name = exceptionClass.getName();
295 
296         List<FieldInfo> fields;
297         try {
298             @SuppressWarnings("unchecked")
299             final Map<?, FieldMetaData> metaDataMap =
300                     (Map<?, FieldMetaData>) exceptionClass.getDeclaredField("metaDataMap").get(null);
301 
302             fields = metaDataMap.values().stream()
303                                 .map(fieldMetaData -> newFieldInfo(exceptionClass, fieldMetaData))
304                                 .collect(toImmutableList());
305         } catch (IllegalAccessException e) {
306             throw new AssertionError("will not happen", e);
307         } catch (NoSuchFieldException ignored) {
308             fields = Collections.emptyList();
309         }
310 
311         return new ExceptionInfo(name, fields);
312     }

314     @VisibleForTesting
315     static FieldInfo newFieldInfo(Class<?> parentType, FieldMetaData fieldMetaData) {
316         requireNonNull(fieldMetaData, "fieldMetaData");
317         final FieldValueMetaData fieldValueMetaData = fieldMetaData.valueMetaData;
318         final TypeSignature typeSignature;
319 
320         if (fieldValueMetaData.isStruct() && fieldValueMetaData.isTypedef() &&
321             parentType.getSimpleName().equals(fieldValueMetaData.getTypedefName())) {
322             // Handle the special case where a struct field refers to itself,
323             // where the Thrift compiler handles it as a typedef.
324             typeSignature = TypeSignature.ofNamed(parentType);
325         } else {
326             typeSignature = toTypeSignature(fieldValueMetaData);
327         }
328 
329         return new FieldInfo(fieldMetaData.fieldName,
330                              convertRequirement(fieldMetaData.requirementType),
331                              typeSignature);
332     }

334     @VisibleForTesting
335     static TypeSignature toTypeSignature(FieldValueMetaData fieldValueMetaData) {
336         if (fieldValueMetaData instanceof StructMetaData) {
337             return TypeSignature.ofNamed(((StructMetaData) fieldValueMetaData).structClass);
338         }
339 
340         if (fieldValueMetaData instanceof EnumMetaData) {
341             return TypeSignature.ofNamed(((EnumMetaData) fieldValueMetaData).enumClass);
342         }
343 
344         if (fieldValueMetaData instanceof ListMetaData) {
345             return TypeSignature.ofList(toTypeSignature(((ListMetaData) fieldValueMetaData).elemMetaData));
346         }
347 
348         if (fieldValueMetaData instanceof SetMetaData) {
349             return TypeSignature.ofSet(toTypeSignature(((SetMetaData) fieldValueMetaData).elemMetaData));
350         }
351 
352         if (fieldValueMetaData instanceof MapMetaData) {
353             return TypeSignature.ofMap(toTypeSignature(((MapMetaData) fieldValueMetaData).keyMetaData),
354                                        toTypeSignature(((MapMetaData) fieldValueMetaData).valueMetaData));
355         }
356 
357         if (fieldValueMetaData.isBinary()) {
358             return BINARY;
359         }
360 
361         switch (fieldValueMetaData.type) {
362             case TType.VOID:
363                 return VOID;
364             case TType.BOOL:
365                 return BOOL;
366             case TType.BYTE:
367                 return I8;
368             case TType.DOUBLE:
369                 return DOUBLE;
370             case TType.I16:
371                 return I16;
372             case TType.I32:
373                 return I32;
374             case TType.I64:
375                 return I64;
376             case TType.STRING:
377                 return STRING;
378         }
379 
380         final String unresolvedName;
381         if (fieldValueMetaData.isTypedef()) {
382             unresolvedName = fieldValueMetaData.getTypedefName();
383         } else {
384             unresolvedName = null;
385         }
386 
387         return TypeSignature.ofUnresolved(firstNonNull(unresolvedName, "unknown"));
388     }

390     private static FieldRequirement convertRequirement(byte value) {
391         switch (value) {
392             case TFieldRequirementType.REQUIRED:
393                 return FieldRequirement.REQUIRED;
394             case TFieldRequirementType.OPTIONAL:
395                 return FieldRequirement.OPTIONAL;
396             case TFieldRequirementType.DEFAULT:
397                 return FieldRequirement.DEFAULT;
398             default:
399                 throw new IllegalArgumentException("unknown requirement type: " + value);
400         }
401     }

414     @VisibleForTesting
415     static final class EntryBuilder {
416         private final Class<?> serviceType;
417         private final List<EndpointInfo> endpointInfos = new ArrayList<>();
418 
419         EntryBuilder(Class<?> serviceType) {
420             this.serviceType = requireNonNull(serviceType, "serviceType");
421         }
422 
423         EntryBuilder endpoint(EndpointInfo endpointInfo) {
424             endpointInfos.add(requireNonNull(endpointInfo, "endpointInfo"));
425             return this;
426         }
427 
428         Entry build() {
429             return new Entry(serviceType, endpointInfos);
430         }

435     @Override
436     public Map<String, String> loadDocStrings(Set<ServiceConfig> serviceConfigs) {
437         return serviceConfigs.stream()
438                              .flatMap(c -> c.service().as(THttpService.class).get().entries().values().stream())
439                              .flatMap(entry -> entry.interfaces().stream().map(Class::getClassLoader))
440                              .flatMap(loader -> docstringExtractor.getAllDocStrings(loader)
441                                                                   .entrySet().stream())
442                              .collect(toImmutableMap(Map.Entry::getKey, Map.Entry::getValue, (a, b) -> a));
443     }

452     @Override
453     public Optional<String> guessServiceName(Object exampleRequest) {
454         final TBase<?, ?> exampleTBase = asTBase(exampleRequest);
455         if (exampleTBase == null) {
456             return Optional.empty();
457         }
458 
459         return Optional.of(exampleTBase.getClass().getEnclosingClass().getName());
460     }
461 
462     @Override
463     public Optional<String> guessServiceMethodName(Object exampleRequest) {
464         final TBase<?, ?> exampleTBase = asTBase(exampleRequest);
465         if (exampleTBase == null) {
466             return Optional.empty();
467         }
468 
469         final String typeName = exampleTBase.getClass().getName();
470         return Optional.of(typeName.substring(typeName.lastIndexOf('$') + 1,
471                                               typeName.length() - REQUEST_STRUCT_SUFFIX.length()));
472     }
473 
474     @Override
475     public Optional<String> serializeExampleRequest(String serviceName, String methodName,
476                                                     Object exampleRequest) {
477         if (!(exampleRequest instanceof TBase)) {
478             return Optional.empty();
479         }
480 
481         final TBase<?, ?> exampleTBase = (TBase<?, ?>) exampleRequest;
482         final TSerializer serializer = new TSerializer(ThriftProtocolFactories.TEXT);
483         try {
484             return Optional.of(serializer.toString(exampleTBase, StandardCharsets.UTF_8.name()));
485         } catch (TException e) {
486             throw new Error("should never reach here", e);
487         }
488     }
489 
490     private static TBase<?, ?> asTBase(Object exampleRequest) {
491         final TBase<?, ?> exampleTBase = (TBase<?, ?>) exampleRequest;
492         final Class<?> type = exampleTBase.getClass();
493         if (!type.getName().endsWith(REQUEST_STRUCT_SUFFIX)) {
494             return null;
495         }
496 
497         final Class<?> serviceType = type.getEnclosingClass();
498         if (serviceType == null) {
499             return null;
500         }
501 
502         if (serviceType.getEnclosingClass() != null) {
503             return null;
504         }
505 
506         return exampleTBase;
507     }
