17  package com.linecorp.armeria.server;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.Set;
22  import java.util.concurrent.CompletionStage;
23  import java.util.function.BiFunction;
24  
25  import com.google.common.base.MoreObjects;
26  import com.google.common.collect.Sets;
27  
28  import com.linecorp.armeria.common.HttpMethod;
29  import com.linecorp.armeria.common.HttpRequest;
30  import com.linecorp.armeria.common.HttpResponse;
31  import com.linecorp.armeria.common.HttpStatus;
32  
33  /**
34   * {@link HttpMethod}, {@link PathMapping} and their corresponding {@link BiFunction}.
35   */
36  final class AnnotatedHttpService implements HttpService {
37      // TODO(trustin): Allow binding multiple functions.
38      //                https://github.com/line/armeria/issues/579
39  
40      /**
41       * EnumSet of HTTP Method, e.g., GET, POST, PUT, ...
42       */
43      private final Set<HttpMethod> methods;
44  
45      /**
46       * Path param extractor with placeholders, e.g., "/const1/{var1}/{var2}/const2"
47       */
48      private final PathMapping pathMapping;
49  
50      /**
51       * The {@link BiFunction} that will handle the request actually.
52       */
53      private final BiFunction<ServiceRequestContext, HttpRequest, Object> function;
54  
55      /**
56       * Creates a new instance.
57       */
58      AnnotatedHttpService(Iterable<HttpMethod> methods, PathMapping pathMapping,
59                           BiFunction<ServiceRequestContext, HttpRequest, Object> function) {
60          this.methods = Sets.immutableEnumSet(requireNonNull(methods, "methods"));
61          this.pathMapping = requireNonNull(pathMapping, "pathMapping");
62          this.function = requireNonNull(function, "function");
63      }
64  
65      /**
66       * Returns the {@link PathMapping} of this service.
67       */
68      PathMapping pathMapping() {
69          return pathMapping;
70      }
71  
72      /**
73       * Returns whether it's mapping overlaps with given {@link AnnotatedHttpService} instance.
74       */
75      boolean overlaps(AnnotatedHttpService entry) {
76          return false;
77          // FIXME(trustin): Make the path overlap detection work again.
78          //return !Sets.intersection(methods, entry.methods).isEmpty() &&
79          //       pathMapping.skeleton().equals(entry.pathMapping.skeleton());
80      }
81  
82      @Override
83      public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
84          if (!methods.contains(ctx.method())) {
85              return HttpResponse.of(HttpStatus.METHOD_NOT_ALLOWED);
86          }
87  
88          final Object ret = function.apply(ctx, req);
89          if (!(ret instanceof CompletionStage)) {
90              return HttpResponse.ofFailure(new IllegalStateException(
91                      "illegal return type: " + ret.getClass().getSimpleName()));
92          }
93  
94          @SuppressWarnings("unchecked")
95          CompletionStage<HttpResponse> castStage = (CompletionStage<HttpResponse>) ret;
96          return HttpResponse.from(castStage);
97      }
98  
99      @Override
100     public String toString() {
101         return MoreObjects.toStringHelper(this)
102                           .add("methods", methods)
103                           .add("pathMapping", pathMapping)
104                           .add("function", function).toString();
105     }
106 }
