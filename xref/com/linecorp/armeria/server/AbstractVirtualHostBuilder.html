60   * @see ChainedVirtualHostBuilder

63  @SuppressWarnings("rawtypes")

65  
66      private static final Logger logger = LoggerFactory.getLogger(AbstractVirtualHostBuilder.class);
67      
68      private static final ApplicationProtocolConfig HTTPS_ALPN_CFG = new ApplicationProtocolConfig(
69              Protocol.ALPN,
70              // NO_ADVERTISE is currently the only mode supported by both OpenSsl and JDK providers.
71              SelectorFailureBehavior.NO_ADVERTISE,
72              // ACCEPT is currently the only mode supported by both OpenSsl and JDK providers.
73              SelectedListenerFailureBehavior.ACCEPT,
74              ApplicationProtocolNames.HTTP_2,
75              ApplicationProtocolNames.HTTP_1_1);
76  
77      private static final String LOCAL_HOSTNAME;
78  
79      static {
80          // Try the '/usr/bin/hostname' command first, which is more reliable.
81          Process process = null;
82          String hostname = null;
83          try {
84              process = Runtime.getRuntime().exec("hostname");
85              final String line = new BufferedReader(new InputStreamReader(process.getInputStream())).readLine();
86              if (line == null) {
87                  logger.warn("The 'hostname' command returned nothing; " +
88                              "using InetAddress.getLocalHost() instead", line);
89              } else {
90                  hostname = normalizeDefaultHostname(line.trim());
91                  logger.info("Hostname: {} (via 'hostname' command)", hostname);
92              }
93          } catch (Exception e) {
94              logger.warn("Failed to get the hostname using the 'hostname' command; " +
95                          "using InetAddress.getLocalHost() instead", e);
96          } finally {
97              if (process != null) {
98                  process.destroy();
99              }
100         }
101 
102         if (hostname == null) {
103             try {
104                 hostname = normalizeDefaultHostname(InetAddress.getLocalHost().getHostName());
105                 logger.info("Hostname: {} (via InetAddress.getLocalHost())", hostname);
106             } catch (Exception e) {
107                 hostname = "localhost";
108                 logger.warn("Failed to get the hostname using InetAddress.getLocalHost(); " +
109                             "using 'localhost' instead", e);
110             }
111         }
112 
113         LOCAL_HOSTNAME = hostname;
114     }

116     private final String defaultHostname;

122     /**
123      * Creates a new {@link VirtualHostBuilder} whose hostname pattern is {@code "*"} (match-all).
124      */
125     AbstractVirtualHostBuilder() {
126         this(LOCAL_HOSTNAME, "*");
127     }
128 
129     /**
130      * Creates a new {@link VirtualHostBuilder} with the specified hostname pattern.
131      */
132     AbstractVirtualHostBuilder(String hostnamePattern) {
133         hostnamePattern = normalizeHostnamePattern(hostnamePattern);
134 
135         if ("*".equals(hostnamePattern)) {
136             defaultHostname = LOCAL_HOSTNAME;
137         } else if (hostnamePattern.startsWith("*.")) {
138             defaultHostname = hostnamePattern.substring(2);
139         } else {
140             defaultHostname = hostnamePattern;
141         }
142 
143         this.hostnamePattern = hostnamePattern;
144     }
145 
146     /**
147      * Creates a new {@link VirtualHostBuilder} with
148      * the default host name and the specified hostname pattern.
149      */
150     AbstractVirtualHostBuilder(String defaultHostname, String hostnamePattern) {
151         requireNonNull(defaultHostname, "defaultHostname");
152 
153         defaultHostname = normalizeDefaultHostname(defaultHostname);
154         hostnamePattern = normalizeHostnamePattern(hostnamePattern);
155         ensureHostnamePatternMatchesDefaultHostname(hostnamePattern, defaultHostname);
156 
157         this.defaultHostname = defaultHostname;
158         this.hostnamePattern = hostnamePattern;
159     }

176         return self();
177     }
178 
179     /**
180      * Sets the {@link SslContext} of this {@link VirtualHost} from the specified {@link SessionProtocol},
181      * {@code keyCertChainFile}, {@code keyFile} and {@code keyPassword}.
182      */
183     public B sslContext(
184             SessionProtocol protocol,
185             File keyCertChainFile, File keyFile, String keyPassword) throws SSLException {
186 
187         if (requireNonNull(protocol, "protocol") != SessionProtocol.HTTPS) {
188             throw new IllegalArgumentException("unsupported protocol: " + protocol);
189         }
190 
191         final SslContextBuilder builder = SslContextBuilder.forServer(keyCertChainFile, keyFile, keyPassword);
192 
193         builder.sslProvider(Flags.useOpenSsl() ? SslProvider.OPENSSL : SslProvider.JDK);
194         builder.ciphers(Http2SecurityUtil.CIPHERS, SupportedCipherSuiteFilter.INSTANCE);
195         builder.applicationProtocolConfig(HTTPS_ALPN_CFG);
196 
197         sslContext(builder.build());
198         return self();
199     }

217     /**
218      * Binds the specified {@link Service} at the the specified path pattern. e.g.
219      * <ul>
220      *   <li>{@code /login} (no path parameters)</li>
221      *   <li>{@code /users/{userId}} (curly-brace style)</li>
222      *   <li>{@code /list/:productType/by/:ordering} (colon style)</li>
223      *   <li>{@code exact:/foo/bar} (exact match)</li>
224      *   <li>{@code prefix:/files} (prefix match)</li>
225      *   <li><code>glob:/~&#42;/downloads/**</code> (glob pattern)</li>
226      *   <li>{@code regex:^/files/(?<filePath>.*)$} (regular expression)</li>
227      * </ul>
228      *
229      * @throws IllegalArgumentException if the specified path pattern is invalid
230      */
231     public B service(String pathPattern, Service<HttpRequest, HttpResponse> service) {
232         service(PathMapping.of(pathPattern), service);
233         return self();
234     }

312     /**
313      * Binds the specified annotated service object under the specified path prefix.
314      */
315     public B annotatedService(
316             String pathPrefix, Object service, Map<Class<?>, ResponseConverter> converters,
317             Function<Service<HttpRequest, HttpResponse>,
318                      ? extends Service<HttpRequest, HttpResponse>> decorator) {
319 
320         requireNonNull(pathPrefix, "pathPrefix");
321         requireNonNull(service, "service");
322         requireNonNull(converters, "converters");
323         requireNonNull(decorator, "decorator");
324 
325         final List<AnnotatedHttpService> entries =
326                 AnnotatedHttpServices.build(pathPrefix, service, converters);
327         entries.forEach(e -> service(e.pathMapping(), decorator.apply(e)));
328         return self();
329     }
330 
331     /**
332      * Decorates all {@link Service}s with the specified {@code decorator}.
333      *
334      * @param decorator the {@link Function} that decorates a {@link Service}
335      * @param <T> the type of the {@link Service} being decorated
336      * @param <R> the type of the {@link Service} {@code decorator} will produce
337      */
338     public <T extends Service<HttpRequest, HttpResponse>, R extends Service<HttpRequest, HttpResponse>>
339     B decorator(Function<T, R> decorator) {
340 
341         requireNonNull(decorator, "decorator");
342 
343         @SuppressWarnings("unchecked")
344         final Function<Service<HttpRequest, HttpResponse>, Service<HttpRequest, HttpResponse>> castDecorator =
345                 (Function<Service<HttpRequest, HttpResponse>, Service<HttpRequest, HttpResponse>>) decorator;
346 
347         if (this.decorator != null) {
348             this.decorator = this.decorator.andThen(castDecorator);
349         } else {
350             this.decorator = castDecorator;
351         }
352 
353         return self();
354     }
355 
356     @SuppressWarnings("unchecked")
357     final B self() {
358         return (B) this;
359     }
360 
361     /**
362      * Creates a new {@link VirtualHost}.
363      */
364     protected VirtualHost build() {
365         final VirtualHost virtualHost = new VirtualHost(defaultHostname, hostnamePattern, sslContext, services);
366         return decorator != null ? virtualHost.decorate(decorator) : virtualHost;
367     }
368 
369     @Override
370     public String toString() {
371         return VirtualHost.toString(getClass(), defaultHostname, hostnamePattern, sslContext, services);
372     }
