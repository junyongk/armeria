78  /**
79   * Encapsulates the state of a single server call, reading messages from the client, passing to business logic
80   * via {@link ServerCall.Listener}, and writing messages passed back to the response.
81   */
82  class ArmeriaServerCall<I, O> extends ServerCall<I, O>
83          implements ArmeriaMessageDeframer.Listener, TransportStatusListener {

123     ArmeriaServerCall(HttpHeaders clientHeaders,
124                       MethodDescriptor<I, O> method,
125                       CompressorRegistry compressorRegistry,
126                       DecompressorRegistry decompressorRegistry,
127                       HttpResponseWriter res,
128                       int maxInboundMessageSizeBytes,
129                       int maxOutboundMessageSizeBytes,
130                       ServiceRequestContext ctx,
131                       SerializationFormat serializationFormat,
132                       MessageMarshaller jsonMarshaller) {
133         requireNonNull(clientHeaders, "clientHeaders");
134         this.method = requireNonNull(method, "method");
135         this.ctx = requireNonNull(ctx, "ctx");
136         this.serializationFormat = requireNonNull(serializationFormat, "serializationFormat");
137         this.messageReader = new HttpStreamReader(
138                 requireNonNull(decompressorRegistry, "decompressorRegistry"),
139                 new ArmeriaMessageDeframer(
140                         this,
141                         maxInboundMessageSizeBytes,
142                         ctx.alloc())
143                         .decompressor(clientDecompressor(clientHeaders, decompressorRegistry)),
144                 this);
145         this.messageFramer = new ArmeriaMessageFramer(ctx.alloc(), maxOutboundMessageSizeBytes);
146         this.res = requireNonNull(res, "res");
147         this.compressorRegistry = requireNonNull(compressorRegistry, "compressorRegistry");
148         this.clientAcceptEncoding =
149                 Strings.emptyToNull(clientHeaders.get(GrpcHeaderNames.GRPC_ACCEPT_ENCODING));
150         this.decompressorRegistry = requireNonNull(decompressorRegistry, "decompressorRegistry");
151         marshaller = new GrpcMessageMarshaller<>(ctx.alloc(), serializationFormat, method, jsonMarshaller);
152     }

154     @Override
155     public void request(int numMessages) {
156         if (ctx.eventLoop().inEventLoop()) {
157             messageReader.request(numMessages);
158         } else {
159             ctx.eventLoop().submit(() -> messageReader.request(numMessages));
160         }
161     }

163     @Override
164     public void sendHeaders(Metadata unusedGrpcMetadata) {
165         checkState(!sendHeadersCalled, "sendHeaders already called");
166         checkState(!closeCalled, "call is closed");
167 
168         HttpHeaders headers = HttpHeaders.of(HttpStatus.OK);
169 
170         headers.setObject(HttpHeaderNames.CONTENT_TYPE, serializationFormat.mediaType());
171 
172         if (compressor == null || !messageCompression || clientAcceptEncoding == null) {
173             compressor = Codec.Identity.NONE;
174         } else {
175             List<String> acceptedEncodingsList =
176                     ACCEPT_ENCODING_SPLITTER.splitToList(clientAcceptEncoding);
177             if (!acceptedEncodingsList.contains(compressor.getMessageEncoding())) {
178                 // resort to using no compression.
179                 compressor = Codec.Identity.NONE;
180             }
181         }
182         messageFramer.setCompressor(compressor);
183 
184         // Always put compressor, even if it's identity.
185         headers.add(GrpcHeaderNames.GRPC_ENCODING, compressor.getMessageEncoding());
186 
187         String advertisedEncodings = String.join(",", decompressorRegistry.getAdvertisedMessageEncodings());
188         if (!advertisedEncodings.isEmpty()) {
189             headers.add(GrpcHeaderNames.GRPC_ACCEPT_ENCODING, advertisedEncodings);
190         }
191 
192         sendHeadersCalled = true;
193         res.write(headers);
194     }
195 
196     @Override
197     public void sendMessage(O message) {
198         checkState(sendHeadersCalled, "sendHeaders has not been called");
199         checkState(!closeCalled, "call is closed");
200 
201         try {
202             res.write(messageFramer.writePayload(marshaller.serializeResponse(message)));
203         } catch (RuntimeException e) {
204             close(Status.fromThrowable(e), EMPTY_METADATA);
205             throw e;
206         } catch (Throwable t) {
207             close(Status.fromThrowable(t), EMPTY_METADATA);
208             throw new RuntimeException(t);
209         }
210 
211         // We don't have a good way of listening to actual stream writes, and this functionality
212         // probably isn't used in most applications anyways, so just say we're always ready for more messages.
213         // In cases where flow control is activated, this may result in excessive internal buffering.
214         final boolean notifyReady = isReady();
215         if (notifyReady) {
216             try (SafeCloseable ignored = RequestContext.push(ctx)) {
217                 listener.onReady();
218             } catch (Throwable t) {
219                 close(Status.fromThrowable(t), EMPTY_METADATA);
220             }
221         }
222     }
223 
224     @Override
225     public boolean isReady() {
226         return !closeCalled;
227     }
228 
229     @Override
230     public void close(Status status, Metadata unusedGrpcMetadata) {
231         checkState(!closeCalled, "call already closed");
232 
233         closeCalled = true;
234 
235         HttpHeaders trailers = statusToTrailers(status, sendHeadersCalled);
236         final HttpObject trailersObj;
237         if (sendHeadersCalled && GrpcSerializationFormats.isGrpcWeb(serializationFormat)) {
238             // Normal trailers are not supported in grpc-web and must be encoded as a message.
239             // Message compression is not supported in grpc-web, so we don't bother using the normal
240             // ArmeriaMessageFramer.
241             trailersObj = serializeTrailersAsMessage(trailers);
242         } else {
243             trailersObj = trailers;
244         }
245         try {
246             res.write(trailersObj);
247             res.close();
248         } finally {
249             closeListener(status);
250         }
251     }

266         checkState(!sendHeadersCalled, "sendHeaders has been called");

277     @Override
278     public void messageRead(ByteBufOrStream message) {
279         boolean success = false;
280         final I request;
281 
282         try {
283             // Special case for unary calls.
284             if (messageReceived && method.getType() == MethodType.UNARY) {
285                 closeListener(Status.INTERNAL.withDescription(
286                         "More than one request messages for unary call or server streaming call"));
287                 return;
288             }
289             messageReceived = true;
290 
291             if (isCancelled()) {
292                 return;
293             }
294 
295             try {
296                 request = marshaller.deserializeRequest(message);
297                 success = true;
298             } catch (IOException e) {
299                 throw new UncheckedIOException(e);
300             }
301         } finally {
302             if (message.buf() != null && !success) {
303                 message.buf().release();
304             }
305         }
306 
307         try (SafeCloseable ignored = RequestContext.push(ctx)) {
308             listener.onMessage(request);
309         } catch (Throwable t) {
310             close(Status.fromThrowable(t), EMPTY_METADATA);
311         }
312     }
313 
314     @Override
315     public void endOfStream() {
316         clientStreamClosed = true;
317         if (!closeCalled) {
318             try (SafeCloseable ignored = RequestContext.push(ctx)) {
319                 listener.onHalfClose();
320             } catch (Throwable t) {
321                 close(Status.fromThrowable(t), EMPTY_METADATA);
322             }
323         }
324     }
325 
326     @Override
327     public void transportReportStatus(Status status) {
328         closeListener(status);
329     }

331     private void closeListener(Status newStatus) {
332         if (!listenerClosed) {
333             listenerClosed = true;
334             if (!clientStreamClosed) {
335                 messageReader().cancel();
336                 clientStreamClosed = true;
337             }
338             messageFramer.close();
339             ctx.logBuilder().responseContent(GrpcLogUtil.rpcResponse(newStatus), null);
340             if (newStatus.isOk()) {
341                 try (SafeCloseable ignored = RequestContext.push(ctx)) {
342                     listener.onComplete();
343                 } catch (Throwable t) {
344                     // This should not be possible with normal generated stubs which do not implement
345                     // onComplete, but is conceivable for a completely manually constructed stub.
346                     logger.warn("Error in gRPC onComplete handler.", t);
347                 }
348             } else {
349                 cancelled = true;
350                 try (SafeCloseable ignored = RequestContext.push(ctx)) {
351                     listener.onCancel();
352                 } catch (Throwable t) {
353                     if (!closeCalled) {
354                         // A custom error when dealing with client cancel or transport issues should be
355                         // returned. We have already closed the listener, so it will not receive any more
356                         // callbacks as designed.
357                         close(Status.fromThrowable(t), EMPTY_METADATA);
358                     }
359                 }
360                 // Transport error, not business logic error, so reset the stream.
361                 if (!closeCalled) {
362                     res.close(newStatus.asException());
363                 }
364             }
365         }
366     }

368     static HttpHeaders statusToTrailers(Status status, boolean headersSent) {
369         final HttpHeaders trailers;
370         if (headersSent) {
371             // Normal trailers.
372             trailers = new DefaultHttpHeaders();
373         } else {
374             // Trailers only response
375             trailers = new DefaultHttpHeaders(true, 3, true)
376                     .status(HttpStatus.OK)
377                     .set(HttpHeaderNames.CONTENT_TYPE, "application/grpc+proto");
378         }
379         trailers.add(GrpcHeaderNames.GRPC_STATUS, Integer.toString(status.getCode().value()));
380         if (status.getDescription() != null) {
381             trailers.add(GrpcHeaderNames.GRPC_MESSAGE,
382                          StatusMessageEscaper.escape(status.getDescription()));
383         }
384 
385         return trailers;
386     }

397     private HttpData serializeTrailersAsMessage(HttpHeaders trailers) {
398         ByteBuf serialized = ctx.alloc().buffer();
399         boolean success = false;
400         try {
401             serialized.writeByte(TRAILERS_FRAME_HEADER);
402             // Skip, we'll set this after serializing the headers.
403             serialized.writeInt(0);
404             for (Map.Entry<AsciiString, String> trailer : trailers) {
405                 encodeHeader(trailer.getKey(), trailer.getValue(), serialized);
406             }
407             int messageSize = serialized.readableBytes() - 5;
408             serialized.setInt(1, messageSize);
409             success = true;
410         } finally {
411             if (!success) {
412                 serialized.release();
413             }
414         }
415         return new ByteBufHttpData(serialized, true);
416     }

418     private static Decompressor clientDecompressor(HttpHeaders headers, DecompressorRegistry registry) {
419         String encoding = headers.get(GrpcHeaderNames.GRPC_ENCODING);
420         if (encoding == null) {
421             return Identity.NONE;
422         }
423         Decompressor decompressor = registry.lookupDecompressor(encoding);
424         return firstNonNull(decompressor, Identity.NONE);
425     }

427     // Copied from io.netty.handler.codec.http.HttpHeadersEncoder
428     private static void encodeHeader(CharSequence name, CharSequence value, ByteBuf buf) {
429         final int nameLen = name.length();
430         final int valueLen = value.length();
431         final int entryLen = nameLen + valueLen + 4;
432         buf.ensureWritable(entryLen);
433         int offset = buf.writerIndex();
434         writeAscii(buf, offset, name, nameLen);
435         offset += nameLen;
436         buf.setByte(offset++, ':');
437         buf.setByte(offset++, ' ');
438         writeAscii(buf, offset, value, valueLen);
439         offset += valueLen;
440         buf.setByte(offset++, '\r');
441         buf.setByte(offset++, '\n');
442         buf.writerIndex(offset);
443     }
444 
445     private static void writeAscii(ByteBuf buf, int offset, CharSequence value, int valueLen) {
446         if (value instanceof AsciiString) {
447             ByteBufUtil.copy((AsciiString) value, 0, buf, offset, valueLen);
448         } else {
449             writeCharSequence(buf, offset, value, valueLen);
450         }
451     }
452 
453     private static void writeCharSequence(ByteBuf buf, int offset, CharSequence value, int valueLen) {
454         for (int i = 0; i < valueLen; ++i) {
455             buf.setByte(offset++, c2b(value.charAt(i)));
456         }
457     }
