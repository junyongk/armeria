17  package com.linecorp.armeria.server.grpc;
18  
19  import static com.google.common.base.MoreObjects.firstNonNull;
20  import static com.google.common.base.Preconditions.checkArgument;
21  import static java.util.Objects.requireNonNull;
22  
23  import java.util.Set;
24  
25  import javax.annotation.Nullable;
26  
27  import com.google.common.collect.ImmutableSet;
28  
29  import com.linecorp.armeria.common.HttpRequest;
30  import com.linecorp.armeria.common.HttpResponse;
31  import com.linecorp.armeria.common.SerializationFormat;
32  import com.linecorp.armeria.common.grpc.GrpcSerializationFormats;
33  import com.linecorp.armeria.internal.grpc.ArmeriaMessageFramer;
34  import com.linecorp.armeria.server.ServerConfig;
35  import com.linecorp.armeria.server.Service;
36  import com.linecorp.armeria.server.encoding.HttpEncodingService;
37  
38  import io.grpc.BindableService;
39  import io.grpc.CompressorRegistry;
40  import io.grpc.DecompressorRegistry;
41  import io.grpc.ServerServiceDefinition;
42  
43  /**
44   * Constructs a {@link GrpcService} to serve GRPC services from within Armeria.
45   */
46  public final class GrpcServiceBuilder {
47  
48      private static final Set<SerializationFormat> DEFAULT_SUPPORTED_SERIALIZATION_FORMATS =
49              ImmutableSet.of(GrpcSerializationFormats.PROTO, GrpcSerializationFormats.PROTO_WEB);
50  
51      private final HandlerRegistry.Builder registryBuilder = new HandlerRegistry.Builder();
52  
53      @Nullable
54      private DecompressorRegistry decompressorRegistry;
55  
56      @Nullable
57      private CompressorRegistry compressorRegistry;
58  
59      private Set<SerializationFormat> supportedSerializationFormats = DEFAULT_SUPPORTED_SERIALIZATION_FORMATS;
60  
61      private int maxInboundMessageSizeBytes = GrpcService.NO_MAX_INBOUND_MESSAGE_SIZE;
62  
63      private int maxOutboundMessageSizeBytes = ArmeriaMessageFramer.NO_MAX_OUTBOUND_MESSAGE_SIZE;
64  
65      private boolean enableUnframedRequests;
66  
67      /**
68       * Adds a GRPC {@link ServerServiceDefinition} to this {@link GrpcServiceBuilder}, such as
69       * what's returned by {@link BindableService#bindService()}.
70       */
71      public GrpcServiceBuilder addService(ServerServiceDefinition service) {
72          registryBuilder.addService(requireNonNull(service, "service"));
73          return this;
74      }
75  
76      /**
77       * Adds a GRPC {@link BindableService} to this {@link GrpcServiceBuilder}. Most GRPC service
78       * implementations are {@link BindableService}s.
79       */
80      public GrpcServiceBuilder addService(BindableService bindableService) {
81          return addService(bindableService.bindService());
82      }
83  
84      /**
85       * Sets the {@link DecompressorRegistry} to use when decompressing messages. If not set, will use
86       * the default, which supports gzip only.
87       */
88      public GrpcServiceBuilder decompressorRegistry(DecompressorRegistry registry) {
89          decompressorRegistry = requireNonNull(registry, "registry");
90          return this;
91      }
92  
93      /**
94       * Sets the {@link CompressorRegistry} to use when compressing messages. If not set, will use the
95       * default, which supports gzip only.
96       */
97      public GrpcServiceBuilder compressorRegistry(CompressorRegistry registry) {
98          compressorRegistry = requireNonNull(registry, "registry");
99          return this;
100     }
101 
102     /**
103      * Sets the {@link SerializationFormat}s supported by this server. If not set, defaults to supporting binary
104      * protobuf formats. Enabling JSON can be useful, e.g., when migrating existing JSON services to gRPC.
105      */
106     public GrpcServiceBuilder supportedSerializationFormats(SerializationFormat... formats) {
107         return supportedSerializationFormats(ImmutableSet.copyOf(requireNonNull(formats, "formats")));
108     }
109 
110     /**
111      * Sets the {@link SerializationFormat}s supported by this server. If not set, defaults to supporting binary
112      * protobuf formats. JSON formats are currently very inefficient and not recommended for use in production.
113      *
114      * <p>TODO(anuraaga): Use faster JSON marshalling.
115      */
116     public GrpcServiceBuilder supportedSerializationFormats(Iterable<SerializationFormat> formats) {
117         requireNonNull(formats, "formats");
118         for (SerializationFormat format : formats) {
119             if (!GrpcSerializationFormats.isGrpc(format)) {
120                 throw new IllegalArgumentException("Not a GRPC serialization format: " + format);
121             }
122         }
123         supportedSerializationFormats = ImmutableSet.copyOf(formats);
124         return this;
125     }
126 
127     /**
128      * Sets the maximum size in bytes of an individual incoming message. If not set, will use
129      * {@link ServerConfig#defaultMaxRequestLength}. To support long-running RPC streams, it is recommended to
130      * set {@link ServerConfig#defaultMaxRequestLength} and {@link ServerConfig#defaultRequestTimeoutMillis} to
131      * very high values and set this to the expected limit of individual messages in the stream.
132      */
133     public GrpcServiceBuilder setMaxInboundMessageSizeBytes(int maxInboundMessageSizeBytes) {
134         checkArgument(maxInboundMessageSizeBytes > 0,
135                       "maxInboundMessageSizeBytes must be >0");
136         this.maxInboundMessageSizeBytes = maxInboundMessageSizeBytes;
137         return this;
138     }
139 
140     /**
141      * Sets the maximum size in bytes of an individual outgoing message. If not set, all messages will be sent.
142      * This can be a safety valve to prevent overflowing network connections with large messages due to business
143      * logic bugs.
144      */
145     public GrpcServiceBuilder setMaxOutboundMessageSizeBytes(int maxOutboundMessageSizeBytes) {
146         checkArgument(maxOutboundMessageSizeBytes > 0,
147                       "maxOutboundMessageSizeBytes must be >0");
148         this.maxOutboundMessageSizeBytes = maxOutboundMessageSizeBytes;
149         return this;
150     }
151 
152     /**
153      * Sets whether the service handles requests not framed using the GRPC wire protocol. Such requests should
154      * only have the serialized message as the request content, and the response content will only have the
155      * serialized response message. Supporting unframed requests can be useful, for example, when migrating an
156      * existing service to GRPC.
157      *
158      * <p>Limitations:
159      * <ul>
160      *     <li>Only unary methods (single request, single response) are supported.</li>
161      *     <li>
162      *         Message compression is not supported.
163      *         {@link HttpEncodingService} should be used instead for
164      *         transport level encoding.
165      *     </li>
166      * </ul>
167      */
168     public GrpcServiceBuilder enableUnframedRequests(boolean enableUnframedRequests) {
169         this.enableUnframedRequests = enableUnframedRequests;
170         return this;
171     }
172 
173     /**
174      * Constructs a new {@link GrpcService} that can be bound to
175      * {@link com.linecorp.armeria.server.ServerBuilder}. As GRPC services themselves are mounted at a path that
176      * corresponds to their protobuf package, you will almost always want to bind to a prefix, e.g. by using
177      * {@link com.linecorp.armeria.server.ServerBuilder#serviceUnder(String, Service)}.
178      */
179     public Service<HttpRequest, HttpResponse> build() {
180         GrpcService grpcService = new GrpcService(
181                 registryBuilder.build(),
182                 firstNonNull(decompressorRegistry, DecompressorRegistry.getDefaultInstance()),
183                 firstNonNull(compressorRegistry, CompressorRegistry.getDefaultInstance()),
184                 supportedSerializationFormats, maxOutboundMessageSizeBytes,
185                 maxInboundMessageSizeBytes);
186         return enableUnframedRequests ? grpcService.decorate(UnframedGrpcService::new) : grpcService;
187     }
188 }
