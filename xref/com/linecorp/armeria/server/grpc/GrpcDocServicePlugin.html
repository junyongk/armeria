112     @Override
113     public ServiceSpecification generateSpecification(Set<ServiceConfig> serviceConfigs) {
114         final Map<String, ServiceEntryBuilder> map = new LinkedHashMap<>();
115         for (ServiceConfig serviceConfig : serviceConfigs) {
116             final GrpcService grpcService = serviceConfig.service().as(GrpcService.class).get();
117             ImmutableSet.Builder<MediaType> supportedMediaTypesBuilder = ImmutableSet.builder();
118             supportedMediaTypesBuilder.addAll(grpcService.supportedSerializationFormats()
119                                                          .stream()
120                                                          .map(SerializationFormat::mediaType)::iterator);
121             if (serviceConfig.service().as(UnframedGrpcService.class).isPresent()) {
122                 if (grpcService.supportedSerializationFormats().contains(GrpcSerializationFormats.PROTO)) {
123                     // Normal clients of a GrpcService are not required to set a protocol when using unframed
124                     // requests but we set it here for clarity in DocService, where there may be multiple
125                     // services with similar mime types but different protocols.
126                     supportedMediaTypesBuilder.add(MediaType.PROTOBUF.withParameter("protocol", "gRPC"));
127                 }
128                 if (grpcService.supportedSerializationFormats().contains(GrpcSerializationFormats.JSON)) {
129                     supportedMediaTypesBuilder.add(MediaType.JSON_UTF_8.withParameter("protocol", "gRPC"));
130                 }
131             }
132             Set<MediaType> supportedMediaTypes = supportedMediaTypesBuilder.build();
133             for (ServerServiceDefinition service : grpcService.services()) {
134                 map.computeIfAbsent(
135                         service.getServiceDescriptor().getName(),
136                         s -> {
137                             FileDescriptor fileDescriptor =
138                                     ((ProtoFileDescriptorSupplier)
139                                             service.getServiceDescriptor().getSchemaDescriptor())
140                                             .getFileDescriptor();
141                             ServiceDescriptor serviceDescriptor = fileDescriptor.getServices().stream()
142                                           .filter(sd -> sd.getFullName().equals(
143                                                   service.getServiceDescriptor().getName()))
144                                           .findFirst()
145                                           .orElseThrow(IllegalStateException::new);
146                             return new ServiceEntryBuilder(serviceDescriptor);
147                         });
148             }
149             serviceConfig.pathMapping().prefix().ifPresent(
150                     path -> {
151                         for (ServerServiceDefinition service : grpcService.services()) {
152                             final String serviceName = service.getServiceDescriptor().getName();
153                             map.get(serviceName).endpoint(
154                                     new EndpointInfo(
155                                             serviceConfig.virtualHost().hostnamePattern(),
156                                             // Only the URL prefix, each method is served at a different path.
157                                             path + serviceName + '/',
158                                             "",
159                                             // No default mime type for GRPC, so just pick arbitrarily for now.
160                                             // TODO(anuraag): Consider allowing default mime type to be null.
161                                             Iterables.getFirst(supportedMediaTypes,
162                                                                GrpcSerializationFormats.PROTO.mediaType()),
163                                             supportedMediaTypes));
164                         }
165                     });
166         }
167         return generate(map.values().stream()
168                            .map(ServiceEntryBuilder::build)
169                            .collect(toImmutableList()));
170     }

172     @Override
173     public Map<String, String> loadDocStrings(Set<ServiceConfig> serviceConfigs) {
174         return serviceConfigs.stream()
175                 .flatMap(c -> c.service().as(GrpcService.class).get().services().stream())
176                 .flatMap(s -> docstringExtractor.getAllDocStrings(s.getClass().getClassLoader())
177                                                 .entrySet().stream())
178                 .collect(toImmutableMap(Map.Entry::getKey, Map.Entry::getValue, (a, b) -> a));
179     }

186     @Override
187     public Optional<String> serializeExampleRequest(String serviceName, String methodName,
188                                                     Object exampleRequest) {
189         try {
190             return Optional.of(JsonFormat.printer().print((Message) exampleRequest));
191         } catch (InvalidProtocolBufferException e) {
192             throw new UncheckedIOException(
193                     "Invalid example request protobuf. Is it missing required fields?", e);
194         }
195     }

197     @VisibleForTesting
198     ServiceSpecification generate(List<ServiceEntry> entries) {
199         final List<ServiceInfo> services = entries.stream()
200                                                   .map(this::newServiceInfo)
201                                                   .collect(toImmutableList());
202         return ServiceSpecification.generate(services, this::newNamedTypeInfo);
203     }

205     private NamedTypeInfo newNamedTypeInfo(TypeSignature typeSignature) {
206         Object descriptor = typeSignature.namedTypeDescriptor().get();
207         if (descriptor instanceof Descriptor) {
208             return newStructInfo((Descriptor) descriptor);
209         }
210 
211         assert descriptor instanceof EnumDescriptor;
212         return newEnumInfo((EnumDescriptor) descriptor);
213     }
214 
215     ServiceInfo newServiceInfo(ServiceEntry entry) {
216         final List<MethodInfo> functions = entry.methods().stream()
217                                                 .map(m -> newMethodInfo(m, entry))
218                                                 .collect(toImmutableList());
219         return new ServiceInfo(entry.name(), functions);
220     }
221 
222     @VisibleForTesting
223     MethodInfo newMethodInfo(MethodDescriptor method, ServiceEntry service) {
224         Set<EndpointInfo> methodEndpoints =
225                 service.endpointInfos.stream()
226                                      .map(e -> new EndpointInfo(
227                                              e.hostnamePattern(),
228                                              e.path() + method.getName(),
229                                              e.fragment(),
230                                              e.defaultMimeType(),
231                                              e.availableMimeTypes()))
232                                      .collect(toImmutableSet());
233         return new MethodInfo(
234                 method.getName(),
235                 namedMessageSignature(method.getOutputType()),
236                 // GRPC methods always take a single request parameter of message type.
237                 ImmutableList.of(
238                         new FieldInfo(
239                                 "request",
240                                 FieldRequirement.REQUIRED,
241                                 namedMessageSignature(method.getInputType()))),
242                 ImmutableList.of(),
243                 methodEndpoints);
244     }

255     private FieldInfo newFieldInfo(FieldDescriptor fieldDescriptor) {
256         return new FieldInfo(
257                 fieldDescriptor.getName(),
258                 fieldDescriptor.isRequired() ? FieldRequirement.REQUIRED : FieldRequirement.OPTIONAL,
259                 newFieldTypeInfo(fieldDescriptor));
260     }

262     @VisibleForTesting
263     TypeSignature newFieldTypeInfo(FieldDescriptor fieldDescriptor) {
264         if (fieldDescriptor.isMapField()) {
265             return TypeSignature.ofMap(
266                     newFieldTypeInfo(fieldDescriptor.getMessageType().findFieldByNumber(1)),
267                     newFieldTypeInfo(fieldDescriptor.getMessageType().findFieldByNumber(2)));
268         }
269         final TypeSignature fieldType;
270         switch (fieldDescriptor.getType()) {
271             case BOOL:
272                 fieldType = BOOL;
273                 break;
274             case BYTES:
275                 fieldType = BYTES;
276                 break;
277             case DOUBLE:
278                 fieldType = DOUBLE;
279                 break;
280             case FIXED32:
281                 fieldType = FIXED32;
282                 break;
283             case FIXED64:
284                 fieldType = FIXED64;
285                 break;
286             case FLOAT:
287                 fieldType = FLOAT;
288                 break;
289             case INT32:
290                 fieldType = INT32;
291                 break;
292             case INT64:
293                 fieldType = INT64;
294                 break;
295             case SFIXED32:
296                 fieldType = SFIXED32;
297                 break;
298             case SFIXED64:
299                 fieldType = SFIXED64;
300                 break;
301             case SINT32:
302                 fieldType = SINT32;
303                 break;
304             case SINT64:
305                 fieldType = SINT64;
306                 break;
307             case STRING:
308                 fieldType = STRING;
309                 break;
310             case UINT32:
311                 fieldType = UINT32;
312                 break;
313             case UINT64:
314                 fieldType = UINT64;
315                 break;
316             case MESSAGE:
317                 fieldType = namedMessageSignature(fieldDescriptor.getMessageType());
318                 break;
319             case GROUP:
320                 // This type has been deprecated since the launch of protocol buffers to open source.
321                 // There is no real metadata for this in the descriptor so we just treat as UNKNOWN
322                 // since it shouldn't happen in practice anyways.
323                 fieldType = UNKNOWN;
324                 break;
325             case ENUM:
326                 fieldType = TypeSignature.ofNamed(
327                         fieldDescriptor.getEnumType().getFullName(), fieldDescriptor.getEnumType());
328                 break;
329             default:
330                 fieldType = UNKNOWN;
331                 break;
332         }
333         return fieldDescriptor.isRepeated() ? TypeSignature.ofContainer("repeated", fieldType) : fieldType;
334     }

336     @VisibleForTesting
337     EnumInfo newEnumInfo(EnumDescriptor enumDescriptor) {
338         return new EnumInfo(
339                 enumDescriptor.getFullName(),
340                 enumDescriptor.getValues().stream()
341                               .map(d -> new EnumValueInfo(d.getName()))
342                               .collect(toImmutableList()));

345     private TypeSignature namedMessageSignature(Descriptor descriptor) {
346         return TypeSignature.ofNamed(descriptor.getFullName(), descriptor);
347     }

349     @VisibleForTesting
350     static final class ServiceEntry {
351         final ServiceDescriptor service;
352         final List<EndpointInfo> endpointInfos;
353 
354         ServiceEntry(ServiceDescriptor service,
355                      List<EndpointInfo> endpointInfos) {
356             this.service = service;
357             this.endpointInfos = ImmutableList.copyOf(endpointInfos);
358         }
359 
360         String name() {
361             return service.getFullName();
362         }
363 
364         List<MethodDescriptor> methods() {
365             return ImmutableList.copyOf(service.getMethods());
366         }
367     }
368 
369     @VisibleForTesting
370     static final class ServiceEntryBuilder {
371         private final ServiceDescriptor service;
372         private final List<EndpointInfo> endpointInfos = new ArrayList<>();
373 
374         ServiceEntryBuilder(ServiceDescriptor service) {
375             this.service = service;
376         }
377 
378         ServiceEntryBuilder endpoint(EndpointInfo endpointInfo) {
379             endpointInfos.add(requireNonNull(endpointInfo, "endpointInfo"));
380             return this;
381         }
382 
383         ServiceEntry build() {
384             return new ServiceEntry(service, endpointInfos);
385         }
