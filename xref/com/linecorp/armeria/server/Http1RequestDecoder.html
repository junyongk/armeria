63  final class Http1RequestDecoder extends ChannelDuplexHandler {
64  
65      private static final Logger logger = LoggerFactory.getLogger(Http1RequestDecoder.class);
66  
67      private static final Http2Settings DEFAULT_HTTP2_SETTINGS = new Http2Settings();
68  
69      private final ServerConfig cfg;
70      private final AsciiString scheme;
71      private final InboundTrafficController inboundTrafficController;
72  
73      /** The request being decoded currently. */
74      private DecodedHttpRequest req;
75      private int receivedRequests;
76      private int sentResponses;
77      private boolean discarding;
78  
79      Http1RequestDecoder(ServerConfig cfg, Channel channel, AsciiString scheme) {
80          this.cfg = cfg;
81          this.scheme = scheme;
82          inboundTrafficController = new InboundTrafficController(channel);
83      }
84  
85      @Override
86      public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
87          if (!(msg instanceof HttpObject)) {
88              ctx.fireChannelRead(msg);
89              return;
90          }
91  
92          // this.req can be set to null by fail(), so we keep it in a local variable.
93          DecodedHttpRequest req = this.req;
94          try {
95              if (discarding) {
96                  return;
97              }
98  
99              if (req == null) {
100                 if (msg instanceof HttpRequest) {
101                     final HttpRequest nettyReq = (HttpRequest) msg;
102                     if (!nettyReq.decoderResult().isSuccess()) {
103                         fail(ctx, HttpResponseStatus.BAD_REQUEST);
104                         return;
105                     }
106 
107                     final HttpHeaders nettyHeaders = nettyReq.headers();
108                     final int id = ++receivedRequests;
109 
110                     // Validate the 'content-length' header.
111                     final String contentLengthStr = nettyHeaders.get(HttpHeaderNames.CONTENT_LENGTH);
112                     if (contentLengthStr != null) {
113                         final long contentLength;
114                         try {
115                             contentLength = Long.parseLong(contentLengthStr);
116                         } catch (NumberFormatException ignored) {
117                             fail(ctx, HttpResponseStatus.BAD_REQUEST);
118                             return;
119                         }
120                         if (contentLength < 0) {
121                             fail(ctx, HttpResponseStatus.BAD_REQUEST);
122                             return;
123                         }
124                     }
125 
126                     nettyHeaders.set(ExtensionHeaderNames.SCHEME.text(), scheme);
127 
128                     this.req = req = new DecodedHttpRequest(
129                             ctx.channel().eventLoop(),
130                             id, 1,
131                             ArmeriaHttpUtil.toArmeria(nettyReq),
132                             HttpUtil.isKeepAlive(nettyReq),
133                             inboundTrafficController,
134                             cfg.defaultMaxRequestLength());
135 
136                     ctx.fireChannelRead(req);
137                 } else {
138                     fail(ctx, HttpResponseStatus.BAD_REQUEST);
139                     return;
140                 }
141             }
142 
143             if (req != null && msg instanceof HttpContent) {
144                 final HttpContent content = (HttpContent) msg;
145                 final DecoderResult decoderResult = content.decoderResult();
146                 if (!decoderResult.isSuccess()) {
147                     fail(ctx, HttpResponseStatus.BAD_REQUEST);
148                     req.close(new ProtocolViolationException(decoderResult.cause()));
149                     return;
150                 }
151 
152                 final ByteBuf data = content.content();
153                 final int dataLength = data.readableBytes();
154                 if (dataLength != 0) {
155                     req.increaseTransferredBytes(dataLength);
156                     final long maxContentLength = req.maxRequestLength();
157                     if (maxContentLength > 0 && req.transferredBytes() > maxContentLength) {
158                         fail(ctx, HttpResponseStatus.REQUEST_ENTITY_TOO_LARGE);
159                         req.close(ContentTooLargeException.get());
160                         return;
161                     }
162 
163                     if (req.isOpen()) {
164                         req.write(HttpData.of(data));
165                     }
166                 }
167 
168                 if (msg instanceof LastHttpContent) {
169                     final HttpHeaders trailingHeaders = ((LastHttpContent) msg).trailingHeaders();
170                     if (!trailingHeaders.isEmpty()) {
171                         req.write(ArmeriaHttpUtil.toArmeria(trailingHeaders));
172                     }
173 
174                     req.close();
175                     this.req = req = null;
176                 }
177             }
178         } catch (URISyntaxException e) {
179             fail(ctx, HttpResponseStatus.BAD_REQUEST);
180             if (req != null) {
181                 req.close(e);
182             }
183         } catch (Throwable t) {
184             fail(ctx, HttpResponseStatus.INTERNAL_SERVER_ERROR);
185             if (req != null) {
186                 req.close(t);
187             } else {
188                 logger.warn("Unexpected exception:", t);
189             }
190         } finally {
191             ReferenceCountUtil.release(msg);
192         }
193     }
194 
195     private void fail(ChannelHandlerContext ctx, HttpResponseStatus status) {
196         discarding = true;
197         req = null;
198 
199         final ChannelFuture future;
200         if (receivedRequests <= sentResponses) {
201             // Just close the connection if sending an error response will make the number of the sent
202             // responses exceed the number of the received requests, which doesn't make sense.
203             future = ctx.writeAndFlush(Unpooled.EMPTY_BUFFER);
204         } else {
205             final ByteBuf content = Unpooled.copiedBuffer(status.toString(), StandardCharsets.UTF_8);
206             final FullHttpResponse res = new DefaultFullHttpResponse(
207                     HttpVersion.HTTP_1_1, status, content);
208 
209             final HttpHeaders headers = res.headers();
210             headers.set(HttpHeaderNames.CONNECTION, HttpHeaderValues.CLOSE);
211             headers.set(HttpHeaderNames.CONTENT_TYPE, MediaType.PLAIN_TEXT_UTF_8);
212             headers.setInt(HttpHeaderNames.CONTENT_LENGTH, content.readableBytes());
213 
214             future = ctx.writeAndFlush(res);
215         }
216 
217         future.addListener(ChannelFutureListener.CLOSE);
218     }
219 
220     @Override
221     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
222         if (evt instanceof UpgradeEvent) {
223             // Generate the initial Http2Settings frame,
224             // so that the next handler knows the protocol upgrade occurred as well.
225             ctx.fireChannelRead(DEFAULT_HTTP2_SETTINGS);
226 
227             // Continue handling the upgrade request after the upgrade is complete.
228             final FullHttpRequest nettyReq = ((UpgradeEvent) evt).upgradeRequest();
229 
230             // Remove the headers related with the upgrade.
231             nettyReq.headers().remove(HttpHeaderNames.CONNECTION);
232             nettyReq.headers().remove(HttpHeaderNames.UPGRADE);
233             nettyReq.headers().remove(Http2CodecUtil.HTTP_UPGRADE_SETTINGS_HEADER);
234 
235             if (logger.isDebugEnabled()) {
236                 logger.debug("{} Handling the pre-upgrade request ({}): {} {} {} ({}B)",
237                              ctx.channel(), ((UpgradeEvent) evt).protocol(),
238                              nettyReq.method(), nettyReq.uri(), nettyReq.protocolVersion(),
239                              nettyReq.content().readableBytes());
240             }
241 
242             channelRead(ctx, nettyReq);
243             channelReadComplete(ctx);
244             return;
245         }
246 
247         ctx.fireUserEventTriggered(evt);
248     }
249 
250     @Override
251     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
252         if (msg instanceof HttpResponse &&
253             ((HttpResponse) msg).status().codeClass() != HttpStatusClass.INFORMATIONAL) {
254             sentResponses++;
255         }
256         ctx.write(msg, promise);
257     }
