17  package com.linecorp.armeria.server;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.AbstractMap.SimpleEntry;
22  import java.util.ArrayList;
23  import java.util.List;
24  import java.util.Map;
25  import java.util.Map.Entry;
26  import java.util.function.BiFunction;
27  
28  import javax.annotation.Nullable;
29  
30  import com.linecorp.armeria.common.util.LruMap;
31  import com.linecorp.armeria.server.composition.SimpleCompositeService;
32  
33  /**
34   * Maps a request path to a value associated with a matching {@link PathMapping}. Useful when building a
35   * service that delegates some or all of its requests to other services. e.g. {@link SimpleCompositeService}.
36   *
37   * @param <T> the type of the mapped value
38   */
39  public class PathMappings<T> implements BiFunction<String, String, PathMapped<T>> {
40  
41      private final ThreadLocal<Map<String, PathMapped<T>>> threadLocalCache;
42      private final List<Entry<PathMapping, T>> patterns = new ArrayList<>();
43      private boolean frozen;
44  
45      /**
46       * Creates a new instance with the default thread-local cache size (1024).
47       */
48      public PathMappings() {
49          this(1024);
50      }
51  
52      /**
53       * Creates a new instance with the specified thread-local LRU {@code cacheSize}.
54       */
55      public PathMappings(int cacheSize) {
56          if (cacheSize < 0) {
57              throw new IllegalArgumentException("cacheSize: " + cacheSize + " (expected: >= 0)");
58          }
59  
60          if (cacheSize == 0) {
61              threadLocalCache = null;
62          } else {
63              threadLocalCache = ThreadLocal.withInitial(() -> new LruMap<>(cacheSize));
64          }
65      }
66  
67      /**
68       * Adds the mapping from the specified {@link PathMapping} to the specified {@code value}.
69       *
70       * @return {@code this}
71       * @throws IllegalStateException if {@link #freeze()} or {@link #apply(String, String)} has been called
72       *                               already
73       */
74      public PathMappings<T> add(PathMapping pathMapping, T value) {
75          if (frozen) {
76              throw new IllegalStateException("can't add a new mapping once apply() was called");
77          }
78  
79          requireNonNull(pathMapping, "mapping");
80          requireNonNull(value, "value");
81  
82          patterns.add(new SimpleEntry<>(pathMapping, value));
83          return this;
84      }
85  
86      /**
87       * Prevents adding a new mapping via {@link #add(PathMapping, Object)}.
88       */
89      public PathMappings<T> freeze() {
90          frozen = true;
91          return this;
92      }
93  
94      /**
95       * Finds the {@link Service} whose {@link PathMapping} matches the specified {@code path}.
96       *
97       * @return a {@link PathMapped} that wraps the matching value if there's a match.
98       *         {@link PathMapped#empty()} if there's no match.
99       */
100     @Override
101     public PathMapped<T> apply(String path, @Nullable String query) {
102         freeze();
103 
104         // Look up the cache if the cache is available and query string does not exist.
105         final Map<String, PathMapped<T>> cache =
106                 query == null && threadLocalCache != null ? threadLocalCache.get() : null;
107 
108         if (cache != null) {
109             final PathMapped<T> value = cache.get(path);
110             if (value != null) {
111                 return value;
112             }
113         }
114 
115         // Cache miss or disabled cache
116         PathMapped<T> result = PathMapped.empty();
117         final int size = patterns.size();
118         for (int i = 0; i < size; i++) {
119             final Entry<PathMapping, T> e = patterns.get(i);
120             final PathMapping mapping = e.getKey();
121             final PathMappingResult mappingResult = mapping.apply(path, query);
122             if (mappingResult.isPresent()) {
123                 result = PathMapped.of(mapping, mappingResult, e.getValue());
124                 break;
125             }
126         }
127 
128         // Cache the result.
129         if (cache != null) {
130             cache.put(path, result);
131         }
132 
133         return result;
134     }
135 
136     @Override
137     public String toString() {
138         return patterns.toString();
139     }
140 }
