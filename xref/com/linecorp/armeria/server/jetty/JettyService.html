104     static JettyService forConfig(JettyServiceConfig config) {
105         final Function<ExecutorService, Server> serverFactory = blockingTaskExecutor -> {
106             final Server server = new Server(new ArmeriaThreadPool(blockingTaskExecutor));
107 
108             config.dumpAfterStart().ifPresent(server::setDumpAfterStart);
109             config.dumpBeforeStop().ifPresent(server::setDumpBeforeStop);
110             config.stopTimeoutMillis().ifPresent(server::setStopTimeout);
111 
112             config.handler().ifPresent(server::setHandler);
113             config.requestLog().ifPresent(server::setRequestLog);
114             config.sessionIdManagerFactory().ifPresent(
115                     factory -> server.setSessionIdManager(factory.apply(server)));
116 
117             config.handlerWrappers().forEach(server::insertHandler);
118             config.attrs().forEach(server::setAttribute);
119             config.beans().forEach(bean -> {
120                 final Boolean managed = bean.isManaged();
121                 if (managed == null) {
122                     server.addBean(bean.bean());
123                 } else {
124                     server.addBean(bean.bean(), managed);
125                 }
126             });
127 
128             config.eventListeners().forEach(server::addEventListener);
129             config.lifeCycleListeners().forEach(server::addLifeCycleListener);
130 
131             config.configurators().forEach(c -> c.accept(server));
132 
133             return server;
134         };
135 
136         final Consumer<Server> postStopTask = server -> {
137             try {
138                 logger.info("Destroying an embedded Jetty: {}", server);
139                 server.destroy();
140             } catch (Exception e) {
141                 logger.warn("Failed to destroy an embedded Jetty: {}", server, e);
142             }
143         };
144 
145         return new JettyService(config.hostname().orElse(null), serverFactory, postStopTask);
146     }

162     private JettyService(String hostname,

173     @Override
174     public void serviceAdded(ServiceConfig cfg) throws Exception {
175         if (armeriaServer != null) {
176             if (armeriaServer != cfg.server()) {
177                 throw new IllegalStateException("cannot be added to more than one server");
178             } else {
179                 return;
180             }
181         }
182 
183         armeriaServer = cfg.server();
184         armeriaServer.addListener(configurator);
185         if (hostname == null) {
186             hostname = armeriaServer.defaultHostname();
187         }
188     }
189 
190     void start() throws Exception {
191         boolean success = false;
192         try {
193             server = serverFactory.apply(armeriaServer.config().blockingTaskExecutor());
194             connector = new ArmeriaConnector(server);
195             server.addConnector(connector);
196 
197             if (!server.isStarted()) {
198                 logger.info("Starting an embedded Jetty: {}", server);
199                 server.start();
200                 startedServer = true;
201             } else {
202                 startedServer = false;
203             }
204             success = true;
205         } finally {
206             if (!success) {
207                 server = null;
208                 connector = null;
209             }
210         }
211     }
212 
213     void stop() throws Exception {
214         final Server server = this.server;
215         this.server = null;
216         connector = null;
217 
218         if (server == null || !startedServer) {
219             return;
220         }
221 
222         try {
223             logger.info("Stopping an embedded Jetty: {}", server);
224             server.stop();
225         } catch (Exception e) {
226             logger.warn("Failed to stop an embedded Jetty: {}", server, e);
227         }
228 
229         postStopTask.accept(server);
230     }

232     @Override
233     public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
234         final ArmeriaConnector connector = this.connector;
235 
236         final DefaultHttpResponse res = new DefaultHttpResponse();
237 
238         req.aggregate().handle(voidFunction((aReq, cause) -> {
239             if (cause != null) {
240                 logger.warn("{} Failed to aggregate a request:", ctx, cause);
241                 res.respond(HttpStatus.INTERNAL_SERVER_ERROR);
242                 return;
243             }
244 
245             boolean success = false;
246             try {
247                 final ArmeriaHttpTransport transport = new ArmeriaHttpTransport();
248                 final HttpChannel httpChannel = new HttpChannel(
249                         connector,
250                         connector.getHttpConfiguration(),
251                         new ArmeriaEndPoint(hostname, connector.getScheduler(),
252                                             ctx.localAddress(), ctx.remoteAddress()),
253                         transport);
254 
255                 fillRequest(ctx, aReq, httpChannel.getRequest());
256 
257                 ctx.blockingTaskExecutor().execute(() -> invoke(ctx, res, transport, httpChannel));
258                 success = true;
259             } finally {
260                 if (!success) {
261                     res.close();
262                 }
263             }
264         })).exceptionally(CompletionActions::log);
265 
266         return res;
267     }
268 
269     private void invoke(ServiceRequestContext ctx, HttpResponseWriter res,
270                         ArmeriaHttpTransport transport, HttpChannel httpChannel) {
271 
272         final Queue<HttpData> out = transport.out;
273         try {
274             server.handle(httpChannel);
275             httpChannel.getResponse().getHttpOutput().flush();
276 
277             final Throwable cause = transport.cause;
278             if (cause != null) {
279                 throw cause;
280             }
281 
282             final HttpHeaders headers = toResponseHeaders(transport);
283             res.write(headers);
284             for (;;) {
285                 final HttpData data = out.poll();
286                 if (data == null) {
287                     break;
288                 }
289                 res.write(data);
290             }
291             res.close();
292         } catch (Throwable t) {
293             logger.warn("{} Failed to produce a response:", ctx, t);
294             res.close();
295         }
296     }

298     private static void fillRequest(
299             ServiceRequestContext ctx, AggregatedHttpMessage aReq, Request jReq) {
300 
301         jReq.setDispatcherType(DispatcherType.REQUEST);
302         jReq.setAsyncSupported(true, "armeria");
303         jReq.setSecure(ctx.sessionProtocol().isTls());
304         jReq.setMetaData(toRequestMetadata(ctx, aReq));
305 
306         final HttpData content = aReq.content();
307         if (!content.isEmpty()) {
308             jReq.getHttpInput().addContent(new Content(ByteBuffer.wrap(
309                     content.array(), content.offset(), content.length())));
310         }
311         jReq.getHttpInput().eof();
312     }

314     private static MetaData.Request toRequestMetadata(ServiceRequestContext ctx, AggregatedHttpMessage aReq) {
315         // Construct the HttpURI
316         final StringBuilder uriBuf = new StringBuilder();
317         final HttpHeaders aHeaders = aReq.headers();
318 
319         uriBuf.append(ctx.sessionProtocol().isTls() ? "https" : "http");
320         uriBuf.append("://");
321         uriBuf.append(aHeaders.authority());
322         uriBuf.append(aHeaders.path());
323 
324         final HttpURI uri = new HttpURI(uriBuf.toString());
325         uri.setPath(ctx.mappedPath());
326 
327         // Convert HttpHeaders to HttpFields
328         final HttpFields jHeaders = new HttpFields(aHeaders.size());
329         aHeaders.forEach(e -> {
330             final AsciiString key = e.getKey();
331             if (!key.isEmpty() && key.byteAt(0) != ':') {
332                 jHeaders.add(key.toString(), e.getValue());
333             }
334         });
335 
336         return new MetaData.Request(
337                 aHeaders.method().name(), uri, HttpVersion.HTTP_1_1, jHeaders, aReq.content().length());
338     }

340     private static HttpHeaders toResponseHeaders(ArmeriaHttpTransport transport) {
341         final MetaData.Response info = transport.info;
342         if (info == null) {
343             throw new IllegalStateException("response metadata unavailable");
344         }
345 
346         final HttpHeaders headers = HttpHeaders.of(HttpStatus.valueOf(info.getStatus()));
347         info.getFields().forEach(e -> headers.add(HttpHeaderNames.of(e.getName()), e.getValue()));
348         return headers;
349     }

351     private static final class ArmeriaHttpTransport implements HttpTransport {
352 
353         final Queue<HttpData> out = new ArrayDeque<>();
354         MetaData.Response info;
355         Throwable cause;
356 
357         @Override
358         public void send(MetaData.Response info, boolean head,
359                          ByteBuffer content, boolean lastContent, Callback callback) {
360 
361             if (info != null) {
362                 this.info = info;
363             }
364 
365             final int length = content.remaining();
366             if (length == 0) {
367                 callback.succeeded();
368                 return;
369             }
370 
371             if (content.hasArray()) {
372                 final int from = content.arrayOffset() + content.position();
373                 out.add(HttpData.of(Arrays.copyOfRange(content.array(), from, from + length)));
374                 content.position(content.position() + length);
375             } else {
376                 final byte[] data = new byte[length];
377                 content.get(data);
378                 out.add(HttpData.of(data));
379             }
380 
381             callback.succeeded();
382         }
383 
384         @Override
385         public boolean isPushSupported() {
386             return false;
387         }
388 
389         @Override
390         public void push(MetaData.Request request) {}
391 
392         @Override
393         public void onCompleted() {}
394 
395         @Override
396         public void abort(Throwable failure) {
397             cause = failure;
398         }
399

401         public boolean isOptimizedForDirectBuffers() {
402             return false;
403         }

406     private static final class ArmeriaEndPoint extends AbstractEndPoint {
407 
408         private final InetSocketAddress localAddress;
409         private final InetSocketAddress remoteAddress;
410 
411         ArmeriaEndPoint(String hostname, Scheduler scheduler, SocketAddress local, SocketAddress remote) {
412             super(scheduler);
413 
414             localAddress = addHostname((InetSocketAddress) local, hostname);
415             remoteAddress = (InetSocketAddress) remote;
416 
417             setIdleTimeout(getIdleTimeout());
418         }
419 
420         @Override
421         public InetSocketAddress getLocalAddress() {
422             return localAddress;
423         }
424 
425         @Override
426         public InetSocketAddress getRemoteAddress() {
427             return remoteAddress;
428         }
429 
430         /**
431          * Adds the hostname string to the specified {@link InetSocketAddress} so that
432          * Jetty's {@code ServletRequest.getLocalName()} implementation returns the configured hostname.
433          */
434         private static InetSocketAddress addHostname(InetSocketAddress address, String hostname) {
435             try {
436                 return new InetSocketAddress(InetAddress.getByAddress(
437                         hostname, address.getAddress().getAddress()), address.getPort());
438             } catch (UnknownHostException e) {
439                 throw new Error(e); // Should never happen
440             }
441         }
