36      // These maps contain the entries whose key is a service name and value is a multimap.
37      // The multimap contains the entries whose key is the method name.
38      // An empty service name denotes 'all services'.
39      // An empty method name denotes 'all methods'.
40      // When a service name is empty, its method name shouldn't be empty.
41      // For exampleRequests, both service name and method name shouldn't be empty.
42  
43      private final Map<String, ListMultimap<String, HttpHeaders>> exampleHttpHeaders = new HashMap<>();
44      private final Map<String, ListMultimap<String, String>> exampleRequests = new HashMap<>();

51          return exampleHttpHeaders(ImmutableList.copyOf(exampleHttpHeaders));

154     private DocServiceBuilder exampleHttpHeaders0(String serviceName, String methodName,
155                                                   Iterable<? extends HttpHeaders> exampleHttpHeaders) {
156         for (HttpHeaders h : exampleHttpHeaders) {
157             requireNonNull(h, "exampleHttpHeaders contains null.");
158             this.exampleHttpHeaders.computeIfAbsent(serviceName, unused -> ArrayListMultimap.create())
159                                    .put(methodName, HttpHeaders.copyOf(h).asImmutable());
160         }
161         return this;
162     }

202     public DocServiceBuilder exampleRequestForMethod(String serviceName, String methodName,
203                                                      Iterable<?> exampleRequests) {
204         requireNonNull(serviceName, "serviceName");
205         requireNonNull(methodName, "methodName");
206         requireNonNull(exampleRequests, "exampleRequests");
207 
208         for (Object e : exampleRequests) {
209             requireNonNull(e, "exampleRequests contains null.");
210             exampleRequest0(serviceName, methodName,
211                             serializeExampleRequest(serviceName, methodName, e));
212         }
213         return this;
214     }

228     /**
229      * Adds the example requests which are applicable to the method denoted by the specified example requests.
230      * Please note that this method may fail if the specified requests object do not provide the information
231      * about their service and method names.
232      *
233      * @throws IllegalArgumentException if failed to get the service and method name from an example request
234      */
235     public DocServiceBuilder exampleRequest(Iterable<?> exampleRequests) {
236         requireNonNull(exampleRequests, "exampleRequests");
237         for (Object e : exampleRequests) {
238             requireNonNull(e, "exampleRequests contains null.");
239             final String[] result = guessAndSerializeExampleRequest(e);
240             exampleRequest0(result[0], result[1], result[2]);
241         }
242         return this;
243     }

245     private void exampleRequest0(String serviceName, String methodName, String serializedExampleRequest) {
246         exampleRequests.computeIfAbsent(serviceName, unused -> ArrayListMultimap.create())
247                        .put(methodName, serializedExampleRequest);
248     }

250     private static String serializeExampleRequest(
251             String serviceName, String methodName, Object exampleRequest) {
252 
253         if (exampleRequest instanceof CharSequence) {
254             return exampleRequest.toString();
255         }
256 
257         for (DocServicePlugin generator : DocService.plugins) {
258             final Optional<String> result =
259                     generator.serializeExampleRequest(serviceName, methodName, exampleRequest);
260             if (result.isPresent()) {
261                 return result.get();
262             }
263         }
264 
265         throw new IllegalArgumentException("could not find a plugin that can serialize: " + exampleRequest);
266     }
267 
268     /**
269      * Returns a tuple of a service name, a method name and a serialized example request.
270      */
271     private static String[] guessAndSerializeExampleRequest(Object exampleRequest) {
272         checkArgument(!(exampleRequest instanceof CharSequence),
273                       "can't guess service or method name from a string: ", exampleRequest);
274 
275         boolean guessed = false;
276         for (DocServicePlugin plugin : DocService.plugins) {
277             // Skip if the plugin does not support it.
278             if (plugin.supportedExampleRequestTypes().stream()
279                       .noneMatch(type -> type.isInstance(exampleRequest))) {
280                 continue;
281             }
282 
283             final Optional<String> serviceName = plugin.guessServiceName(exampleRequest);
284             final Optional<String> methodName = plugin.guessServiceMethodName(exampleRequest);
285 
286             // Skip if the plugin cannot guess the service and method name.
287             if (!serviceName.isPresent() || !methodName.isPresent()) {
288                 continue;
289             }
290 
291             guessed = true;
292             final String s = serviceName.get();
293             final String f = methodName.get();
294             Optional<String> serialized = plugin.serializeExampleRequest(s, f, exampleRequest);
295             if (serialized.isPresent()) {
296                 return new String[] { s, f, serialized.get() };
297             }
298         }
299 
300         if (guessed) {
301             throw new IllegalArgumentException(
302                     "could not find a plugin that can serialize: " + exampleRequest);
303         } else {
304             throw new IllegalArgumentException(
305                     "could not find a plugin that can guess the service and method name from: " +
306                     exampleRequest);
307         }
308     }

314         return new DocService(exampleHttpHeaders, exampleRequests);
