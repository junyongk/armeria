17  package com.linecorp.armeria.server.docs;
18  
19  import java.util.Map;
20  import java.util.Optional;
21  import java.util.Set;
22  
23  import com.google.common.collect.ImmutableMap;
24  import com.google.common.collect.ImmutableSet;
25  
26  import com.linecorp.armeria.server.Service;
27  import com.linecorp.armeria.server.ServiceConfig;
28  
29  /**
30   * Generates the {@link ServiceSpecification}s of the supported {@link Service}s.
31   */
32  public interface DocServicePlugin {
33  
34      // Methods related with generating a ServiceSpecification.
35  
36      /**
37       * Returns the {@link Set} of the {@link Service} types supported by this plugin.
38       */
39      Set<Class<? extends Service<?, ?>>> supportedServiceTypes();
40  
41      /**
42       * Generates a new {@link ServiceSpecification} that provides the information about the supported
43       * {@link Service}s.
44       *
45       * @param serviceConfigs the {@link ServiceConfig}s of the {@link Service}s that are instances of the
46       *                       {@link #supportedServiceTypes()}
47       */
48      ServiceSpecification generateSpecification(Set<ServiceConfig> serviceConfigs);
49  
50      // Methods related with extracting documentation strings.
51      // TODO(trustin): Define the docstring format.
52      // TODO(trustin): How do we specify the docstring of an exception thrown by a method?
53      // TODO(trustin): How do we specify the docstring of a method return value?
54  
55      /**
56       * Loads the documentation strings that describes services and their methods, enums and their values and
57       * structs/exceptions and their fields. The {@link Map} returned by this method will contain the
58       * documentation strings identified by the key strings that conforms to one of the following formats:
59       * <ul>
60       *   <li>{@code "{service name}"} - a docstring that describes a service</li>
61       *   <li>{@code "{service name}/{method name}"} - a docstring that describes a service method</li>
62       *   <li>{@code "{service name}/{method name}/{parameter name}"} - a docstring that describes
63       *       a service method parameter</li>
64       *   <li>{@code "{type name}"} - a docstring that describes an enum, a struct or an exception</li>
65       *   <li>{@code "{type name}/{field name}"} - a docstring that describes a field of an enum, a struct or
66       *       an exception</li>
67       * </ul>
68       */
69      default Map<String, String> loadDocStrings(Set<ServiceConfig> serviceConfigs) {
70          return ImmutableMap.of();
71      }
72  
73      // Methods related with serializing example requests.
74  
75      /**
76       * Returns the {@link Set} of the example request types supported by this plugin.
77       */
78      default Set<Class<?>> supportedExampleRequestTypes() {
79          return ImmutableSet.of();
80      }
81  
82      /**
83       * Guesses the name of the service who handles the specified example request.
84       *
85       * @param exampleRequest the example request object which is an instance of one of the {@link Class}es
86       *                       returned by {@link #supportedExampleRequestTypes()}
87       */
88      default Optional<String> guessServiceName(Object exampleRequest) {
89          return Optional.empty();
90      }
91  
92      /**
93       * Guesses the name of the service method who handles the specified example request.
94       *
95       * @param exampleRequest the example request object which is an instance of one of the {@link Class}es
96       *                       returned by {@link #supportedExampleRequestTypes()}
97       */
98      default Optional<String> guessServiceMethodName(Object exampleRequest) {
99          return Optional.empty();
100     }
101 
102     /**
103      * Serializes the specified example request into a string.
104      *
105      * @param serviceName the name of the service
106      * @param methodName the name of the method
107      * @param exampleRequest the example request
108      */
109     default Optional<String> serializeExampleRequest(
110             String serviceName, String methodName, Object exampleRequest) {
111         return Optional.empty();
112     }
113 }
