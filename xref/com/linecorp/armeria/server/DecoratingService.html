17  package com.linecorp.armeria.server;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.Optional;
22  
23  import com.linecorp.armeria.common.Request;
24  import com.linecorp.armeria.common.Response;
25  
26  /**
27   * A {@link Service} that decorates another {@link Service}. Use {@link SimpleDecoratingService} or
28   * {@link Service#decorate(DecoratingServiceFunction)} if your {@link Service} has the same {@link Request}
29   * and {@link Response} type with the {@link Service} being decorated.
30   *
31   * @param <T_I> the {@link Request} type of the {@link Service} being decorated
32   * @param <T_O> the {@link Response} type of the {@link Service} being decorated
33   * @param <R_I> the {@link Request} type of this {@link Service}
34   * @param <R_O> the {@link Response} type of this {@link Service}
35   */
36  public abstract class DecoratingService<T_I extends Request, T_O extends Response,
37                                          R_I extends Request, R_O extends Response>
38          implements Service<R_I, R_O> {
39  
40      private final Service<T_I, T_O> delegate;
41  
42      /**
43       * Creates a new instance that decorates the specified {@link Service}.
44       */
45      protected DecoratingService(Service<T_I, T_O> delegate) {
46          this.delegate = requireNonNull(delegate, "delegate");
47      }
48  
49      /**
50       * Returns the {@link Service} being decorated.
51       */
52      @SuppressWarnings("unchecked")
53      protected final <T extends Service<T_I, T_O>> T delegate() {
54          return (T) delegate;
55      }
56  
57      @Override
58      public void serviceAdded(ServiceConfig cfg) throws Exception {
59          ServiceCallbackInvoker.invokeServiceAdded(cfg, delegate);
60      }
61  
62      @Override
63      public final <T> Optional<T> as(Class<T> serviceType) {
64          final Optional<T> result = Service.super.as(serviceType);
65          return result.isPresent() ? result : delegate.as(serviceType);
66      }
67  
68      @Override
69      public String toString() {
70          final String simpleName = getClass().getSimpleName();
71          final String name = simpleName.isEmpty() ? getClass().getName() : simpleName;
72          return name + '(' + delegate + ')';
73      }
74  }
