17  package com.linecorp.armeria.server.auth;
18  
19  import java.util.Map;
20  import java.util.Map.Entry;
21  import java.util.Set;
22  
23  import com.google.common.base.Ascii;
24  import com.google.common.base.MoreObjects;
25  import com.google.common.base.Strings;
26  import com.google.common.collect.ImmutableMap;
27  import com.google.common.collect.ImmutableSet;
28  import com.google.common.collect.Sets;
29  
30  /**
31   * The bearer token of <a href="https://oauth.net/core/1.0a/#anchor12">OAuth 1.0a authentication</a>.
32   */
33  public final class OAuth1aToken {
34  
35      /**
36       * realm parameter. (optional)
37       */
38      private static final String REALM = "realm";
39  
40      /**
41       * oauth_consumer_key parameter.
42       */
43      private static final String OAUTH_CONSUMER_KEY = "oauth_consumer_key";
44  
45      /**
46       * oauth_token parameter.
47       */
48      private static final String OAUTH_TOKEN = "oauth_token";
49  
50      /**
51       * oauth_signature_method parameter.
52       */
53      private static final String OAUTH_SIGNATURE_METHOD = "oauth_signature_method";
54  
55      /**
56       * oauth_signature parameter.
57       */
58      private static final String OAUTH_SIGNATURE = "oauth_signature";
59  
60      /**
61       * oauth_timestamp parameter.
62       */
63      private static final String OAUTH_TIMESTAMP = "oauth_timestamp";
64  
65      /**
66       * oauth_nonce parameter.
67       */
68      private static final String OAUTH_NONCE = "oauth_nonce";
69  
70      /**
71       * version parameter. (optional)
72       * If not set, the default value is 1.0.
73       */
74      private static final String OAUTH_VERSION = "version";
75  
76      /**
77       * Set of required parameters.
78       */
79      private static final Set<String> REQUIRED_PARAM_KEYS = ImmutableSet.of(OAUTH_CONSUMER_KEY, OAUTH_TOKEN,
80                                                                             OAUTH_SIGNATURE_METHOD,
81                                                                             OAUTH_SIGNATURE, OAUTH_TIMESTAMP,
82                                                                             OAUTH_NONCE);
83  
84      /**
85       * Set of optional parameters.
86       */
87      private static final Set<String> OPTIONAL_PARAM_KEYS = ImmutableSet.of(REALM, OAUTH_VERSION);
88  
89      /**
90       * Set of defined parameters, regardless of it is required or not.
91       */
92      private static final Set<String> DEFINED_PARAM_KEYS = Sets.union(REQUIRED_PARAM_KEYS, OPTIONAL_PARAM_KEYS);
93  
94      /**
95       * Creates a new {@link OAuth1aToken} from the given arguments.
96       */
97      public static OAuth1aToken of(Map<String, String> params) {
98          return new OAuth1aToken(params);
99      }
100 
101     private final Map<String, String> params;
102 
103     private OAuth1aToken(Map<String, String> params) {
104         // Map builder with default version value.
105         ImmutableMap.Builder<String, String> builder = ImmutableMap.builder();
106 
107         for (Entry<String, String> param : params.entrySet()) {
108             String key = param.getKey();
109             String value = param.getValue();
110 
111             // Empty values are ignored.
112             if (!Strings.isNullOrEmpty(value)) {
113                 final String lowerCased = Ascii.toLowerCase(key);
114                 if (DEFINED_PARAM_KEYS.contains(lowerCased)) {
115                     // If given parameter is defined by Oauth1a protocol, add with lower-cased key.
116                     builder.put(lowerCased, value);
117                 } else {
118                     // Otherwise, just add.
119                     builder.put(key, value);
120                 }
121             }
122         }
123 
124         this.params = builder.build();
125 
126         if (!this.params.keySet().containsAll(REQUIRED_PARAM_KEYS)) {
127             Set<String> missing = Sets.difference(REQUIRED_PARAM_KEYS, this.params.keySet());
128             throw new IllegalArgumentException("Missing OAuth1a parameter exists: " + missing);
129         }
130 
131         try {
132             Long.parseLong(this.params.get(OAUTH_TIMESTAMP));
133         } catch (NumberFormatException e) {
134             throw new IllegalArgumentException(
135                     "Illegal " + OAUTH_TIMESTAMP + " value: " + this.params.get(OAUTH_TIMESTAMP));
136         }
137     }
138 
139     public String realm() {
140         return params.get(REALM);
141     }
142 
143     public String consumerKey() {
144         return params.get(OAUTH_CONSUMER_KEY);
145     }
146 
147     public String token() {
148         return params.get(OAUTH_TOKEN);
149     }
150 
151     public String signatureMethod() {
152         return params.get(OAUTH_SIGNATURE_METHOD);
153     }
154 
155     public String signature() {
156         return params.get(OAUTH_SIGNATURE);
157     }
158 
159     public String timestamp() {
160         return params.get(OAUTH_TIMESTAMP);
161     }
162 
163     public String nonce() {
164         return params.get(OAUTH_NONCE);
165     }
166 
167     /**
168      * Returns version. If not set, returns default value (1.0).
169      */
170     public String version() {
171         return params.getOrDefault(OAUTH_VERSION, "1.0");
172     }
173 
174     /**
175      * Returns additional (or, user-defined) parameters.
176      */
177     public Map<String, String> additionals() {
178         ImmutableMap.Builder<String, String> builder = ImmutableMap.builder();
179         for (String key : params.keySet()) {
180             if (!DEFINED_PARAM_KEYS.contains(key)) {
181                 builder.put(key, params.get(key));
182             }
183         }
184         return builder.build();
185     }
186 
187     @Override
188     public boolean equals(Object o) {
189         if (this == o) {
190             return true;
191         }
192         if (o == null || getClass() != o.getClass()) {
193             return false;
194         }
195         OAuth1aToken that = (OAuth1aToken) o;
196         return params.equals(that.params);
197     }
198 
199     @Override
200     public int hashCode() {
201         return params.hashCode();
202     }
203 
204     @Override
205     public String toString() {
206         return MoreObjects.toStringHelper(this)
207                           .add("realm", realm())
208                           .add("consumerKey", consumerKey())
209                           .add("token", "****")
210                           .add("signatureMethod", signatureMethod())
211                           .add("signature", signature())
212                           .add("timestamp", timestamp())
213                           .add("nonce", nonce())
214                           .add("version", version())
215                           .add("additionals", additionals())
216                           .toString();
217     }
218 }
