74  @Configuration
75  @EnableConfigurationProperties(ArmeriaSettings.class)
76  @ConditionalOnMissingBean(Server.class)
77  @EnableMetrics(proxyTargetClass = true)
78  public class ArmeriaAutoConfiguration {
79  
80      private static final Logger logger = LoggerFactory.getLogger(ArmeriaAutoConfiguration.class);
81  
82      private static final HealthChecker[] EMPTY_HEALTH_CHECKERS = new HealthChecker[0];
83  
84      private static final Port DEFAULT_PORT = new Port().setPort(8080)
85                                                         .setProtocol(SessionProtocol.HTTP);
86  
87      /**
88       * Create a {@link Server} bean.
89       */
90      @Bean
91      public Server armeriaServer(
92              ArmeriaSettings armeriaSettings,
93              MetricRegistry metricRegistry,
94              Optional<List<HealthChecker>> healthCheckers,
95              Optional<List<ArmeriaServerConfigurator>> armeriaServiceInitializers,
96              Optional<List<ThriftServiceRegistrationBean>> thriftServiceRegistrationBeans,
97              Optional<List<HttpServiceRegistrationBean>> httpServiceRegistrationBeans,
98              Optional<List<AnnotatedServiceRegistrationBean>> annotatedServiceRegistrationBeans)
99              throws InterruptedException {
100         if (!armeriaServiceInitializers.isPresent() &&
101             !thriftServiceRegistrationBeans.isPresent() &&
102             !httpServiceRegistrationBeans.isPresent()) {
103             // No services to register, no need to start up armeria server.
104             return null;
105         }
106 
107         final ServerBuilder server = new ServerBuilder();
108         if (armeriaSettings.getGracefulShutdownQuietPeriodMillis() != -1 &&
109             armeriaSettings.getGracefulShutdownTimeoutMillis() != -1) {
110             server.gracefulShutdownTimeout(
111                     armeriaSettings.getGracefulShutdownQuietPeriodMillis(),
112                     armeriaSettings.getGracefulShutdownTimeoutMillis());
113         }
114 
115         configurePorts(armeriaSettings, server);
116 
117         List<TBase<?, ?>> docServiceRequests = new ArrayList<>();
118         Map<String, Collection<HttpHeaders>> docServiceHeaders = new HashMap<>();
119         thriftServiceRegistrationBeans.ifPresent(beans -> beans.forEach(bean -> {
120             Service<HttpRequest, HttpResponse> service = bean.getService().decorate(bean.getDecorator());
121             if (armeriaSettings.isEnableDropwizardMetrics()) {
122                 service = service.decorate(
123                         DropwizardMetricCollectingService.newDecorator(
124                                 metricRegistry, serviceMetricName(bean.getServiceName())));
125             }
126 
127             server.service(bean.getPath(), service);
128             docServiceRequests.addAll(bean.getExampleRequests());
129             bean.getService().as(THttpService.class).ifPresent(
130                     beanService -> beanService.entries().forEach((serviceName, entry) -> {
131                         for (Class<?> iface : entry.interfaces()) {
132                             docServiceHeaders.put(iface.getEnclosingClass().getName(),
133                                                   bean.getExampleHeaders());
134                         }
135                     }));
136         }));
137 
138         httpServiceRegistrationBeans.ifPresent(beans -> beans.forEach(bean -> {
139             Service<HttpRequest, HttpResponse> service = bean.getService().decorate(bean.getDecorator());
140             if (armeriaSettings.isEnableDropwizardMetrics()) {
141                 service = service.decorate(
142                         DropwizardMetricCollectingService.newDecorator(
143                                 metricRegistry, serviceMetricName(bean.getServiceName())));
144             }
145             server.service(bean.getPathMapping(), service);
146         }));
147 
148         annotatedServiceRegistrationBeans.ifPresent(beans -> beans.forEach(bean -> {
149             Function<Service<HttpRequest, HttpResponse>,
150                      ? extends Service<HttpRequest, HttpResponse>> decorator = bean.getDecorator();
151             if (armeriaSettings.isEnableDropwizardMetrics()) {
152                 decorator = decorator.andThen(DropwizardMetricCollectingService.newDecorator(
153                         metricRegistry, serviceMetricName(bean.getServiceName())));
154             }
155             server.annotatedService(bean.getPathPrefix(), bean.getService(), decorator);
156         }));
157 
158         if (!Strings.isNullOrEmpty(armeriaSettings.getHealthCheckPath())) {
159             server.service(armeriaSettings.getHealthCheckPath(),
160                            new HttpHealthCheckService(healthCheckers.orElseGet(Collections::emptyList)
161                                                                     .toArray(EMPTY_HEALTH_CHECKERS)));
162         }
163 
164         if (!Strings.isNullOrEmpty(armeriaSettings.getDocsPath())) {
165             DocServiceBuilder docServiceBuilder = new DocServiceBuilder();
166             docServiceBuilder.exampleRequest(docServiceRequests);
167             for (Entry<String, Collection<HttpHeaders>> entry : docServiceHeaders.entrySet()) {
168                 docServiceBuilder.exampleHttpHeaders(entry.getKey(), entry.getValue());
169             }
170             server.serviceUnder(armeriaSettings.getDocsPath(), docServiceBuilder.build());
171         }
172 
173         if (!Strings.isNullOrEmpty(armeriaSettings.getMetricsPath())) {
174             ObjectMapper objectMapper = new ObjectMapper()
175                     .enable(SerializationFeature.INDENT_OUTPUT)
176                     .registerModule(new MetricsModule(TimeUnit.SECONDS,
177                                                       TimeUnit.MILLISECONDS,
178                                                       true));
179             server.service(
180                     armeriaSettings.getMetricsPath(),
181                     new AbstractHttpService() {
182                         @Override
183                         protected void doGet(ServiceRequestContext ctx, HttpRequest req,
184                                              HttpResponseWriter res) throws Exception {
185                             res.respond(HttpStatus.OK, MediaType.JSON_UTF_8,
186                                         objectMapper.writeValueAsBytes(metricRegistry));
187                         }
188                     });
189         }
190 
191         armeriaServiceInitializers.ifPresent(
192                 initializers -> initializers.forEach(
193                         initializer -> initializer.configure(server)));
194 
195         Server s = server.build();
196         s.start().join();
197         logger.info("Armeria server started at ports: {}", s.activePorts());
198         return s;
199     }
200 
201     private static void configurePorts(ArmeriaSettings armeriaSettings, ServerBuilder server) {
202         if (armeriaSettings.getPorts().isEmpty()) {
203             server.port(DEFAULT_PORT.getPort(), DEFAULT_PORT.getProtocol());
204             return;
205         }
206 
207         for (Port p : armeriaSettings.getPorts()) {
208             final String ip = p.getIp();
209             final String iface = p.getIface();
210             final int port = p.getPort();
211             final SessionProtocol proto = p.getProtocol();
212 
213             if (ip == null) {
214                 if (iface == null) {
215                     server.port(new InetSocketAddress(port), proto);
216                 } else {
217                     try {
218                         Enumeration<InetAddress> e = NetworkInterface.getByName(iface).getInetAddresses();
219                         while (e.hasMoreElements()) {
220                             server.port(new InetSocketAddress(e.nextElement(), port), proto);
221                         }
222                     } catch (SocketException e) {
223                         throw new IllegalStateException("Failed to find an iface: " + iface, e);
224                     }
225                 }
226             } else if (iface == null) {
227                 if (NetUtil.isValidIpV4Address(ip) || NetUtil.isValidIpV6Address(ip)) {
228                     final byte[] bytes = NetUtil.createByteArrayFromIpAddressString(ip);
229                     try {
230                         server.port(new InetSocketAddress(InetAddress.getByAddress(bytes), port), proto);
231                     } catch (UnknownHostException e) {
232                         // Should never happen.
233                         throw new Error(e);
234                     }
235                 } else {
236                     throw new IllegalStateException("invalid IP address: " + ip);
237                 }
238             } else {
239                 throw new IllegalStateException("A port cannot have both IP and iface: " + p);
240             }
241         }
242     }
