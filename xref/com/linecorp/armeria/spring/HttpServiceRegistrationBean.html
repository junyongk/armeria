16  package com.linecorp.armeria.spring;
17  
18  import java.util.function.Function;
19  
20  import javax.validation.constraints.NotNull;
21  
22  import com.linecorp.armeria.common.HttpRequest;
23  import com.linecorp.armeria.common.HttpResponse;
24  import com.linecorp.armeria.server.PathMapping;
25  import com.linecorp.armeria.server.Service;
26  
27  /**
28   * A bean with information for registering a http service. It enables dropwizard
29   * monitoring of the service automatically.
30   * <pre>{@code
31   * > @Bean
32   * > public HttpServiceRegistrationBean okService() {
33   * >     return new HttpServiceRegistrationBean()
34   * >             .setServiceName("okService")
35   * >             .setService(new OkService())
36   * >             .setPathMapping(PathMapping.ofExact("/ok"))
37   * >             .setDecorator(LoggingService.newDecorator());
38   * > }
39   * }</pre>
40   */
41  public class HttpServiceRegistrationBean {
42  
43      /**
44       * The http service to register.
45       */
46      @NotNull
47      private Service<HttpRequest, HttpResponse> service;
48  
49      /**
50       * The pathMapping for the http service. For example, {@code PathMapping.ofPrefix("/foobar")}.
51       */
52      @NotNull
53      private PathMapping pathMapping;
54  
55      /**
56       * A service name to use in monitoring.
57       */
58      @NotNull
59      private String serviceName;
60  
61      /**
62       * The decorator of the HTTP service.
63       */
64      @NotNull
65      private Function<Service<HttpRequest, HttpResponse>,
66                       ? extends Service<HttpRequest, HttpResponse>> decorator = Function.identity();
67  
68      /**
69       * Returns the http {@link Service} registered to this bean.
70       */
71      @NotNull
72      public Service<HttpRequest, HttpResponse> getService() {
73          return service;
74      }
75  
76      /**
77       * Register a http {@link Service}.
78       */
79      public HttpServiceRegistrationBean setService(@NotNull Service<HttpRequest, HttpResponse> service) {
80          this.service = service;
81          return this;
82      }
83  
84      /**
85       * Returns the {@link PathMapping} that this service map to.
86       */
87      @NotNull
88      public PathMapping getPathMapping() {
89          return pathMapping;
90      }
91  
92      /**
93       * Sets a {@link PathMapping} that this service map to.
94       */
95      public HttpServiceRegistrationBean setPathMapping(@NotNull PathMapping pathMapping) {
96          this.pathMapping = pathMapping;
97          return this;
98      }
99  
100     /**
101      * Sets the path pattern of the service.
102      */
103     public HttpServiceRegistrationBean setPathPattern(@NotNull String pathPattern) {
104         return setPathMapping(PathMapping.of(pathPattern));
105     }
106 
107     /**
108      * Returns this service name to use in monitoring.
109      */
110     @NotNull
111     public String getServiceName() {
112         return serviceName;
113     }
114 
115     /**
116      * Sets service name to use in monitoring.
117      */
118     public HttpServiceRegistrationBean setServiceName(@NotNull String serviceName) {
119         this.serviceName = serviceName;
120         return this;
121     }
122 
123     /**
124      * Returns the decorator of the HTTP service.
125      */
126     @NotNull
127     public Function<Service<HttpRequest, HttpResponse>,
128                     ? extends Service<HttpRequest, HttpResponse>> getDecorator() {
129         return decorator;
130     }
131 
132     /**
133      * Sets the decorator of the HTTP service.
134      */
135     public HttpServiceRegistrationBean setDecorator(
136             @NotNull Function<Service<HttpRequest, HttpResponse>,
137                               ? extends Service<HttpRequest, HttpResponse>> decorator) {
138         this.decorator = decorator;
139         return this;
140     }
141 }
