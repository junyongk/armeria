38  /**
39   * Provides the metadata of a Thrift service function.
40   */
41  public final class ThriftFunction {
42  
43      private enum Type {
44          SYNC,
45          ASYNC
46      }
47  
48      private final Object func;
49      private final Type type;
50      private final Class<?> serviceType;
51      private final String name;
52      private final TBase<?, ?> result;
53      private final TFieldIdEnum[] argFields;
54      private final TFieldIdEnum successField;
55      private final Map<Class<Throwable>, TFieldIdEnum> exceptionFields;
56      private final Class<?>[] declaredExceptions;
57  
58      ThriftFunction(Class<?> serviceType, ProcessFunction<?, ?> func) throws Exception {
59          this(serviceType, func.getMethodName(), func, Type.SYNC,
60               getArgFields(func), getResult(func), getDeclaredExceptions(func));
61      }
62  
63      ThriftFunction(Class<?> serviceType, AsyncProcessFunction<?, ?, ?> func) throws Exception {
64          this(serviceType, func.getMethodName(), func, Type.ASYNC,
65               getArgFields(func), getResult(func), getDeclaredExceptions(func));
66      }
67  
68      private ThriftFunction(
69              Class<?> serviceType, String name, Object func, Type type,
70              TFieldIdEnum[] argFields, TBase<?, ?> result, Class<?>[] declaredExceptions) throws Exception {
71  
72          this.func = func;
73          this.type = type;
74          this.serviceType = serviceType;
75          this.name = name;
76          this.argFields = argFields;
77          this.result = result;
78          this.declaredExceptions = declaredExceptions;
79  
80          // Determine the success and exception fields of the function.
81          final ImmutableMap.Builder<Class<Throwable>, TFieldIdEnum> exceptionFieldsBuilder =
82                  ImmutableMap.builder();
83          TFieldIdEnum successField = null;
84  
85          if (result != null) { // if not oneway
86              @SuppressWarnings("unchecked")
87              final Class<? extends TBase<?, ?>> resultType = (Class<? extends TBase<?, ?>>) result.getClass();
88              @SuppressWarnings("unchecked")
89              final Map<TFieldIdEnum, FieldMetaData> metaDataMap =
90                      (Map<TFieldIdEnum, FieldMetaData>) FieldMetaData.getStructMetaDataMap(resultType);
91  
92              for (Entry<TFieldIdEnum, FieldMetaData> e : metaDataMap.entrySet()) {
93                  final TFieldIdEnum key = e.getKey();
94                  final String fieldName = key.getFieldName();
95                  if ("success".equals(fieldName)) {
96                      successField = key;
97                      continue;
98                  }
99  
100                 Class<?> fieldType = resultType.getField(fieldName).getType();
101                 if (Throwable.class.isAssignableFrom(fieldType)) {
102                     @SuppressWarnings("unchecked")
103                     Class<Throwable> exceptionFieldType = (Class<Throwable>) fieldType;
104                     exceptionFieldsBuilder.put(exceptionFieldType, key);
105                 }
106             }
107         }
108 
109         this.successField = successField;
110         exceptionFields = exceptionFieldsBuilder.build();
111     }

113     /**
114      * Returns {@code true} if this function is a one-way.
115      */
116     public boolean isOneWay() {
117         return result == null;
118     }
119 
120     /**
121      * Returns {@code true} if this function is asynchronous.
122      */
123     public boolean isAsync() {
124         return type == Type.ASYNC;
125     }

127     /**
128      * Returns the type of this function.
129      *
130      * @return {@link TMessageType#CALL} or {@link TMessageType#ONEWAY}

132     public byte messageType() {
133         return isOneWay() ? TMessageType.ONEWAY : TMessageType.CALL;

194     public TBase<?, ?> newArgs() {
195         if (isAsync()) {
196             return asyncFunc().getEmptyArgsInstance();
197         } else {

199         }
200     }

202     /**
203      * Returns a new arguments instance.
204      */
205     public TBase<?, ?> newArgs(List<Object> args) {
206         requireNonNull(args, "args");
207         final TBase<?, ?> newArgs = newArgs();
208         final int size = args.size();
209         for (int i = 0; i < size; i++) {
210             ThriftFieldAccess.set(newArgs, argFields[i], args.get(i));
211         }
212         return newArgs;
213     }

215     /**
216      * Returns a new empty result instance.
217      */
218     public TBase<?, ?> newResult() {
219         return result.deepCopy();
220     }

222     /**
223      * Sets the success field of the specified {@code result} to the specified {@code value}.
224      */
225     public void setSuccess(TBase<?, ?> result, Object value) {
226         if (successField != null) {
227             ThriftFieldAccess.set(result, successField, value);
228         }
229     }
230 
231     /**
232      * Converts the specified {@code result} into a Java object.
233      */
234     public Object getResult(TBase<?, ?> result) throws TException {
235         for (TFieldIdEnum fieldIdEnum : exceptionFields()) {
236             if (ThriftFieldAccess.isSet(result, fieldIdEnum)) {
237                 throw (TException) ThriftFieldAccess.get(result, fieldIdEnum);
238             }
239         }
240 
241         final TFieldIdEnum successField = successField();
242         if (successField == null) { //void method
243             return null;
244         } else if (ThriftFieldAccess.isSet(result, successField)) {
245             return ThriftFieldAccess.get(result, successField);
246         } else {
247             throw new TApplicationException(
248                     TApplicationException.MISSING_RESULT,
249                     result.getClass().getName() + '.' + successField.getFieldName());
250         }
251     }

253     private static TBase<?, ?> getResult(ProcessFunction<?, ?> func) {
254         return getResult0(Type.SYNC, func.getClass(), func.getMethodName());
255     }
256 
257     private static TBase<?, ?> getResult(AsyncProcessFunction<?, ?, ?> asyncFunc) {
258         return getResult0(Type.ASYNC, asyncFunc.getClass(), asyncFunc.getMethodName());
259     }
260 
261     private static TBase<?, ?> getResult0(Type type, Class<?> funcClass, String methodName) {
262 
263         final String resultTypeName = typeName(type, funcClass, methodName, methodName + "_result");
264         try {
265             @SuppressWarnings("unchecked")
266             final Class<TBase<?, ?>> resultType =
267                     (Class<TBase<?, ?>>) Class.forName(resultTypeName, false, funcClass.getClassLoader());
268             return resultType.newInstance();
269         } catch (ClassNotFoundException ignored) {
270             // Oneway function does not have a result type.
271             return null;
272         } catch (Exception e) {
273             throw new IllegalStateException("cannot determine the result type of method: " + methodName, e);
274         }
275     }
276 
277     /**
278      * Sets the exception field of the specified {@code result} to the specified {@code cause}.
279      */
280     public boolean setException(TBase<?, ?> result, Throwable cause) {
281         Class<?> causeType = cause.getClass();
282         for (Entry<Class<Throwable>, TFieldIdEnum> e : exceptionFields.entrySet()) {
283             if (e.getKey().isAssignableFrom(causeType)) {
284                 ThriftFieldAccess.set(result, e.getValue(), cause);
285                 return true;
286             }
287         }
288         return false;
289     }

299     private static TBase<?, ?> getArgs0(Type type, Class<?> funcClass, String methodName) {
300         final String argsTypeName = typeName(type, funcClass, methodName, methodName + "_args");
301         try {
302             @SuppressWarnings("unchecked")
303             final Class<TBase<?, ?>> argsType =
304                     (Class<TBase<?, ?>>) Class.forName(argsTypeName, false, funcClass.getClassLoader());
305             return argsType.newInstance();
306         } catch (Exception e) {
307             throw new IllegalStateException("cannot determine the args class of method: " + methodName, e);
308         }
309     }

319     private static TFieldIdEnum[] getArgFields0(Type type, Class<?> funcClass, String methodName) {
320         final String fieldIdEnumTypeName = typeName(type, funcClass, methodName, methodName + "_args$_Fields");
321         try {
322             Class<?> fieldIdEnumType = Class.forName(fieldIdEnumTypeName, false, funcClass.getClassLoader());
323             return (TFieldIdEnum[]) requireNonNull(fieldIdEnumType.getEnumConstants(),
324                                                    "field enum may not be empty");
325         } catch (Exception e) {
326             throw new IllegalStateException("cannot determine the arg fields of method: " + methodName, e);
327         }
328     }

338     private static Class<?>[] getDeclaredExceptions0(
339             Type type, Class<?> funcClass, String methodName) {
340 
341         final String ifaceTypeName = typeName(type, funcClass, methodName, "Iface");
342         try {
343             Class<?> ifaceType = Class.forName(ifaceTypeName, false, funcClass.getClassLoader());
344             for (Method m : ifaceType.getDeclaredMethods()) {
345                 if (!m.getName().equals(methodName)) {
346                     continue;
347                 }
348 
349                 return m.getExceptionTypes();
350             }
351 
352             throw new IllegalStateException("failed to find a method: " + methodName);
353         } catch (Exception e) {
354             throw new IllegalStateException(
355                     "cannot determine the declared exceptions of method: " + methodName, e);
356         }
357     }
358 
359     private static String typeName(Type type, Class<?> funcClass, String methodName, String toAppend) {
360         final String funcClassName = funcClass.getName();
361         final int serviceClassEndPos = funcClassName.lastIndexOf(
362                 (type == Type.SYNC ? "$Processor$" : "$AsyncProcessor$") + methodName);
363 
364         if (serviceClassEndPos <= 0) {
365             throw new IllegalStateException("cannot determine the service class of method: " + methodName);
366         }
367 
368         return funcClassName.substring(0, serviceClassEndPos) + '$' + toAppend;
369     }
