73  /**
74   * A deframer of messages transported in the GRPC wire format. See
75   * <a href="http://www.grpc.io/docs/guides/wire.html">GRPC Wire Protocol</a> for more detail on the protocol.
76   *
77   * <p>The logic has been mostly copied from {@code io.grpc.internal.MessageDeframer}, while removing the buffer
78   * abstraction in favor of using {@link ByteBuf} directly, and allowing the delivery of uncompressed frames as
79   * a {@link ByteBuf} to optimize message parsing.
80   */
81  public class ArmeriaMessageDeframer implements AutoCloseable {
82  
83      private static final String DEBUG_STRING = ArmeriaMessageDeframer.class.getName();
84  
85      private static final int HEADER_LENGTH = 5;
86      private static final int COMPRESSED_FLAG_MASK = 1;
87      private static final int RESERVED_MASK = 0xFE;
88  
89      /**
90       * A deframed message. For uncompressed messages, we have the entire buffer available and return it
91       * as is in {@code buf} to optimize parsing. For compressed messages, we will parse incrementally
92       * and thus return a {@link InputStream} in {@code stream}.
93       */
94      public static class ByteBufOrStream {
95          @Nullable
96          private final ByteBuf buf;
97          @Nullable
98          private final InputStream stream;
99  
100         @VisibleForTesting
101         public ByteBufOrStream(ByteBuf buf) {
102             this(requireNonNull(buf, "buf"), null);
103         }
104 
105         @VisibleForTesting
106         public ByteBufOrStream(InputStream stream) {
107             this(null, requireNonNull(stream, "stream"));
108         }
109 
110         private ByteBufOrStream(@Nullable ByteBuf buf, @Nullable InputStream stream) {
111             this.buf = buf;
112             this.stream = stream;
113         }
114 
115         @Nullable
116         public ByteBuf buf() {
117             return buf;
118         }
119 
120         @Nullable
121         public InputStream stream() {
122             return stream;
123         }
124 
125         @Override
126         public boolean equals(Object o) {
127             if (this == o) {
128                 return true;
129             }
130             if (o == null || getClass() != o.getClass()) {
131                 return false;
132             }
133 
134             ByteBufOrStream that = (ByteBufOrStream) o;
135 
136             return Objects.equals(buf, that.buf) && Objects.equals(stream, that.stream);
137         }
138 
139         @Override
140         public int hashCode() {
141             return Objects.hash(buf, stream);
142         }
143     }
144 
145     /**
146      * A listener of deframing events.
147      */
148     public interface Listener {
149 
150         /**
151          * Called to deliver the next complete message. Either {@code message.buf} or {@code message.stream}
152          * will be non-null. {@code message.buf} must be released, or {@code message.stream} must be closed by
153          * the callee.
154          */
155         void messageRead(ByteBufOrStream message);
156 
157         /**
158          * Called when the stream is complete and all messages have been successfully delivered.
159          */
160         void endOfStream();
161     }
162 
163     private enum State {
164         HEADER, BODY
165     }
166 
167     private final Listener listener;
168     private final int maxMessageSizeBytes;
169     private final ByteBufAllocator alloc;
170 
171     private State state = State.HEADER;
172     private int requiredLength = HEADER_LENGTH;
173     private Decompressor decompressor = Identity.NONE;
174 
175     private boolean compressedFlag;
176     private boolean endOfStream;
177     private CompositeByteBuf nextFrame;
178     private CompositeByteBuf unprocessed;
179     private long pendingDeliveries;
180     private boolean deliveryStalled = true;
181     private boolean inDelivery;
182     private boolean startedDeframing;
183 
184     public ArmeriaMessageDeframer(Listener listener,
185                                   int maxMessageSizeBytes,
186                                   ByteBufAllocator alloc) {
187         this.listener = requireNonNull(listener, "listener");
188         this.maxMessageSizeBytes = maxMessageSizeBytes;
189         this.alloc = requireNonNull(alloc, "alloc");
190 
191         unprocessed = alloc.compositeBuffer();
192     }
193 
194     /**
195      * Requests up to the given number of messages from the call to be delivered to
196      * {@link Listener#messageRead(ByteBufOrStream)}. No additional messages will be delivered.
197      *
198      * <p>If {@link #close()} has been called, this method will have no effect.
199      *
200      * @param numMessages the requested number of messages to be delivered to the listener.
201      */
202     public void request(int numMessages) {
203         checkArgument(numMessages > 0, "numMessages must be > 0");
204         if (isClosed()) {
205             return;
206         }
207         pendingDeliveries += numMessages;
208         deliver();
209     }
210 
211     /**
212      * Indicates whether delivery is currently stalled, pending receipt of more data.  This means
213      * that no additional data can be delivered to the application.
214      */
215     public boolean isStalled() {
216         return deliveryStalled;
217     }
218 
219     /**
220      * Adds the given data to this deframer and attempts delivery to the listener.
221      *
222      * @param data the raw data read from the remote endpoint. Must be non-null.
223      * @param endOfStream if {@code true}, indicates that {@code data} is the end of the stream from
224      *        the remote endpoint.  End of stream should not be used in the event of a transport
225      *        error, such as a stream reset.
226      * @throws IllegalStateException if {@link #close()} has been called previously or if
227      *         this method has previously been called with {@code endOfStream=true}.
228      */
229     public void deframe(HttpData data, boolean endOfStream) {
230         requireNonNull(data, "data");
231         checkNotClosed();
232         checkState(!this.endOfStream, "Past end of stream");
233 
234         startedDeframing = true;
235 
236         if (!data.isEmpty()) {
237             ByteBuf buf = alloc.buffer(data.length());
238             buf.writeBytes(data.array(), data.offset(), data.length());
239             unprocessed.addComponent(true, buf);
240         }
241 
242         // Indicate that all of the data for this stream has been received.
243         this.endOfStream = endOfStream;
244         deliver();
245     }
246 
247     /**
248      * Closes this deframer and frees any resources. After this method is called, additional
249      * calls will have no effect.
250      */
251     @Override
252     public void close() {
253         try {
254             if (unprocessed != null) {
255                 unprocessed.release();
256             }
257             if (nextFrame != null) {
258                 nextFrame.release();
259             }
260         } finally {
261             unprocessed = null;
262             nextFrame = null;
263         }
264     }
265 
266     /**
267      * Indicates whether or not this deframer has been closed.
268      */
269     public boolean isClosed() {
270         return unprocessed == null;
271     }
272 
273     public ArmeriaMessageDeframer decompressor(Decompressor decompressor) {
274         checkState(!startedDeframing, "Deframing has already started, cannot change decompressor mid-stream.");
275         this.decompressor = decompressor;
276         return this;
277     }
278 
279     /**
280      * Throws if this deframer has already been closed.
281      */
282     private void checkNotClosed() {
283         checkState(!isClosed(), "MessageDeframer is already closed");
284     }
285 
286     /**
287      * Reads and delivers as many messages to the listener as possible.
288      */
289     private void deliver() {
290         // We can have reentrancy here when using a direct executor, triggered by calls to
291         // request more messages. This is safe as we simply loop until pendingDelivers = 0
292         if (inDelivery) {
293             return;
294         }
295         inDelivery = true;
296         try {
297             // Process the uncompressed bytes.
298             while (pendingDeliveries > 0 && readRequiredBytes()) {
299                 switch (state) {
300                     case HEADER:
301                         processHeader();
302                         break;
303                     case BODY:
304                         // Read the body and deliver the message.
305                         processBody();
306 
307                         // Since we've delivered a message, decrement the number of pending
308                         // deliveries remaining.
309                         pendingDeliveries--;
310                         break;
311                     default:
312                         throw new IllegalStateException("Invalid state: " + state);
313                 }
314             }
315 
316             /*
317             * We are stalled when there are no more bytes to process. This allows delivering errors as
318             * soon as the buffered input has been consumed, independent of whether the application
319             * has requested another message.  At this point in the function, either all frames have been
320             * delivered, or unprocessed is empty.  If there is a partial message, it will be inside next
321             * frame and not in unprocessed.  If there is extra data but no pending deliveries, it will
322             * be in unprocessed.
323             */
324             boolean stalled = !unprocessed.isReadable();
325 
326             if (endOfStream && stalled) {
327                 boolean havePartialMessage = nextFrame != null && nextFrame.isReadable();
328                 if (!havePartialMessage) {
329                     listener.endOfStream();
330                     deliveryStalled = false;
331                     return;
332                 } else {
333                     // We've received the entire stream and have data available but we don't have
334                     // enough to read the next frame ... this is bad.
335                     throw Status.INTERNAL.withDescription(
336                             DEBUG_STRING + ": Encountered end-of-stream mid-frame").asRuntimeException();
337                 }
338             }
339             deliveryStalled = stalled;
340         } finally {
341             inDelivery = false;
342         }
343     }
344 
345     /**
346      * Attempts to read the required bytes into nextFrame.
347      *
348      * @return {@code true} if all of the required bytes have been read.
349      */
350     private boolean readRequiredBytes() {
351         if (nextFrame == null) {
352             nextFrame = alloc.compositeBuffer();
353         }
354 
355         // Read until the buffer contains all the required bytes.
356         int missingBytes;
357         while ((missingBytes = requiredLength - nextFrame.readableBytes()) > 0) {
358             int numUnprocessedBytes = unprocessed.readableBytes();
359             if (numUnprocessedBytes == 0) {
360                 // No more data is available.
361                 return false;
362             }
363             int toRead = Math.min(missingBytes, numUnprocessedBytes);
364             if (toRead > 0) {
365                 nextFrame.addComponent(true, unprocessed.readBytes(toRead));
366                 unprocessed.discardReadComponents();
367             }
368         }
369         return true;
370     }
371 
372     /**
373      * Processes the GRPC compression header which is composed of the compression flag and the outer
374      * frame length.
375      */
376     private void processHeader() {
377         int type = nextFrame.readUnsignedByte();
378         if ((type & RESERVED_MASK) != 0) {
379             throw Status.INTERNAL.withDescription(
380                     DEBUG_STRING + ": Frame header malformed: reserved bits not zero")
381                                  .asRuntimeException();
382         }
383         compressedFlag = (type & COMPRESSED_FLAG_MASK) != 0;
384 
385         // Update the required length to include the length of the frame.
386         requiredLength = nextFrame.readInt();
387         if (requiredLength < 0 || requiredLength > maxMessageSizeBytes) {
388             throw Status.RESOURCE_EXHAUSTED.withDescription(
389                     String.format("%s: Frame size %d exceeds maximum: %d. ",
390                                   DEBUG_STRING, requiredLength,
391                                   maxMessageSizeBytes)).asRuntimeException();
392         }
393 
394         // Continue reading the frame body.
395         state = State.BODY;
396     }
397 
398     /**
399      * Processes the body of the GRPC compression frame. A single compression frame may contain
400      * several GRPC messages within it.
401      */
402     private void processBody() {
403         ByteBufOrStream msg = compressedFlag ? getCompressedBody() : getUncompressedBody();
404         nextFrame = null;
405         listener.messageRead(msg);
406 
407         // Done with this frame, begin processing the next header.
408         state = State.HEADER;
409         requiredLength = HEADER_LENGTH;
410     }
411 
412     private ByteBufOrStream getUncompressedBody() {
413         return new ByteBufOrStream(nextFrame.consolidate());
414     }
415 
416     private ByteBufOrStream getCompressedBody() {
417         if (decompressor == Codec.Identity.NONE) {
418             throw Status.INTERNAL.withDescription(
419                     DEBUG_STRING + ": Can't decode compressed frame as compression not configured.")
420                                  .asRuntimeException();
421         }
422 
423         try {
424             // Enforce the maxMessageSizeBytes limit on the returned stream.
425             InputStream unlimitedStream =
426                     decompressor.decompress(new ByteBufInputStream(nextFrame, true));
427             return new ByteBufOrStream(
428                     new SizeEnforcingInputStream(unlimitedStream, maxMessageSizeBytes, DEBUG_STRING));
429         } catch (IOException e) {
430             throw new RuntimeException(e);
431         }
432     }

434     /**
435      * An {@link InputStream} that enforces the {@link #maxMessageSize} limit for compressed frames.
436      */
437     @VisibleForTesting
438     static final class SizeEnforcingInputStream extends FilterInputStream {
439         private final int maxMessageSize;
440         private final String debugString;
441         private long maxCount;
442         private long count;
443         private long mark = -1;
444 
445         SizeEnforcingInputStream(InputStream in, int maxMessageSize, String debugString) {
446             super(in);
447             this.maxMessageSize = maxMessageSize;
448             this.debugString = debugString;
449         }
450 
451         @Override
452         public int read() throws IOException {
453             int result = in.read();
454             if (result != -1) {
455                 count++;
456             }
457             verifySize();
458             reportCount();
459             return result;
460         }
461 
462         @Override
463         public int read(byte[] b, int off, int len) throws IOException {
464             int result = in.read(b, off, len);
465             if (result != -1) {
466                 count += result;
467             }
468             verifySize();
469             reportCount();
470             return result;
471         }
472 
473         @Override
474         public long skip(long n) throws IOException {
475             long result = in.skip(n);
476             count += result;
477             verifySize();
478             reportCount();
479             return result;
480         }
481 
482         @Override
483         public synchronized void mark(int readlimit) {
484             in.mark(readlimit);
485             mark = count;
486             // it's okay to mark even if mark isn't supported, as reset won't work
487         }
488 
489         @Override
490         public synchronized void reset() throws IOException {
491             if (!in.markSupported()) {
492                 throw new IOException("Mark not supported");
493             }
494             if (mark == -1) {
495                 throw new IOException("Mark not set");
496             }
497 
498             in.reset();
499             count = mark;
500         }
501 
502         private void reportCount() {
503             if (count > maxCount) {
504                 maxCount = count;
505             }
506         }
507 
508         private void verifySize() {
509             if (count > maxMessageSize) {
510                 throw Status.INTERNAL.withDescription(String.format(
511                         "%s: Compressed frame exceeds maximum frame size: %d. Bytes read: %d. ",
512                         debugString, maxMessageSize, count)).asRuntimeException();
513             }
514         }
515     }
