17  package com.linecorp.armeria.internal;
18  
19  import java.util.concurrent.TimeUnit;
20  import java.util.concurrent.atomic.AtomicBoolean;
21  import java.util.concurrent.atomic.AtomicInteger;
22  import java.util.concurrent.atomic.LongAdder;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import io.netty.bootstrap.ServerBootstrap;
28  import io.netty.channel.Channel;
29  import io.netty.channel.ChannelHandler;
30  import io.netty.channel.ChannelHandler.Sharable;
31  import io.netty.channel.ChannelHandlerContext;
32  import io.netty.channel.ChannelInboundHandlerAdapter;
33  import io.netty.channel.ChannelOption;
34  
35  /**
36   * Limit the number of open connections to the configured value.
37   * {@link ConnectionLimitingHandler} instance would be set to {@link ServerBootstrap#handler(ChannelHandler)}.
38   */
39  @Sharable
40  public final class ConnectionLimitingHandler extends ChannelInboundHandlerAdapter {
41  
42      private static final Logger logger = LoggerFactory.getLogger(ConnectionLimitingHandler.class);
43  
44      private final int maxNumConnections;
45      private final AtomicInteger numConnections = new AtomicInteger();
46  
47      private final AtomicBoolean loggingScheduled = new AtomicBoolean();
48      private final LongAdder numDroppedConnections = new LongAdder();
49  
50      public ConnectionLimitingHandler(int maxNumConnections) {
51          this.maxNumConnections = validateMaxNumConnections(maxNumConnections);
52      }
53  
54      @Override
55      @SuppressWarnings("unchecked")
56      public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
57          final Channel child = (Channel) msg;
58  
59          int conn = numConnections.incrementAndGet();
60          if (conn > 0 && conn <= maxNumConnections) {
61              child.closeFuture().addListener(future -> numConnections.decrementAndGet());
62              super.channelRead(ctx, msg);
63          } else {
64              numConnections.decrementAndGet();
65  
66              // Set linger option to 0 so that the server doesn't get too many TIME_WAIT states.
67              child.config().setOption(ChannelOption.SO_LINGER, 0);
68              child.unsafe().closeForcibly();
69  
70              numDroppedConnections.increment();
71  
72              if (loggingScheduled.compareAndSet(false, true)) {
73                  ctx.executor().schedule(this::writeNumDroppedConnectionsLog, 1, TimeUnit.SECONDS);
74              }
75          }
76      }
77  
78      private void writeNumDroppedConnectionsLog() {
79          loggingScheduled.set(false);
80  
81          long dropped = numDroppedConnections.sumThenReset();
82          if (dropped > 0) {
83              logger.warn("Dropped {} connection(s) to limit the number of open connections to {}",
84                          dropped, maxNumConnections);
85          }
86      }
87  
88      /**
89       * Returns the maximum allowed number of open connections.
90       */
91      public int maxNumConnections() {
92          return maxNumConnections;
93      }
94  
95      /**
96       * Returns the number of open connections.
97       */
98      public int numConnections() {
99          return numConnections.get();
100     }
101 
102     /**
103      * Validates the maximum allowed number of open connections. It must be a positive number.
104      */
105     public static int validateMaxNumConnections(int maxNumConnections) {
106         if (maxNumConnections <= 0) {
107             throw new IllegalArgumentException("maxNumConnections: " + maxNumConnections + " (expected: > 0)");
108         }
109         return maxNumConnections;
110     }
111 }
