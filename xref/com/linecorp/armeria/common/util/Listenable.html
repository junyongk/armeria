3   import java.util.function.Consumer;
4   
5   /**
6    * An interface that accepts item change listeners.
7    */
8   public interface Listenable<T> {
9       /**
10       * Adds a {@link Consumer} that will be invoked when a {@link Listenable} changes its value.
11       */
12      void addListener(Consumer<? super T> listener);
13  
14      /**
15       * Remove a listener.
16       */
17      void removeListener(Consumer<?> listener);
18  }
