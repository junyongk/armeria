17  package com.linecorp.armeria.common.logging;
18  
19  import java.util.EventListener;
20  import java.util.Objects;
21  
22  /**
23   * Invoked when {@link RequestLog} meets the {@link RequestLogAvailability} specified with
24   * {@link RequestLog#addListener(RequestLogListener, RequestLogAvailability)}.
25   */
26  @FunctionalInterface
27  public interface RequestLogListener extends EventListener {
28  
29      /**
30       * Invoked when {@link RequestLog} meets the {@link RequestLogAvailability} specified with
31       * {@link RequestLog#addListener(RequestLogListener, RequestLogAvailability)}.
32       */
33      void onRequestLog(RequestLog log) throws Exception;
34  
35      /**
36       * Returns a composed listener that calls this listener first and then the specified one.
37       */
38      default RequestLogListener andThen(RequestLogListener other) {
39          Objects.requireNonNull(other, "other");
40  
41          final RequestLogListener first = this;
42          final RequestLogListener second = other;
43  
44          return log -> {
45              RequestLogListenerInvoker.invokeOnRequestLog(first, log);
46              second.onRequestLog(log);
47          };
48      }
49  }
