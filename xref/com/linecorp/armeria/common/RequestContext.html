58  /**
59   * Provides information about a {@link Request}, its {@link Response} and related utilities.
60   * A server-side {@link Request} has a {@link ServiceRequestContext} and
61   * a client-side {@link Request} has a {@link ClientRequestContext}.
62   */
63  public interface RequestContext extends AttributeMap {
64  
65      /**
66       * Returns the context of the {@link Request} that is being handled in the current thread.
67       *
68       * @throws IllegalStateException if the context is unavailable in the current thread
69       */
70      static <T extends RequestContext> T current() {
71          final T ctx = RequestContextThreadLocal.get();
72          if (ctx == null) {
73              throw new IllegalStateException(RequestContext.class.getSimpleName() + " unavailable");
74          }
75          return ctx;
76      }
77  
78      /**
79       * Maps the context of the {@link Request} that is being handled in the current thread.
80       *
81       * @param mapper the {@link Function} that maps the {@link RequestContext}
82       * @param defaultValueSupplier the {@link Supplier} that provides the value when the context is unavailable
83       *                             in the current thread. If {@code null}, the {@code null} will be returned
84       *                             when the context is unavailable in the current thread.
85       */
86      static <T> T mapCurrent(
87              Function<? super RequestContext, T> mapper, @Nullable Supplier<T> defaultValueSupplier) {
88  
89          final RequestContext ctx = RequestContextThreadLocal.get();
90          if (ctx != null) {
91              return mapper.apply(ctx);
92          }
93  
94          if (defaultValueSupplier != null) {
95              return defaultValueSupplier.get();
96          }
97  
98          return null;
99      }

101     /**
102      * Pushes the specified context to the thread-local stack. To pop the context from the stack, call
103      * {@link SafeCloseable#close()}, which can be done using a {@code try-with-resources} block:
104      * <pre>{@code
105      * try (SafeCloseable ignored = RequestContext.push(ctx)) {
106      *     ...
107      * }
108      * }</pre>
109      *
110      * <p>The callbacks added by {@link #onEnter(Consumer)} and {@link #onExit(Consumer)} will be invoked
111      * when the context is pushed to and removed from the thread-local stack respectively.
112      *
113      * <p>NOTE: In case of re-entrance, the callbacks will never run.
114      */
115     static SafeCloseable push(RequestContext ctx) {
116         return push(ctx, true);
117     }
118 
119     /**
120      * Pushes the specified context to the thread-local stack. To pop the context from the stack, call
121      * {@link SafeCloseable#close()}, which can be done using a {@code try-with-resources} block:
122      * <pre>{@code
123      * try (PushHandle ignored = RequestContext.push(ctx, true)) {
124      *     ...
125      * }
126      * }</pre>
127      *
128      * <p>NOTE: This method is only useful when it is undesirable to invoke the callbacks, such as replacing
129      *          the current context with another. Prefer {@link #push(RequestContext)} otherwise.
130      *
131      * @param runCallbacks if {@code true}, the callbacks added by {@link #onEnter(Consumer)} and
132      *                     {@link #onExit(Consumer)} will be invoked when the context is pushed to and
133      *                     removed from the thread-local stack respectively.
134      *                     If {@code false}, no callbacks will be executed.
135      *                     NOTE: In case of re-entrance, the callbacks will never run.
136      */
137     static SafeCloseable push(RequestContext ctx, boolean runCallbacks) {
138         final RequestContext oldCtx = RequestContextThreadLocal.getAndSet(ctx);
139         if (oldCtx == ctx) {
140             // Reentrance
141             return () -> { /* no-op */ };
142         }
143 
144         if (runCallbacks) {
145             if (oldCtx != null) {
146                 oldCtx.invokeOnChildCallbacks(ctx);
147                 ctx.invokeOnEnterCallbacks();
148                 return () -> {
149                     ctx.invokeOnExitCallbacks();
150                     RequestContextThreadLocal.set(oldCtx);
151                 };
152             } else {
153                 ctx.invokeOnEnterCallbacks();
154                 return () -> {
155                     ctx.invokeOnExitCallbacks();
156                     RequestContextThreadLocal.remove();
157                 };
158             }
159         } else {
160             if (oldCtx != null) {
161                 return () -> RequestContextThreadLocal.set(oldCtx);
162             } else {
163                 return RequestContextThreadLocal::remove;
164             }
165         }
166     }

431     /**
432      * Resolves the specified {@code promise} with the specified {@code result} so that the {@code promise} is
433      * marked as 'done'. If {@code promise} is done already, this method does the following:
434      * <ul>
435      *   <li>Log a warning about the failure, and</li>
436      *   <li>Release {@code result} if it is {@linkplain ReferenceCounted a reference-counted object},
437      *       such as {@link ByteBuf} and {@link FullHttpResponse}.</li>
438      * </ul>
439      * Note that a {@link Promise} can be done already even if you did not call this method in the following
440      * cases:
441      * <ul>
442      *   <li>Invocation timeout - The invocation associated with the {@link Promise} has been timed out.</li>
443      *   <li>User error - A service implementation called any of the following methods more than once:
444      *     <ul>
445      *       <li>{@link #resolvePromise(Promise, Object)}</li>
446      *       <li>{@link #rejectPromise(Promise, Throwable)}</li>
447      *       <li>{@link Promise#setSuccess(Object)}</li>
448      *       <li>{@link Promise#setFailure(Throwable)}</li>
449      *       <li>{@link Promise#cancel(boolean)}</li>
450      *     </ul>
451      *   </li>
452      * </ul>
453      *
454      * @deprecated Use {@link CompletableFuture} instead.
455      */
456     @Deprecated
457     default void resolvePromise(Promise<?> promise, Object result) {
458         @SuppressWarnings("unchecked")
459         final Promise<Object> castPromise = (Promise<Object>) promise;
460 
461         if (castPromise.trySuccess(result)) {
462             // Resolved successfully.
463             return;
464         }
465 
466         try {
467             if (!(promise.cause() instanceof TimeoutException)) {
468                 // Log resolve failure unless it is due to a timeout.
469                 LoggerFactory.getLogger(RequestContext.class).warn(
470                         "Failed to resolve a completed promise ({}) with {}", promise, result);
471             }
472         } finally {
473             ReferenceCountUtil.safeRelease(result);
474         }
475     }
476 
477     /**
478      * Rejects the specified {@code promise} with the specified {@code cause}. If {@code promise} is done
479      * already, this method logs a warning about the failure. Note that a {@link Promise} can be done already
480      * even if you did not call this method in the following cases:
481      * <ul>
482      *   <li>Invocation timeout - The invocation associated with the {@link Promise} has been timed out.</li>
483      *   <li>User error - A service implementation called any of the following methods more than once:
484      *     <ul>
485      *       <li>{@link #resolvePromise(Promise, Object)}</li>
486      *       <li>{@link #rejectPromise(Promise, Throwable)}</li>
487      *       <li>{@link Promise#setSuccess(Object)}</li>
488      *       <li>{@link Promise#setFailure(Throwable)}</li>
489      *       <li>{@link Promise#cancel(boolean)}</li>
490      *     </ul>
491      *   </li>
492      * </ul>
493      *
494      * @deprecated Use {@link CompletableFuture} instead.
495      */
496     @Deprecated
497     default void rejectPromise(Promise<?> promise, Throwable cause) {
498         if (promise.tryFailure(cause)) {
499             // Fulfilled successfully.
500             return;
501         }
502 
503         final Throwable firstCause = promise.cause();
504         if (firstCause instanceof TimeoutException) {
505             // Timed out already.
506             return;
507         }
508 
509         if (Exceptions.isExpected(cause)) {
510             // The exception that was thrown after firstCause (often a transport-layer exception)
511             // was a usual expected exception, not an error.
512             return;
513         }
514 
515         LoggerFactory.getLogger(RequestContext.class).warn(
516                 "Failed to reject a completed promise ({}) with {}", promise, cause, cause);
517     }
