31  package com.linecorp.armeria.common;
32  
33  /**
34   * HTTP request method.
35   */
36  public enum HttpMethod {
37  
38      /**
39       * The OPTIONS method represents a request for information about the communication options
40       * available on the request/response chain identified by the Request-URI. This method allows
41       * the client to determine the options and/or requirements associated with a resource, or the
42       * capabilities of a server, without implying a resource action or initiating a resource
43       * retrieval.
44       */
45      OPTIONS,
46  
47      /**
48       * The GET method means retrieve whatever information (in the form of an entity) is identified
49       * by the Request-URI.  If the Request-URI refers to a data-producing process, it is the
50       * produced data which shall be returned as the entity in the response and not the source text
51       * of the process, unless that text happens to be the output of the process.
52       */
53      GET,
54  
55      /**
56       * The HEAD method is identical to GET except that the server MUST NOT return a message-body
57       * in the response.
58       */
59      HEAD,
60  
61      /**
62       * The POST method is used to request that the origin server accept the entity enclosed in the
63       * request as a new subordinate of the resource identified by the Request-URI in the
64       * Request-Line.
65       */
66      POST,
67  
68      /**
69       * The PUT method requests that the enclosed entity be stored under the supplied Request-URI.
70       */
71      PUT,
72  
73      /**
74       * The PATCH method requests that a set of changes described in the
75       * request entity be applied to the resource identified by the Request-URI.
76       */
77      PATCH,
78  
79      /**
80       * The DELETE method requests that the origin server delete the resource identified by the
81       * Request-URI.
82       */
83      DELETE,
84  
85      /**
86       * The TRACE method is used to invoke a remote, application-layer loop- back of the request
87       * message.
88       */
89      TRACE,
90  
91      /**
92       * The CONNECT method is used for a proxy that can dynamically switch to being a tunnel.
93       */
94      CONNECT;
95  }
