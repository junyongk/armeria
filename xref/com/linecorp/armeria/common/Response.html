17  package com.linecorp.armeria.common;
18  
19  import java.util.concurrent.CompletableFuture;
20  import java.util.concurrent.CompletionStage;
21  
22  import com.linecorp.armeria.common.stream.StreamMessage;
23  
24  /**
25   * A response stream or a holder of the future result value.
26   * It has to be a {@link HttpResponse} or a {@link RpcResponse}.
27   */
28  public interface Response {
29  
30      /**
31       * Returns a {@link CompletableFuture} which completes when
32       * 1) the response stream has been closed (the {@link StreamMessage} has been closed) or
33       * 2) the result value is set (the {@link CompletionStage} has completed.)
34       */
35      default CompletableFuture<?> closeFuture() {
36          if (this instanceof StreamMessage) {
37              return ((StreamMessage<?>) this).closeFuture();
38          }
39  
40          if (this instanceof CompletionStage) {
41              return ((CompletionStage<?>) this).toCompletableFuture();
42          }
43  
44          throw new IllegalStateException(
45                  "response must be a " + StreamMessage.class.getSimpleName() + " or a " +
46                  CompletionStage.class.getSimpleName() + ": " + getClass().getName());
47      }
48  }
