17  package com.linecorp.armeria.common;
18  
19  import java.util.Iterator;
20  import java.util.concurrent.Callable;
21  import java.util.concurrent.TimeUnit;
22  
23  import io.netty.channel.Channel;
24  import io.netty.channel.ChannelFuture;
25  import io.netty.channel.ChannelPromise;
26  import io.netty.channel.EventLoop;
27  import io.netty.channel.EventLoopGroup;
28  import io.netty.util.concurrent.EventExecutor;
29  import io.netty.util.concurrent.Future;
30  import io.netty.util.concurrent.ProgressivePromise;
31  import io.netty.util.concurrent.Promise;
32  import io.netty.util.concurrent.ScheduledFuture;
33  
34  /**
35   * A delegating {@link EventLoop} that makes sure all submitted tasks are
36   * executed within the {@link RequestContext}.
37   */
38  final class RequestContextAwareEventLoop extends RequestContextAwareExecutorService implements EventLoop {
39  
40      RequestContextAwareEventLoop(RequestContext context, EventLoop delegate) {
41          super(context, delegate);
42      }
43  
44      @Override
45      protected EventLoop delegate() {
46          return (EventLoop) super.delegate();
47      }
48  
49      @Override
50      public EventLoop next() {
51          return this;
52      }
53  
54      @Override
55      public EventLoopGroup parent() {
56          return delegate().parent();
57      }
58  
59      @Override
60      public boolean inEventLoop() {
61          return delegate().inEventLoop();
62      }
63  
64      @Override
65      public boolean inEventLoop(Thread thread) {
66          return delegate().inEventLoop(thread);
67      }
68  
69      @Override
70      public <V> Promise<V> newPromise() {
71          return new RequestContextAwarePromise<>(context(), delegate().newPromise());
72      }
73  
74      @Override
75      public <V> ProgressivePromise<V> newProgressivePromise() {
76          return new RequestContextAwareProgressivePromise<>(context(), delegate().newProgressivePromise());
77      }
78  
79      @Override
80      public <V> Future<V> newSucceededFuture(V result) {
81          return new RequestContextAwareFuture<>(context(), delegate().newSucceededFuture(result));
82      }
83  
84      @Override
85      public <V> Future<V> newFailedFuture(Throwable cause) {
86          return new RequestContextAwareFuture<>(context(), delegate().newFailedFuture(cause));
87      }
88  
89      @Override
90      public boolean isShuttingDown() {
91          return delegate().isShuttingDown();
92      }
93  
94      @Override
95      public Future<?> shutdownGracefully() {
96          return delegate().shutdownGracefully();
97      }
98  
99      @Override
100     public Future<?> shutdownGracefully(long quietPeriod, long timeout,
101                                         TimeUnit unit) {
102         return delegate().shutdownGracefully(quietPeriod, timeout, unit);
103     }
104 
105     @Override
106     public Future<?> terminationFuture() {
107         return delegate().terminationFuture();
108     }
109 
110     @Override
111     public Iterator<EventExecutor> iterator() {
112         return delegate().iterator();
113     }
114 
115     @Override
116     public Future<?> submit(Runnable task) {
117         return (Future<?>) super.submit(task);
118     }
119 
120     @Override
121     public <T> Future<T> submit(Runnable task, T result) {
122         return (Future<T>) super.submit(task, result);
123     }
124 
125     @Override
126     public <T> Future<T> submit(Callable<T> task) {
127         return (Future<T>) super.submit(task);
128     }
129 
130     @Override
131     public ScheduledFuture<?> schedule(Runnable command, long delay, TimeUnit unit) {
132         return delegate().schedule(context().makeContextAware(command), delay, unit);
133     }
134 
135     @Override
136     public <V> ScheduledFuture<V> schedule(Callable<V> callable, long delay, TimeUnit unit) {
137         return delegate().schedule(context().makeContextAware(callable), delay, unit);
138     }
139 
140     @Override
141     public ScheduledFuture<?> scheduleAtFixedRate(
142             Runnable command, long initialDelay, long period, TimeUnit unit) {
143         return delegate().scheduleAtFixedRate(context().makeContextAware(command), initialDelay, period, unit);
144     }
145 
146     @Override
147     public ScheduledFuture<?> scheduleWithFixedDelay(
148             Runnable command, long initialDelay, long delay, TimeUnit unit) {
149         return delegate().scheduleWithFixedDelay(
150                 context().makeContextAware(command), initialDelay, delay, unit);
151     }
152 
153     @Override
154     public ChannelFuture register(Channel channel) {
155         return delegate().register(channel);
156     }
157 
158     @Override
159     public ChannelFuture register(ChannelPromise channelPromise) {
160         return delegate().register(channelPromise);
161     }
162 
163     @Override
164     public ChannelFuture register(Channel channel,
165                                   ChannelPromise channelPromise) {
166         return delegate().register(channel, channelPromise);
167     }
168 }
