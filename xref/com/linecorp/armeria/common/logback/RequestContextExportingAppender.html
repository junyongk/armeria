201     private void ensureNotStarted() {
202         if (exporter != null) {
203             throw new IllegalStateException("can't update the export list once started");
204         }
205     }

207     @Override
208     protected void append(ILoggingEvent eventObject) {
209         final RequestContext ctx = RequestContext.mapCurrent(Function.identity(), () -> null);
210         if (ctx != null) {
211             final State state = state(ctx);
212             final RequestLog log = ctx.log();
213             final Set<RequestLogAvailability> availabilities = log.availabilities();
214 
215             // Note: This equality check is extremely fast.
216             //       See RequestLogAvailabilitySet for more information.
217             if (!availabilities.equals(state.availabilities)) {
218                 state.availabilities = availabilities;
219                 export(state, ctx, log);
220             }
221 
222             final Map<String, String> originalMdcMap = eventObject.getMDCPropertyMap();
223             final Map<String, String> mdcMap;
224 
225             // Create a copy of 'state' to avoid the race between:
226             // - the delegate appenders who iterate over the MDC map and
227             // - this class who update 'state'.
228             if (!originalMdcMap.isEmpty()) {
229                 mdcMap = new UnionMap<>(state.clone(), originalMdcMap);
230             } else {
231                 mdcMap = state.clone();
232             }
233 
234             eventObject = new LoggingEventWrapper(eventObject, mdcMap);
235         }
236 
237         aai.appendLoopOnAppenders(eventObject);
238     }
239 
240     private static State state(RequestContext ctx) {
241         final Attribute<State> attr = ctx.attr(STATE);
242         final State state = attr.get();
243         if (state == null) {
244             State newState = new State();
245             State oldState = attr.setIfAbsent(newState);
246             if (oldState != null) {
247                 return oldState;
248             } else {
249                 return newState;
250             }
251         }
252         return state;
253     }

255     /**
256      * Exports the necessary properties to {@link MDC}. By default, this method exports all properties added
257      * to the export list via {@code add*()} calls and {@code <export />} tags. Override this method to export
258      * additional properties.
259      */
260     protected void export(Map<String, String> out, RequestContext ctx, @Nullable RequestLog log) {
261         exporter.export(out, ctx, log);
262     }

264     @Override
265     public void start() {
266         if (!aai.iteratorForAppenders().hasNext()) {
267             addWarn("No appender was attached to " + getClass().getSimpleName() + '.');
268         }
269         if (exporter == null) {
270             exporter = builder.build();
271         }
272         super.start();
273     }

275     @Override
276     public void stop() {
277         try {
278             aai.detachAndStopAllAppenders();
279         } finally {
280             super.stop();
281         }
282     }

325     private static final class LoggingEventWrapper implements ILoggingEvent {
326         private final ILoggingEvent event;
327         private final Map<String, String> mdcPropertyMap;
328         private final LoggerContextVO vo;
329 
330         LoggingEventWrapper(ILoggingEvent event, Map<String, String> mdcPropertyMap) {
331             this.event = event;
332             this.mdcPropertyMap = mdcPropertyMap;
333 
334             final LoggerContextVO oldVo = event.getLoggerContextVO();
335             if (oldVo != null) {
336                 vo = new LoggerContextVO(oldVo.getName(), mdcPropertyMap, oldVo.getBirthTime());
337             } else {
338                 vo = null;
339             }
340         }
