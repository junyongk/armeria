17  package com.linecorp.armeria.common.stream;
18  
19  import java.util.concurrent.CompletableFuture;
20  import java.util.function.Supplier;
21  
22  import org.reactivestreams.Subscriber;
23  
24  /**
25   * Produces the objects to be published by a {@link StreamMessage}.
26   *
27   * @param <T> the type of the stream element
28   */
29  public interface StreamWriter<T> {
30  
31      /**
32       * Returns {@code true} if the {@link StreamMessage} is open.
33       */
34      boolean isOpen();
35  
36      /**
37       * Writes the specified object to the {@link StreamMessage}. The written object will be transferred to the
38       * {@link Subscriber}.
39       *
40       * @return {@code true} if the specified object has been scheduled for publication. {@code false} if the
41       *         stream has been closed already.
42       *
43       * @throws IllegalArgumentException if the publication of the specified object has been rejected
44       */
45      boolean write(T o);
46  
47      /**
48       * Writes the specified object {@link Supplier} to the {@link StreamMessage}. The object provided by the
49       * {@link Supplier} will be transferred to the {@link Subscriber}.
50       *
51       * @return {@code true} if the specified object has been scheduled for publication. {@code false} if the
52       *         stream has been closed already.
53       */
54      boolean write(Supplier<? extends T> o);
55  
56      /**
57       * Performs the specified {@code task} when there's enough demans from the {@link Subscriber}.
58       *
59       * @return the future that completes successfully when the {@code task} finishes or
60       *         exceptionally when the {@link StreamMessage} is closed unexpectedly.
61       */
62      CompletableFuture<Void> onDemand(Runnable task);
63  
64      /**
65       * Closes the {@link StreamMessage} successfully. {@link Subscriber#onComplete()} will be invoked to
66       * signal that the {@link Subscriber} has consumed the stream completely.
67       */
68      void close();
69  
70      /**
71       * Closes the {@link StreamMessage} exceptionally. {@link Subscriber#onError(Throwable)} will be invoked to
72       * signal that the {@link Subscriber} did not consume the stream completely.
73       */
74      void close(Throwable cause);
75  }
