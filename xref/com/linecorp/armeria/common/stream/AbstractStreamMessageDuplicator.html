44  /**
45   * Allows subscribing to a {@link StreamMessage} multiple times by duplicating the stream.
46   * <p>
47   * Only one subscriber can subscribe other stream messages such as {@link DefaultStreamMessage},
48   * {@link DeferredStreamMessage}, etc.
49   * This factory is wrapping one of those {@link StreamMessage}s and spawns duplicated stream messages
50   * which are created using {@link AbstractStreamMessageDuplicator#duplicateStream()} and subscribed
51   * by subscribers one by one.
52   * </p><p>
53   * The published elements can be shared across {@link Subscriber}s, so do not manipulate the
54   * data unless you copy them. Only one case does not share the elements which is when you
55   * {@link StreamMessage#subscribe(Subscriber, boolean)} with the {@code withPooledObjects}
56   * as {@code false} while the elements is one of two pooled object classes which are {@link ByteBufHolder}
57   * and {@link ByteBuf}.
58   * </p><p>
59   * This factory has to be closed by {@link AbstractStreamMessageDuplicator#close()} when
60   * you do not need the contents anymore, otherwise memory leak might happen.
61   * </p>
62   * @param <T> the type of elements
63   * @param <U> the type of the publisher and duplicated stream messages
64   */
65  public abstract class AbstractStreamMessageDuplicator<T, U extends StreamMessage<? extends T>>
66          implements SafeCloseable {
67  
68      private final StreamMessageProcessor<T> processor;
69  
70      /**
71       * Creates a new instance wrapping a {@code publisher} and publishing to multiple subscribers.
72       * @param publisher the publisher who will publish data to subscribers
73       */
74      protected AbstractStreamMessageDuplicator(U publisher) {
75          requireNonNull(publisher, "publisher");
76          processor = new StreamMessageProcessor<>(publisher);
77      }
78  
79      /**
80       * Creates a new {@link U} instance that publishes data from the {@code publisher} you create
81       * this factory with.
82       */
83      public U duplicateStream() {
84          if (!processor.isOpen()) {
85              throw new IllegalStateException("This factory has been closed already.");
86          }
87          return doDuplicateStream(new ChildStreamMessage<>(processor));
88      }
89  
90      /**
91       * Creates a new {@link U} instance that wraps {@link ChildStreamMessage} and forwards its method
92       * invocations to it.
93       * @param delegate {@link ChildStreamMessage}
94       */
95      protected abstract U doDuplicateStream(StreamMessage<T> delegate);
96  
97      /**
98       * Closes this factory and stream messages who are invoked by
99       * {@link AbstractStreamMessageDuplicator#duplicateStream()}.
100      * Also, clean up the data published from {@code publisher}.
101      */
102     @Override
103     public void close() {
104         processor.close();
105     }
106 
107     @VisibleForTesting
108     static class StreamMessageProcessor<T> implements Subscriber<T> {
109 
110         private enum State {
111             /**
112              * The initial state. Will enter {@link #CLOSED}.
113              */
114             OPEN,
115             /**
116               * {@link AbstractStreamMessageDuplicator#close()} has been called.
117              */
118             CLOSED
119         }
120 
121         @SuppressWarnings("rawtypes")
122         private static final AtomicReferenceFieldUpdater<StreamMessageProcessor, State> stateUpdater =
123                 AtomicReferenceFieldUpdater.newUpdater(StreamMessageProcessor.class, State.class, "state");
124 
125         @SuppressWarnings("rawtypes")
126         private static final AtomicLongFieldUpdater<StreamMessageProcessor> requestedDemandUpdater =
127                 AtomicLongFieldUpdater.newUpdater(StreamMessageProcessor.class, "requestedDemand");
128 
129         private final Set<DownstreamSubscription> downstreamSubscriptions =
130                 Collections.newSetFromMap(new ConcurrentHashMap<>());
131 
132         private final List<T> contentList = new ArrayList<>();
133 
134         @SuppressWarnings("unused")
135         private volatile long requestedDemand;
136         private volatile Subscription upstreamSubscription;
137         private volatile boolean upstreamCompleted;
138         private volatile int upstreamOffset;
139         private volatile State state = State.OPEN;
140 
141         private Throwable upstreamCause;
142 
143         StreamMessageProcessor(StreamMessage<? extends T> upstream) {
144             upstream.subscribe(this, true);
145             notifyDownstreamWhenUpstreamCompleted(upstream);
146         }
147 
148         private void notifyDownstreamWhenUpstreamCompleted(StreamMessage<? extends T> upstream) {
149             upstream.closeFuture().whenComplete((unused, cause) -> {
150                 upstreamCompleted = true;
151                 if (cause != null) {
152                     upstreamCause = cause;
153                 }
154                 notifyDownstreams();
155             });
156         }
157 
158         @Override
159         public void onSubscribe(Subscription s) {
160             upstreamSubscription = s;
161             downstreamSubscriptions.forEach(downstream -> {
162                 if (downstream.setSubscribed()) {
163                     if (downstream.executor() != null) {
164                         downstream.executor().execute(() -> downstream.subscriber().onSubscribe(downstream));
165                     } else {
166                         downstream.subscriber().onSubscribe(downstream);
167                     }
168                 }
169             });
170             if (state == State.CLOSED) {
171                 s.cancel();
172             }
173         }
174 
175         @Override
176         public void onNext(T t) {
177             if (upstreamOffset >= Integer.MAX_VALUE) {
178                 //TODO(minwoox) limit by the size of data not by the size of contentList.
179                 upstreamSubscription.cancel();
180                 throw new IllegalStateException("Published numbers of data is greater than Integer.MAX_VALUE");
181             }
182             contentList.add(t);
183             upstreamOffset++;
184             notifyDownstreams();
185         }
186 
187         /**
188          * Handled by {@link #notifyDownstreamWhenUpstreamCompleted(StreamMessage)} instead.
189          */
190         @Override
191         public void onError(Throwable t) {}
192 
193         /**
194          * Handled by {@link #notifyDownstreamWhenUpstreamCompleted(StreamMessage)} instead.
195          */
196         @Override
197         public void onComplete() {}
198 
199         void subscribe(DownstreamSubscription subscription) {
200             if (state == State.OPEN) {
201                 downstreamSubscriptions.add(subscription);
202                 if (state == State.CLOSED) {
203                     downstreamSubscriptions.remove(subscription);
204                     throw new IllegalStateException("This factory has been closed already.");
205                 }
206             } else {
207                 throw new IllegalStateException("This factory has been closed already.");
208             }
209 
210             final Subscriber<Object> subscriber = subscription.subscriber();
211             if (upstreamSubscription != null && subscription.setSubscribed()) {
212                 final Executor executor = subscription.executor();
213                 if (executor != null) {
214                     executor.execute(() -> subscriber.onSubscribe(subscription));
215                 } else {
216                     subscriber.onSubscribe(subscription);
217                 }
218             }
219         }
220 
221         void requestDemand(long cumulativeDemand) {
222             for (;;) {
223                 if (cumulativeDemand <= requestedDemand) {
224                     return;
225                 }
226                 long currentRequested = requestedDemand;
227                 if (requestedDemandUpdater.compareAndSet(this, currentRequested, cumulativeDemand)) {
228                     upstreamSubscription.request(cumulativeDemand - currentRequested);
229                     return;
230                 }
231             }
232         }
233 
234         void notifyDownstreams() {
235             if (downstreamSubscriptions.isEmpty()) {
236                 return;
237             }
238             downstreamSubscriptions.forEach(downstream -> {
239                 final Executor executor = downstream.executor();
240                 if (executor != null) {
241                     executor.execute(() -> notifyDownstream(downstream));
242                 } else {
243                     notifyDownstream(downstream);
244                 }
245             });
246         }
247 
248         void notifyDownstream(DownstreamSubscription downstream) {
249             for (;;) {
250                 int offsetOfSubscriber = downstream.offset;
251                 int upstreamOffset = this.upstreamOffset;
252                 if (upstreamCompleted && offsetOfSubscriber == upstreamOffset) {
253                     if (DownstreamSubscription.notifyingUpdater.compareAndSet(downstream, 0, 1)) {
254                         completeDownstream(downstream, upstreamCause);
255                         // Don't have to bring notifying back to 0 because it's completed.
256                     }
257                     return;
258                 }
259 
260                 if (downstream.cancelled) {
261                     if (DownstreamSubscription.notifyingUpdater.compareAndSet(downstream, 0, 1)) {
262                         completeDownstream(downstream, CancelledSubscriptionException.get());
263                         // Don't have to bring notifying back to 0 because it's completed.
264                     }
265                     return;
266                 }
267 
268                 if (offsetOfSubscriber == upstreamOffset) {
269                     break;
270                 }
271 
272                 if (!publishData(downstream)) {
273                     break;
274                 }
275             }
276         }
277 
278         private void completeDownstream(DownstreamSubscription downstream, Throwable cause) {
279             downstreamSubscriptions.remove(downstream);
280             if (cause == null) {
281                 try {
282                     downstream.subscriber().onComplete();
283                 } finally {
284                     @SuppressWarnings("unchecked")
285                     final CompletableFuture<Void> f =
286                             (CompletableFuture<Void>) downstream.streamMessage().closeFuture();
287                     f.complete(null);
288                 }
289             } else {
290                 try {
291                     if (!(cause instanceof CancelledSubscriptionException)) {
292                         downstream.subscriber().onError(cause);
293                     }
294                 } finally {
295                     downstream.streamMessage().closeFuture().completeExceptionally(cause);
296                 }
297             }
298         }
299 
300         private boolean publishData(DownstreamSubscription downstream) {
301             for (;;) {
302                 final long demand = downstream.demand;
303                 if (demand == 0) {
304                     break;
305                 }
306 
307                 if (demand == Long.MAX_VALUE ||
308                     DownstreamSubscription.demandUpdater.compareAndSet(downstream, demand, demand - 1)) {
309                     final Subscriber<Object> subscriber = downstream.subscriber();
310                     if (DownstreamSubscription.notifyingUpdater.compareAndSet(downstream, 0, 1)) {
311                         T o = contentList.get(downstream.offset++);
312                         ReferenceCountUtil.touch(o);
313                         if (downstream.withPooledObjects()) {
314                             if (o instanceof ByteBufHolder) {
315                                 o = retainedDuplicate((ByteBufHolder) o);
316                             } else if (o instanceof ByteBuf) {
317                                 o = retainedDuplicate((ByteBuf) o);
318                             }
319                         } else {
320                             if (o instanceof ByteBufHolder) {
321                                 o = copy((ByteBufHolder) o);
322                             } else if (o instanceof ByteBuf) {
323                                 o = copy((ByteBuf) o);
324                             }
325                         }
326 
327                         subscriber.onNext(o);
328                         downstream.notifying = 0;
329                         return true;
330                     } else {
331                         if (demand != Long.MAX_VALUE) {
332                             incrementDemand(downstream);
333                         }
334                         return false;
335                     }
336                 }
337             }
338             return false;
339         }
340 
341         @SuppressWarnings("unchecked")
342         private T retainedDuplicate(ByteBufHolder o) {
343             return (T) o.replace(o.content().retainedDuplicate());
344         }
345 
346         @SuppressWarnings("unchecked")
347         private T retainedDuplicate(ByteBuf o) {
348             return (T) o.retainedDuplicate();
349         }
350 
351         @SuppressWarnings("unchecked")
352         private T copy(ByteBufHolder o) {
353            return (T) o.replace(Unpooled.copiedBuffer(o.content()));
354         }
355 
356         @SuppressWarnings("unchecked")
357         private T copy(ByteBuf o) {
358             return (T) Unpooled.copiedBuffer(o);
359         }
360 
361         private void incrementDemand(DownstreamSubscription downstream) {
362             for (;;) {
363                 final long oldDemand = downstream.demand;
364                 if (DownstreamSubscription.demandUpdater.compareAndSet(
365                         downstream, oldDemand, oldDemand + 1)) {
366                     break;
367                 }
368             }
369         }
370 
371         boolean isOpen() {
372             return state == State.OPEN;
373         }
374 
375         void close() {
376             if (stateUpdater.compareAndSet(this, State.OPEN, State.CLOSED)) {
377                 state = State.CLOSED;
378                 final Subscription upstream = this.upstreamSubscription;
379                 if (upstream != null) {
380                     upstream.cancel();
381                 }
382                 cleanup();
383             }
384         }
385 
386         private void cleanup() {
387             final List<CompletableFuture<Void>> closeFutures = new ArrayList<>();
388             downstreamSubscriptions.forEach(s -> {
389                 @SuppressWarnings("unchecked")
390                 final CompletableFuture<Void> future = s.streamMessage().closeFuture();
391                 closeFutures.add(future);
392             });
393             final CompletableFuture<Void> allDoneFuture =
394                     CompletableFuture.allOf(closeFutures.toArray(new CompletableFuture[closeFutures.size()]));
395             allDoneFuture.whenComplete((unused, cause) -> {
396                 contentList.forEach(ReferenceCountUtil::safeRelease);
397                 contentList.clear();
398             });
399         }
400     }

402     private static class ChildStreamMessage<T> implements StreamMessage<T> {
403 
404         @SuppressWarnings("rawtypes")
405         private static final AtomicReferenceFieldUpdater<ChildStreamMessage, DownstreamSubscription>
406                 subscriptionUpdater = AtomicReferenceFieldUpdater.newUpdater(
407                 ChildStreamMessage.class, DownstreamSubscription.class, "subscription");
408 
409         private StreamMessageProcessor processor;
410         @SuppressWarnings("unused")
411         private volatile DownstreamSubscription subscription;
412 
413         private CompletableFuture<Void> closeFuture = new CompletableFuture<>();
414 
415         ChildStreamMessage(StreamMessageProcessor<? extends T> processor) {
416             this.processor = processor;
417         }
418 
419         @Override
420         public boolean isOpen() {
421             return !closeFuture.isDone();
422         }
423 
424         @Override
425         public boolean isEmpty() {
426             return !isOpen() && processor.upstreamOffset == 0;
427         }
428 
429         @Override
430         public CompletableFuture<Void> closeFuture() {
431             return closeFuture;
432         }
433 
434         @Override
435         public void subscribe(Subscriber<? super T> subscriber) {
436             requireNonNull(subscriber, "subscriber");
437             subscribe(subscriber, false);
438         }
439 
440         @Override
441         public void subscribe(Subscriber<? super T> subscriber, boolean withPooledObjects) {
442             requireNonNull(subscriber, "subscriber");
443             subscribe0(subscriber, null, withPooledObjects);
444         }
445 
446         @Override
447         public void subscribe(Subscriber<? super T> subscriber, Executor executor) {
448             requireNonNull(subscriber, "subscriber");
449             requireNonNull(executor, "executor");
450             subscribe(subscriber, executor, false);
451         }
452 
453         @Override
454         public void subscribe(Subscriber<? super T> subscriber, Executor executor,
455                               boolean withPooledObjects) {
456             requireNonNull(subscriber, "subscriber");
457             requireNonNull(executor, "executor");
458             subscribe0(subscriber, executor, withPooledObjects);
459         }
460 
461         private void subscribe0(Subscriber<? super T> subscriber, Executor executor,
462                                 boolean withPooledObjects) {
463             final DownstreamSubscription subscription = new DownstreamSubscription(
464                     this, subscriber, processor, executor, withPooledObjects);
465             if (!subscriptionUpdater.compareAndSet(this, null, subscription)) {
466                 throw new IllegalStateException(
467                         "Subscribed by other subscriber already: " + this.subscription.subscriber());
468             }
469             processor.subscribe(subscription);
470         }
471 
472         @Override
473         public void abort() {
474             final DownstreamSubscription currentSubscription = subscription;
475             if (currentSubscription != null) {
476                 currentSubscription.cancel();
477                 return;
478             }
479 
480             final DownstreamSubscription newSubscription = new DownstreamSubscription(
481                     this, AbortingSubscriber.INSTANCE, processor, null, false);
482             if (subscriptionUpdater.compareAndSet(this, null, newSubscription)) {
483                 processor.subscribe(newSubscription);
484             } else {
485                 subscription.cancel();
486             }
487         }
488     }

490     @VisibleForTesting
491     static class DownstreamSubscription implements Subscription {
492 
493         @SuppressWarnings("rawtypes")
494         private static final AtomicIntegerFieldUpdater<DownstreamSubscription> subscribedUpdater =
495                 AtomicIntegerFieldUpdater.newUpdater(DownstreamSubscription.class, "subscribed");
496 
497         @SuppressWarnings("rawtypes")
498         static final AtomicLongFieldUpdater<DownstreamSubscription> demandUpdater =
499                 AtomicLongFieldUpdater.newUpdater(DownstreamSubscription.class, "demand");
500 
501         @SuppressWarnings("rawtypes")
502         static final AtomicIntegerFieldUpdater<DownstreamSubscription> notifyingUpdater =
503                 AtomicIntegerFieldUpdater.newUpdater(DownstreamSubscription.class, "notifying");
504 
505         private final ChildStreamMessage streamMessage;
506         private final Subscriber<Object> subscriber;
507         private final StreamMessageProcessor processor;
508         private final Executor executor;
509         private final boolean withPooledObjects;
510 
511         @SuppressWarnings("unused")
512         private volatile long demand;
513 
514         private volatile boolean cancelled;
515 
516         @SuppressWarnings("unused")
517         private volatile int subscribed; // 0: not on subscribe, 1: on subscribe
518 
519         private volatile int offset;
520 
521         volatile int notifying;
522 
523         private long cumulativeDemand;
524 
525         @SuppressWarnings("unchecked")
526         DownstreamSubscription(ChildStreamMessage streamMessage,
527                                Subscriber<?> subscriber, StreamMessageProcessor processor,
528                                Executor executor, boolean withPooledObjects) {
529             this.streamMessage = streamMessage;
530             this.subscriber = (Subscriber<Object>) subscriber;
531             this.processor = processor;
532             this.executor = executor;
533             this.withPooledObjects = withPooledObjects;
534         }
535 
536         StreamMessage streamMessage() {
537             return streamMessage;
538         }
539 
540         Subscriber<Object> subscriber() {
541             return subscriber;
542         }
543 
544         Executor executor() {
545             return executor;
546         }
547 
548         boolean withPooledObjects() {
549             return withPooledObjects;
550         }
551 
552         @Override
553         public void request(long n) {
554             if (n <= 0) {
555                 throw new IllegalArgumentException("n: " + n + " (expected: > 0)");
556             }
557 
558             accumulateDemand(n);
559             processor.requestDemand(cumulativeDemand);
560 
561             for (;;) {
562                 final long oldDemand = demand;
563                 final long newDemand;
564                 if (oldDemand >= Long.MAX_VALUE - n) {
565                     newDemand = Long.MAX_VALUE;
566                 } else {
567                     newDemand = oldDemand + n;
568                 }
569 
570                 if (demandUpdater.compareAndSet(this, oldDemand, newDemand)) {
571                     if (oldDemand == 0) {
572                         processor.notifyDownstream(this);
573                     }
574                     break;
575                 }
576             }
577         }
578 
579         private void accumulateDemand(long n) {
580             if (n == Long.MAX_VALUE || Long.MAX_VALUE - n >= cumulativeDemand) {
581                 cumulativeDemand = Long.MAX_VALUE;
582             } else {
583                 cumulativeDemand += n;
584             }
585         }
586 
587         @Override
588         public void cancel() {
589             if (!cancelled) {
590                 cancelled = true;
591             }
592             if (executor != null) {
593                 executor.execute(() -> processor.notifyDownstream(this));
594             } else {
595                 processor.notifyDownstream(this);
596             }
597         }
598 
599         boolean setSubscribed() {
600             return subscribedUpdater.compareAndSet(this, 0, 1);
601         }
602     }
