17  package com.linecorp.armeria.common.stream;
18  
19  import java.util.concurrent.CompletableFuture;
20  import java.util.concurrent.Executor;
21  
22  import org.reactivestreams.Publisher;
23  import org.reactivestreams.Subscriber;
24  import org.reactivestreams.Subscription;
25  
26  import io.netty.buffer.ByteBuf;
27  import io.netty.buffer.ByteBufHolder;
28  import io.netty.util.ReferenceCounted;
29  
30  /**
31   * A variant of <a href="http://www.reactive-streams.org/">Reactive Streams</a> {@link Publisher}, which allows
32   * only one {@link Subscriber}. Unlike a usual {@link Publisher}, a {@link StreamMessage} can stream itself
33   * only once.  It has the following additional operations on top of what the Reactive Streams API provides:
34   * <ul>
35   *   <li>{@link #isOpen()}</li>
36   *   <li>{@link #isEmpty()}</li>
37   *   <li>{@link #closeFuture()}</li>
38   *   <li>{@link #abort()}</li>
39   * </ul>
40   *
41   * <h3>When is a {@link StreamMessage} open?</h3>
42   *
43   * <p>A {@link StreamMessage} is open since its instantiation until:
44   * <ul>
45   *   <li>the {@link Subscriber} consumes all elements and {@link Subscriber#onComplete()} is invoked,</li>
46   *   <li>an error occurred and {@link Subscriber#onError(Throwable)} is invoked,</li>
47   *   <li>the {@link Subscription} has been cancelled or</li>
48   *   <li>{@link #abort()} has been requested.</li>
49   * </ul>
50   *
51   * <h3>Getting notified when a {@link StreamMessage} is closed</h3>
52   *
53   * <p>{@link Subscriber#onComplete()} and {@link Subscriber#onError(Throwable)} will not be invoked when its
54   * {@link Subscription} has been {@linkplain Subscription#cancel() cancelled} or {@linkplain #abort() aborted}.
55   * Use {@link #closeFuture()} if you need to get notified when a {@link StreamMessage} is closed regardless of
56   * whether it's been cancelled, aborted or closed.
57   *
58   * <h3>Publication and Consumption of {@link ReferenceCounted} objects</h3>
59   *
60   * <p>{@link StreamMessage} will reject the publication request of a {@link ReferenceCounted} object except
61   * {@link ByteBuf} and {@link ByteBufHolder}.
62   *
63   * <p>{@link StreamMessage} will discard the publication request of a {@link ByteBuf} or a {@link ByteBufHolder}
64   * silently and release it automatically when the publication is attempted after the stream is closed.
65   *
66   * <p>For {@link ByteBuf} and {@link ByteBufHolder}, {@link StreamMessage} will convert them into their
67   * respective unpooled versions that never leak, so that the {@link Subscriber} does not need to worry about
68   * leaks.
69   *
70   * <p>If a {@link Subscriber} does not want a {@link StreamMessage} to make a copy of a {@link ByteBufHolder},
71   * use {@link #subscribe(Subscriber, boolean)} or {@link #subscribe(Subscriber, Executor, boolean)} and
72   * specify {@code true} for {@code withPooledObjects}. Note that the {@link Subscriber} is responsible for
73   * releasing the objects given with {@link Subscriber#onNext(Object)}.
74   *
75   * @param <T> the type of element signaled
76   */
77  public interface StreamMessage<T> extends Publisher<T> {
78      /**
79       * Returns {@code true} if this publisher is not closed yet.
80       */
81      boolean isOpen();
82  
83      /**
84       * Returns {@code true} if this stream has been closed and did not publish any elements.
85       * Note that this method will not return {@code true} when the stream is open even if it has not
86       * published anything so far, because it may publish something later.
87       */
88      boolean isEmpty();
89  
90      /**
91       * Returns a {@link CompletableFuture} that completes when this publisher is complete,
92       * either successfully or exceptionally.
93       */
94      CompletableFuture<Void> closeFuture();
95  
96      /**
97       * Requests to start streaming data to the specified {@link Subscriber}.
98       *
99       * @throws IllegalStateException if there is a {@link Subscriber} who subscribed to this stream already
100      */
101     @Override
102     void subscribe(Subscriber<? super T> s);
103 
104     /**
105      * Requests to start streaming data to the specified {@link Subscriber}.
106      *
107      * @param withPooledObjects if {@code true}, receives the pooled {@link ByteBuf} and {@link ByteBufHolder}
108      *                          as is, without making a copy. If you don't know what this means, use
109      *                          {@link StreamMessage#subscribe(Subscriber)}.
110      * @throws IllegalStateException if there is a {@link Subscriber} who subscribed to this stream already
111      */
112     void subscribe(Subscriber<? super T> s, boolean withPooledObjects);
113 
114     /**
115      * Requests to start streaming data, invoking the specified {@link Subscriber} from the specified
116      * {@link Executor}.
117      *
118      * @throws IllegalStateException if there is a {@link Subscriber} who subscribed to this stream already
119      */
120     void subscribe(Subscriber<? super T> s, Executor executor);
121 
122     /**
123      * Requests to start streaming data, invoking the specified {@link Subscriber} from the specified
124      * {@link Executor}.
125      *
126      * @param withPooledObjects if {@code true}, receives the pooled {@link ByteBuf} and {@link ByteBufHolder}
127      *                          as is, without making a copy. If you don't know what this means, use
128      *                          {@link StreamMessage#subscribe(Subscriber)}.
129      * @throws IllegalStateException if there is a {@link Subscriber} who subscribed to this stream already
130      */
131     void subscribe(Subscriber<? super T> s, Executor executor, boolean withPooledObjects);
132 
133     /**
134      * Cancels the {@link Subscription} if any and closes this publisher.
135      */
136     void abort();
137 }
