17  package com.linecorp.armeria.client;
18  
19  import java.net.InetSocketAddress;
20  import java.util.concurrent.ScheduledFuture;
21  import java.util.concurrent.TimeUnit;
22  
23  import org.reactivestreams.Subscriber;
24  import org.reactivestreams.Subscription;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  import com.linecorp.armeria.client.HttpResponseDecoder.HttpResponseWrapper;
29  import com.linecorp.armeria.common.AbstractRequestContext;
30  import com.linecorp.armeria.common.ClosedSessionException;
31  import com.linecorp.armeria.common.HttpData;
32  import com.linecorp.armeria.common.HttpHeaders;
33  import com.linecorp.armeria.common.HttpObject;
34  import com.linecorp.armeria.common.HttpRequest;
35  import com.linecorp.armeria.common.logging.RequestLogBuilder;
36  import com.linecorp.armeria.common.stream.ClosedPublisherException;
37  import com.linecorp.armeria.common.util.Exceptions;
38  import com.linecorp.armeria.internal.HttpObjectEncoder;
39  
40  import io.netty.channel.Channel;
41  import io.netty.channel.ChannelFuture;
42  import io.netty.channel.ChannelFutureListener;
43  import io.netty.channel.ChannelHandlerContext;
44  import io.netty.channel.EventLoop;
45  import io.netty.handler.codec.http2.Http2Error;
46  
47  final class HttpRequestSubscriber implements Subscriber<HttpObject>, ChannelFutureListener {
48  
49      private static final Logger logger = LoggerFactory.getLogger(HttpRequestSubscriber.class);
50  
51      enum State {
52          NEEDS_TO_WRITE_FIRST_HEADER,
53          NEEDS_DATA_OR_TRAILING_HEADERS,
54          DONE
55      }
56  
57      private final ChannelHandlerContext ctx;
58      private final HttpObjectEncoder encoder;
59      private final int id;
60      private final HttpRequest request;
61      private final HttpResponseWrapper response;
62      private final ClientRequestContext reqCtx;
63      private final RequestLogBuilder logBuilder;
64      private final long timeoutMillis;
65      private Subscription subscription;
66      private ScheduledFuture<?> timeoutFuture;
67      private State state = State.NEEDS_TO_WRITE_FIRST_HEADER;
68  
69      HttpRequestSubscriber(Channel ch, HttpObjectEncoder encoder,
70                            int id, HttpRequest request, HttpResponseWrapper response,
71                            ClientRequestContext reqCtx, long timeoutMillis) {
72  
73          ctx = ch.pipeline().lastContext();
74  
75          this.encoder = encoder;
76          this.id = id;
77          this.request = request;
78          this.response = response;
79          this.reqCtx = reqCtx;
80          logBuilder = reqCtx.logBuilder();
81          this.timeoutMillis = timeoutMillis;
82      }
83  
84      /**
85       * Invoked on each write of an {@link HttpObject}.
86       */
87      @Override
88      public void operationComplete(ChannelFuture future) throws Exception {
89          if (future.isSuccess()) {
90              if (state == State.DONE) {
91                  // Successfully sent the request; schedule the response timeout.
92                  response.scheduleTimeout(ctx);
93              } else {
94                  subscription.request(1);
95              }
96              return;
97          }
98  
99          fail(future.cause());
100 
101         final Throwable cause = future.cause();
102         if (!(cause instanceof ClosedPublisherException)) {
103             final Channel ch = future.channel();
104             Exceptions.logIfUnexpected(logger, ch, HttpSession.get(ch).protocol(), cause);
105             ch.close();
106         }
107     }
108 
109     @Override
110     public void onSubscribe(Subscription subscription) {
111         assert this.subscription == null;
112         this.subscription = subscription;
113 
114         final EventLoop eventLoop = ctx.channel().eventLoop();
115         if (timeoutMillis > 0) {
116             timeoutFuture = eventLoop.schedule(
117                     () -> {
118                         if (state == State.NEEDS_TO_WRITE_FIRST_HEADER) {
119                             if (reqCtx instanceof AbstractRequestContext) {
120                                 ((AbstractRequestContext) reqCtx).setTimedOut();
121                             }
122                             failAndRespond(WriteTimeoutException.get());
123                         }
124                     },
125                     timeoutMillis, TimeUnit.MILLISECONDS);
126         }
127 
128         // NB: This must be invoked at the end of this method because otherwise the callback methods in this
129         //     class can be called before the member fields (subscription and timeoutFuture) are initialized.
130         //     It is because the successful write of the first headers will trigger subscription.request(1).
131         eventLoop.execute(this::writeFirstHeader);
132     }
133 
134     private void writeFirstHeader() {
135         final Channel ch = ctx.channel();
136         final HttpSession session = HttpSession.get(ch);
137         if (!session.isActive()) {
138             failAndRespond(ClosedSessionException.get());
139             return;
140         }
141 
142         final HttpHeaders firstHeaders = request.headers();
143         String host = firstHeaders.authority();
144         if (host == null) {
145             host = ((InetSocketAddress) ch.remoteAddress()).getHostString();
146         } else {
147             final int colonIdx = host.lastIndexOf(':');
148             if (colonIdx > 0) {
149                 host = host.substring(0, colonIdx);
150             }
151         }
152 
153         logBuilder.startRequest(ch, session.protocol(), host);
154         logBuilder.requestHeaders(firstHeaders);
155 
156         if (request.isEmpty()) {
157             setDone();
158             write0(firstHeaders, true, true);
159         } else {
160             write0(firstHeaders, false, true);
161         }
162         state = State.NEEDS_DATA_OR_TRAILING_HEADERS;
163         cancelTimeout();
164     }
165 
166     @Override
167     public void onNext(HttpObject o) {
168         if (!(o instanceof HttpData) && !(o instanceof HttpHeaders)) {
169             throw newIllegalStateException(
170                     "published an HttpObject that's neither Http2Headers nor Http2Data: " + o);
171         }
172 
173         boolean endOfStream = o.isEndOfStream();
174         switch (state) {
175             case NEEDS_DATA_OR_TRAILING_HEADERS: {
176                 if (o instanceof HttpHeaders) {
177                     final HttpHeaders trailingHeaders = (HttpHeaders) o;
178                     if (trailingHeaders.status() != null) {
179                         throw newIllegalStateException("published a trailing HttpHeaders with status: " + o);
180                     }
181                     // Trailing headers always end the stream even if not explicitly set.
182                     endOfStream = true;
183                 }
184                 break;
185             }
186             case DONE:
187                 return;
188         }
189 
190         write(o, endOfStream, true);
191     }
192 
193     @Override
194     public void onError(Throwable cause) {
195         failAndRespond(cause);
196     }
197 
198     @Override
199     public void onComplete() {
200         if (!cancelTimeout()) {
201             return;
202         }
203 
204         if (state != State.DONE) {
205             write(HttpData.EMPTY_DATA, true, true);
206         }
207     }
208 
209     private void write(HttpObject o, boolean endOfStream, boolean flush) {
210         if (state == State.DONE) {
211             throw newIllegalStateException(
212                     "a request publisher published an HttpObject after a trailing HttpHeaders: " + o);
213         }
214 
215         final Channel ch = ctx.channel();
216         if (!ch.isActive()) {
217             fail(ClosedSessionException.get());
218             return;
219         }
220 
221         if (endOfStream) {
222             setDone();
223         }
224 
225         ch.eventLoop().execute(() -> write0(o, endOfStream, flush));
226     }
227 
228     private void write0(HttpObject o, boolean endOfStream, boolean flush) {
229         final ChannelFuture future;
230         if (o instanceof HttpData) {
231             final HttpData data = (HttpData) o;
232             future = encoder.writeData(ctx, id, streamId(), data, endOfStream);
233             logBuilder.increaseRequestLength(data.length());
234         } else if (o instanceof HttpHeaders) {
235             future = encoder.writeHeaders(ctx, id, streamId(), (HttpHeaders) o, endOfStream);
236         } else {
237             // Should never reach here because we did validation in onNext().
238             throw new Error();
239         }
240 
241         if (endOfStream) {
242             logBuilder.endRequest();
243         }
244 
245         future.addListener(this);
246         if (flush) {
247             ctx.flush();
248         }
249     }
250 
251     private int streamId() {
252         return (id << 1) + 1;
253     }
254 
255     private void fail(Throwable cause) {
256         setDone();
257         logBuilder.endRequest(cause);
258     }
259 
260     private void setDone() {
261         cancelTimeout();
262         state = State.DONE;
263         subscription.cancel();
264     }
265 
266     private void failAndRespond(Throwable cause) {
267         fail(cause);
268 
269         final Channel ch = ctx.channel();
270         final Http2Error error;
271         if (response.isOpen()) {
272             response.close(cause);
273             error = Http2Error.INTERNAL_ERROR;
274         } else if (cause instanceof WriteTimeoutException) {
275             error = Http2Error.CANCEL;
276         } else {
277             Exceptions.logIfUnexpected(logger, ch,
278                                        HttpSession.get(ch).protocol(),
279                                        "a request publisher raised an exception", cause);
280             error = Http2Error.INTERNAL_ERROR;
281         }
282 
283         if (ch.isActive()) {
284             encoder.writeReset(ctx, id, streamId(), error);
285             ctx.flush();
286         }
287     }
288 
289     private boolean cancelTimeout() {
290         final ScheduledFuture<?> timeoutFuture = this.timeoutFuture;
291         if (timeoutFuture == null) {
292             return true;
293         }
294 
295         this.timeoutFuture = null;
296         return timeoutFuture.cancel(false);
297     }
298 
299     private IllegalStateException newIllegalStateException(String msg) {
300         final IllegalStateException cause = new IllegalStateException(msg);
301         fail(cause);
302         return cause;
303     }
304 }
