54  /**
55   * A {@link DecoratingClientFactory} that creates a GRPC client.
56   */
57  public class GrpcClientFactory extends DecoratingClientFactory {
58  
59      private static final Set<Scheme> SUPPORTED_SCHEMES =
60              Arrays.stream(SessionProtocol.values())
61                    .flatMap(p -> GrpcSerializationFormats.values()
62                                                          .stream()
63                                                          .map(f -> Scheme.of(f, p)))
64                    .collect(toImmutableSet());
65  
66      /**
67       * Creates a new instance from the specified {@link ClientFactory} that supports HTTP, such as
68       * {@link HttpClientFactory}.
69       *
70       * @throws IllegalArgumentException if the specified {@link ClientFactory} does not support HTTP
71       */
72      public GrpcClientFactory(ClientFactory httpClientFactory) {
73          super(validate(httpClientFactory));
74      }
75  
76      @Override
77      public Set<Scheme> supportedSchemes() {
78          return SUPPORTED_SCHEMES;
79      }
80  
81      @Override
82      public <T> T newClient(URI uri, Class<T> clientType, ClientOptions options) {
83          Scheme scheme = validateScheme(uri);
84          SerializationFormat serializationFormat = scheme.serializationFormat();
85  
86  
87          Class<?> stubClass = clientType.getEnclosingClass();
88          if (stubClass == null) {
89              throw new IllegalArgumentException("Client type not a GRPC client stub class, " +
90                                                 "should be something like ServiceNameGrpc.ServiceNameXXStub: " +
91                                                 clientType);
92          }
93          final Method newStubMethod;
94          final Method newBlockingStubMethod;
95          final Method newFutureStubMethod;
96          try {
97              newStubMethod = stubClass.getDeclaredMethod("newStub", Channel.class);
98              newBlockingStubMethod = stubClass.getDeclaredMethod("newBlockingStub", Channel.class);
99              newFutureStubMethod = stubClass.getDeclaredMethod("newFutureStub", Channel.class);
100         } catch (NoSuchMethodException e) {
101             throw new IllegalArgumentException("Client type not a GRPC client stub class, " +
102                                                "should be something like ServiceNameGrpc.ServiceNameXXStub: " +
103                                                clientType);
104         }
105         final Method createClientMethod;
106         if (newStubMethod.getReturnType() == clientType) {
107             createClientMethod = newStubMethod;
108         } else if (newBlockingStubMethod.getReturnType() == clientType) {
109             createClientMethod = newBlockingStubMethod;
110         } else if (newFutureStubMethod.getReturnType() == clientType) {
111             createClientMethod = newFutureStubMethod;
112         } else {
113             throw new IllegalArgumentException("Client type not a GRPC client stub class, " +
114                                                "should be something like ServiceNameGrpc.ServiceNameXXStub: " +
115                                                clientType);
116         }
117 
118         Client<HttpRequest, HttpResponse> httpClient = newHttpClient(uri, scheme, options);
119 
120         MessageMarshaller jsonMarshaller = GrpcSerializationFormats.isJson(serializationFormat) ?
121                                            GrpcJsonUtil.jsonMarshaller(stubMethods(stubClass)) : null;
122         ArmeriaChannel channel = new ArmeriaChannel(
123                 new DefaultClientBuilderParams(this, uri, clientType, options),
124                 httpClient,
125                 scheme.sessionProtocol(),
126                 newEndpoint(uri),
127                 serializationFormat,
128                 jsonMarshaller);
129 
130         try {
131             // Verified createClientMethod.getReturnType == clientType
132             @SuppressWarnings("unchecked")
133             T stub = (T) createClientMethod.invoke(null, channel);
134             return stub;
135         } catch (IllegalAccessException | InvocationTargetException e) {
136             throw new IllegalStateException("Could not create stub through reflection.", e);
137         }
138     }
139 
140     @Override
141     public <T> Optional<ClientBuilderParams> clientBuilderParams(T client) {
142         if (!(client instanceof AbstractStub)) {
143             return Optional.empty();
144         }
145         AbstractStub<?> stub = (AbstractStub<?>) client;
146         if (!(stub.getChannel() instanceof ArmeriaChannel)) {
147             return Optional.empty();
148         }
149         return Optional.of((ArmeriaChannel) stub.getChannel());
150     }
151 
152     private static ClientFactory validate(ClientFactory httpClientFactory) {
153         requireNonNull(httpClientFactory, "httpClientFactory");
154 
155         for (SessionProtocol p : SessionProtocol.values()) {
156             if (!httpClientFactory.supportedSchemes().contains(Scheme.of(SerializationFormat.NONE, p))) {
157                 throw new IllegalArgumentException(p.uriText() + " not supported by: " + httpClientFactory);
158             }
159         }
160 
161         return httpClientFactory;
162     }
163 
164     private static List<MethodDescriptor<?, ?>> stubMethods(Class<?> stubClass) {
165         return Arrays.stream(stubClass.getDeclaredFields())
166                      .filter(field -> ((field.getModifiers() & Modifier.PUBLIC) != 0) &&
167                                       ((field.getModifiers() & Modifier.STATIC) != 0) &&
168                                       MethodDescriptor.class.isAssignableFrom(field.getType()))
169                      .map(field -> {
170                          try {
171                              return (MethodDescriptor<?, ?>) field.get(null);
172                          } catch (IllegalAccessException e) {
173                              throw new IllegalStateException("Modifier is public so can't happen.");
174                          }
175                      })
176                      .collect(toImmutableList());
177     }
178 
179     private Client<HttpRequest, HttpResponse> newHttpClient(URI uri, Scheme scheme, ClientOptions options) {
180         try {
181             @SuppressWarnings("unchecked")
182             Client<HttpRequest, HttpResponse> client = delegate().newClient(
183                     new URI(Scheme.of(SerializationFormat.NONE, scheme.sessionProtocol()).uriText(),
184                             uri.getAuthority(), null, null, null),
185                     Client.class,
186                     options);
187             return client;
188         } catch (URISyntaxException e) {
189             throw new Error(e); // Should never happen.
190         }
191     }
192 }
