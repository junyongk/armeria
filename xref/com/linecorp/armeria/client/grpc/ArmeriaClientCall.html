63  /**
64   * Encapsulates the state of a single client call, writing messages from the client and reading responses
65   * from the server, passing to business logic via {@link ClientCall.Listener}.
66   */
67  class ArmeriaClientCall<I, O> extends ClientCall<I, O>
68          implements ArmeriaMessageDeframer.Listener, TransportStatusListener {
69  
70      private static final Runnable NO_OP = () -> { };
71  
72      private static final Metadata EMPTY_METADATA = new Metadata();
73  
74      private static final Logger logger = LoggerFactory.getLogger(ArmeriaClientCall.class);
75  
76      private final ClientRequestContext ctx;
77      private final Client<HttpRequest, HttpResponse> httpClient;
78      private final DefaultHttpRequest req;
79      private final CallOptions callOptions;
80      private final ArmeriaMessageFramer messageFramer;
81      private final GrpcMessageMarshaller<I, O> marshaller;
82      private final CompressorRegistry compressorRegistry;
83      private final DecompressorRegistry decompressorRegistry;
84      private final HttpStreamReader responseReader;
85      @Nullable
86      private final Executor executor;
87  
88      // Effectively final, only set once during start()
89      private Listener<O> listener;
90  
91      private boolean cancelCalled;
92  
93      ArmeriaClientCall(
94              ClientRequestContext ctx,
95              Client<HttpRequest, HttpResponse> httpClient,
96              DefaultHttpRequest req,
97              MethodDescriptor<I, O> method,
98              int maxOutboundMessageSizeBytes,
99              int maxInboundMessageSizeBytes,
100             CallOptions callOptions,
101             CompressorRegistry compressorRegistry,
102             DecompressorRegistry decompressorRegistry,
103             SerializationFormat serializationFormat,
104             @Nullable MessageMarshaller jsonMarshaller) {
105         this.ctx = ctx;
106         this.httpClient = httpClient;
107         this.req = req;
108         this.callOptions = callOptions;
109         this.compressorRegistry = compressorRegistry;
110         this.decompressorRegistry = decompressorRegistry;
111         this.messageFramer = new ArmeriaMessageFramer(ctx.alloc(), maxOutboundMessageSizeBytes);
112         this.marshaller = new GrpcMessageMarshaller<>(
113                 ctx.alloc(), serializationFormat, method, jsonMarshaller);
114         responseReader = new HttpStreamReader(
115                 decompressorRegistry,
116                 new ArmeriaMessageDeframer(this, maxInboundMessageSizeBytes, ctx.alloc()),
117                 this);
118         executor = callOptions.getExecutor();
119     }
120 
121     @Override
122     public void start(Listener<O> responseListener, Metadata unused) {
123         requireNonNull(responseListener, "responseListener");
124         final Compressor compressor;
125         if (callOptions.getCompressor() != null) {
126             compressor = compressorRegistry.lookupCompressor(callOptions.getCompressor());
127             if (compressor == null) {
128                 responseListener.onClose(
129                         Status.INTERNAL.withDescription(
130                                 "Unable to find compressor by name " + callOptions.getCompressor()),
131                         EMPTY_METADATA);
132                 return;
133             }
134         } else {
135             compressor = Identity.NONE;
136         }
137         messageFramer.setCompressor(compressor);
138         prepareHeaders(req.headers(), compressor);
139         listener = responseListener;
140         final HttpResponse res;
141         try {
142             res = httpClient.execute(ctx, req);
143         } catch (Exception e) {
144             try (SafeCloseable ignored = RequestContext.push(ctx)) {
145                 listener.onClose(Status.fromThrowable(e), EMPTY_METADATA);
146             }
147             return;
148         }
149         res.subscribe(responseReader);
150     }
151 
152     @Override
153     public void request(int numMessages) {
154         if (ctx.eventLoop().inEventLoop()) {
155             responseReader.request(numMessages);
156         } else {
157             ctx.eventLoop().submit(() -> responseReader.request(numMessages));
158         }
159     }
160 
161     @Override
162     public void cancel(@Nullable String message, @Nullable Throwable cause) {
163         if (message == null && cause == null) {
164             cause = new CancellationException("Cancelled without a message or cause");
165             logger.warn("Cancelling without a message or cause is suboptimal", cause);
166         }
167         if (cancelCalled) {
168             return;
169         }
170         cancelCalled = true;
171         Status status = Status.CANCELLED;
172         if (message != null) {
173             status = status.withDescription(message);
174         }
175         if (cause != null) {
176             status = status.withCause(cause);
177         }
178         responseReader.cancel();
179         req.close(status.asException());
180         if (listener != null) {
181             try (SafeCloseable ignored = RequestContext.push(ctx)) {
182                 listener.onClose(status, EMPTY_METADATA);
183             }
184             notifyExecutor();
185         }
186     }
187 
188     @Override
189     public void halfClose() {
190         req.close();
191     }
192 
193     @Override
194     public void sendMessage(I message) {
195         try {
196             ByteBuf serialized = marshaller.serializeRequest(message);
197             boolean success = false;
198             final HttpData frame;
199             try {
200                 frame = messageFramer.writePayload(serialized);
201                 success = true;
202             } finally {
203                 if (!success) {
204                     serialized.release();
205                 }
206             }
207             req.write(frame);
208         } catch (Throwable t) {
209             cancel(null, t);
210         }
211     }
212 
213     @Override
214     public void setMessageCompression(boolean enabled) {
215         checkState(req != null, "Not started");
216         messageFramer.setMessageCompression(enabled);
217     }
218 
219     @Override
220     public void messageRead(ByteBufOrStream message) {
221         try {
222             O msg = marshaller.deserializeResponse(message);
223             try (SafeCloseable ignored = RequestContext.push(ctx)) {
224                 listener.onMessage(msg);
225             }
226         } catch (Throwable t) {
227             req.close(Status.fromThrowable(t).asException());
228             throw (t instanceof RuntimeException) ? (RuntimeException) t : new RuntimeException(t);
229         }
230     }
231 
232     @Override
233     public void endOfStream() {
234         // Ignore - the client call is terminated by headers, not data.
235     }
236 
237     @Override
238     public void transportReportStatus(Status status) {
239         responseReader.cancel();
240         try (SafeCloseable ignored = RequestContext.push(ctx)) {
241             listener.onClose(status, EMPTY_METADATA);
242         }
243         ctx.logBuilder().responseContent(GrpcLogUtil.rpcResponse(status), null);
244         notifyExecutor();
245     }
246 
247     private void prepareHeaders(HttpHeaders headers, Compressor compressor) {
248         if (compressor != Identity.NONE) {
249             headers.set(GrpcHeaderNames.GRPC_ENCODING, compressor.getMessageEncoding());
250         }
251         String advertisedEncodings = String.join(",", decompressorRegistry.getAdvertisedMessageEncodings());
252         if (!advertisedEncodings.isEmpty()) {
253             headers.add(GrpcHeaderNames.GRPC_ACCEPT_ENCODING, advertisedEncodings);
254         }
255         headers.add(GrpcHeaderNames.GRPC_TIMEOUT,
256                     TimeoutHeaderUtil.toHeaderValue(
257                             TimeUnit.MILLISECONDS.toNanos(ctx.responseTimeoutMillis())));
258     }
259 
260     /**
261      * Armeria does not support {@link CallOptions} set by the user, however GRPC stubs set an {@link Executor}
262      * within blocking stubs which is used to notify the stub when processing is finished. It's unclear why
263      * the stubs use a loop and {@link java.util.concurrent.Future#isDone()} instead of just blocking on
264      * {@link java.util.concurrent.Future#get}, but we make sure to run the {@link Executor} so the stub can
265      * be notified of completion.
266      */
267     private void notifyExecutor() {
268         if (executor != null) {
269             executor.execute(NO_OP);
270         }
271     }
