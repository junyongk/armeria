17  package com.linecorp.armeria.client.endpoint;
18  
19  import java.util.List;
20  import java.util.function.Consumer;
21  
22  import com.linecorp.armeria.client.Endpoint;
23  import com.linecorp.armeria.common.util.Listenable;
24  import com.linecorp.armeria.common.util.SafeCloseable;
25  
26  /**
27   * A list of {@link Endpoint}s.
28   */
29  @FunctionalInterface
30  public interface EndpointGroup extends Listenable<List<Endpoint>>, SafeCloseable {
31      /**
32       * Return the endpoints held by this {@link EndpointGroup}.
33       */
34      List<Endpoint> endpoints();
35  
36      @Override
37      default void addListener(Consumer<? super List<Endpoint>> listener) {}
38  
39      @Override
40      default void removeListener(Consumer<?> listener) {}
41  
42      @Override
43      default void close() {}
44  
45      /*
46       * Creates a new {@link EndpointGroup} that tries this {@link EndpointGroup} first and then the specified
47       * {@link EndpointGroup} when this {@link EndpointGroup} does not have a requested resource.
48       *
49       * @param nextEndpointGroup the {@link EndpointGroup} to try secondly.
50       */
51      default EndpointGroup orElse(EndpointGroup nextEndpointGroup) {
52          return new OrElseEndpointGroup(this, nextEndpointGroup);
53      }
54  
55  }
