17  package com.linecorp.armeria.client;
18  
19  import static java.util.Objects.requireNonNull;
20  
21  import com.google.common.net.HostAndPort;
22  
23  import com.linecorp.armeria.client.endpoint.EndpointGroupRegistry;
24  
25  /**
26   * A remote endpoint that refers to a single host or a group of multiple hosts.
27   *
28   * <p>A host endpoint has {@link #host()} and optional {@link #port()} and it can be represented as
29   * {@code "<host>"} or {@code "<host>:<port>"} in the authority part of a URI.
30   *
31   * <p>A group endpoint has {@link #groupName()} and it can be represented as {@code "group:<groupName>"}
32   * in the authority part of a URI. It can be resolved into a host endpoint with {@link #resolve()}.
33   */
34  public final class Endpoint {
35  
36      /**
37       * Parse the authority part of a URI. The authority part may have one of the following formats:
38       * <ul>
39       *   <li>{@code "group:<groupName>"} for a group endpoint</li>
40       *   <li>{@code "<host>:<port>"} for a host endpoint</li>
41       *   <li>{@code "<host>"} for a host endpoint with no port number specified</li>
42       * </ul>
43       */
44      public static Endpoint parse(String authority) {
45          requireNonNull(authority, "authority");
46          if (authority.startsWith("group:")) {
47              return ofGroup(authority.substring(6));
48          }
49  
50          final HostAndPort parsed = HostAndPort.fromString(authority).withDefaultPort(0);
51          return new Endpoint(parsed.getHost(), parsed.getPort(), 1000);
52      }
53  
54      /**
55       * Creates a new group {@link Endpoint}.
56       */
57      public static Endpoint ofGroup(String name) {
58          requireNonNull(name, "name");
59          return new Endpoint(name);
60      }
61  
62      /**
63       * Creates a new host {@link Endpoint}.
64       */
65      public static Endpoint of(String host, int port) {
66          return of(host, port, 1000);
67      }
68  
69      /**
70       * Creates a new host {@link Endpoint} with unspecified port number.
71       */
72      public static Endpoint of(String host) {
73          return new Endpoint(host, 0, 1000);
74      }
75  
76      // TODO(trustin): Remove weight and make Endpoint a pure endpoint representation.
77      //                We could specify an additional attributes such as weight/priority
78      //                when adding an Endpoint to an EndpointGroup.
79  
80      /**
81       * Creates a new host {@link Endpoint}.
82       */
83      public static Endpoint of(String host, int port, int weight) {
84          requireNonNull(host, "host");
85          validatePort("port", port);
86          validateWeight(weight);
87  
88          return new Endpoint(host, port, weight);
89      }
90  
91      private final String groupName;
92      private final String host;
93      private final int port;
94      private final int weight;
95      private String authority;
96  
97      private Endpoint(String groupName) {
98          this.groupName = groupName;
99          host = null;
100         port = 0;
101         weight = 0;
102     }
103 
104     private Endpoint(String host, int port, int weight) {
105         this.host = host;
106         this.port = port;
107         this.weight = weight;
108         groupName = null;
109     }
110 
111     /**
112      * Returns {@code true} if this endpoint refers to a group.
113      */
114     public boolean isGroup() {
115         return groupName != null;
116     }
117 
118     /**
119      * Resolves this endpoint into a host endpoint.
120      *
121      * @return the {@link Endpoint} resolved by {@link EndpointGroupRegistry}.
122      *         {@code this} if this endpoint is already a host endpoint.
123      */
124     public Endpoint resolve() {
125         if (isGroup()) {
126             return EndpointGroupRegistry.selectNode(groupName);
127         } else {
128             return this;
129         }
130     }
131 
132     /**
133      * Returns the group name of this endpoint.
134      *
135      * @throws IllegalStateException if this endpoint is not a group endpoint
136      */
137     public String groupName() {
138         ensureGroup();
139         return groupName;
140     }
141 
142     /**
143      * Returns the host name of this endpoint.
144      *
145      * @throws IllegalStateException if this endpoint is not a host endpoint
146      */
147     public String host() {
148         ensureSingle();
149         return host;
150     }
151 
152     /**
153      * Returns the port number of this endpoint.
154      *
155      * @throws IllegalStateException if this endpoint is not a host endpoint or
156      *                               this endpoint does not have its port specified.
157      */
158     public int port() {
159         ensureSingle();
160         if (port == 0) {
161             throw new IllegalStateException("port not specified");
162         }
163         return port;
164     }
165 
166     /**
167      * Returns the port number of this endpoint.
168      *
169      * @param defaultPort the default port number to use when this endpoint does not have its port specified
170      *
171      * @throws IllegalStateException if this endpoint is not a host endpoint
172      */
173     public int port(int defaultPort) {
174         ensureSingle();
175         validatePort("defaultPort", defaultPort);
176         return port != 0 ? port : defaultPort;
177     }
178 
179     /**
180      * Returns a new host endpoint with the specified default port number.
181      *
182      * @return the new endpoint whose port is {@code defaultPort} if this endpoint does not have its port
183      *         specified. {@code this} if this endpoint already has its port specified.
184      *
185      * @throws IllegalStateException if this endpoint is not a host endpoint
186      */
187     public Endpoint withDefaultPort(int defaultPort) {
188         ensureSingle();
189         validatePort("defaultPort", defaultPort);
190 
191         return port != 0 ? this : new Endpoint(host(), defaultPort, weight());
192     }
193 
194     /**
195      * Returns a new host endpoint with the specified weight.
196      *
197      * @return the new endpoint with the specified weight. {@code this} if this endpoint has the same weight.
198      *
199      * @throws IllegalStateException if this endpoint is not a host endpoint
200      */
201     public Endpoint withWeight(int weight) {
202         ensureSingle();
203         validateWeight(weight);
204 
205         return this.weight == weight ? this : new Endpoint(host(), port, weight);
206     }
207 
208     /**
209      * Returns the weight of this endpoint.
210      */
211     public int weight() {
212         ensureSingle();
213         return weight;
214     }
215 
216     /**
217      * Converts this endpoint into the authority part of a URI.
218      *
219      * @return the authority string
220      */
221     public String authority() {
222         String authority = this.authority;
223         if (authority != null) {
224             return authority;
225         }
226 
227         if (isGroup()) {
228             authority = "group:" + groupName;
229         } else if (port != 0) {
230             authority = host() + ':' + port;
231         } else {
232             authority = host();
233         }
234 
235         return this.authority = authority;
236     }
237 
238     private void ensureGroup() {
239         if (!isGroup()) {
240             throw new IllegalStateException("not a group endpoint");
241         }
242     }
243 
244     private void ensureSingle() {
245         if (isGroup()) {
246             throw new IllegalStateException("not a host:port endpoint");
247         }
248     }
249 
250     private static void validatePort(String name, int port) {
251         if (port <= 0 || port > 65535) {
252             throw new IllegalArgumentException(name + ": " + port + " (expected: 1-65535)");
253         }
254     }
255 
256     private static void validateWeight(int weight) {
257         if (weight <= 0) {
258             throw new IllegalArgumentException("weight: " + weight + " (expected: > 0)");
259         }
260     }
261 
262     @Override
263     public String toString() {
264         return "Endpoint(" + authority() + '/' + weight + ')';
265     }
266 
267     @Override
268     public boolean equals(Object obj) {
269         if (this == obj) {
270             return true;
271         }
272 
273         if (!(obj instanceof Endpoint)) {
274             return false;
275         }
276 
277         final Endpoint that = (Endpoint) obj;
278         return authority().equals(that.authority()) && weight() == that.weight();
279     }
280 
281     @Override
282     public int hashCode() {
283         return authority().hashCode();
284     }
285 }
