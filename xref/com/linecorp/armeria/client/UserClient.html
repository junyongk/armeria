17  package com.linecorp.armeria.client;
18  
19  import java.net.URI;
20  import java.util.function.Function;
21  
22  import javax.annotation.Nullable;
23  
24  import com.linecorp.armeria.common.DefaultHttpHeaders;
25  import com.linecorp.armeria.common.HttpHeaders;
26  import com.linecorp.armeria.common.HttpMethod;
27  import com.linecorp.armeria.common.Request;
28  import com.linecorp.armeria.common.RequestContext;
29  import com.linecorp.armeria.common.Response;
30  import com.linecorp.armeria.common.SessionProtocol;
31  import com.linecorp.armeria.common.util.SafeCloseable;
32  
33  import io.netty.channel.EventLoop;
34  import io.netty.util.Attribute;
35  
36  /**
37   * A base class for implementing a user's entry point for sending a {@link Request}.
38   *
39   * <p>It provides the utility methods for easily forwarding a {@link Request} from a user to a {@link Client}.
40   *
41   * <p>Note that this class is not a subtype of {@link Client}, although its name may mislead.
42   *
43   * @param <I> the request type
44   * @param <O> the response type
45   */
46  public abstract class UserClient<I extends Request, O extends Response> implements ClientBuilderParams {
47  
48      static final ThreadLocal<Function<HttpHeaders, HttpHeaders>> THREAD_LOCAL_HEADER_MANIPULATOR =
49              new ThreadLocal<>();
50  
51      private final ClientBuilderParams params;
52      private final Client<I, O> delegate;
53      private final SessionProtocol sessionProtocol;
54      private final Endpoint endpoint;
55  
56      /**
57       * Creates a new instance.
58       *
59       * @param params the parameters used for constructing the client
60       * @param delegate the {@link Client} that will process {@link Request}s
61       * @param sessionProtocol the {@link SessionProtocol} of the {@link Client}
62       * @param endpoint the {@link Endpoint} of the {@link Client}
63       */
64      protected UserClient(ClientBuilderParams params,
65                           Client<I, O> delegate, SessionProtocol sessionProtocol, Endpoint endpoint) {
66          this.params = params;
67          this.delegate = delegate;
68          this.sessionProtocol = sessionProtocol;
69          this.endpoint = endpoint;
70      }
71  
72      @Override
73      public ClientFactory factory() {
74          return params.factory();
75      }
76  
77      @Override
78      public URI uri() {
79          return params.uri();
80      }
81  
82      @Override
83      public Class<?> clientType() {
84          return params.clientType();
85      }
86  
87      @Override
88      public final ClientOptions options() {
89          return params.options();
90      }
91  
92      /**
93       * Returns the {@link Client} that will process {@link Request}s.
94       */
95      @SuppressWarnings("unchecked")
96      protected final <U extends Client<I, O>> U delegate() {
97          return (U) delegate;
98      }
99  
100     /**
101      * Retrieves an {@link EventLoop} supplied by the {@link ClientFactory}.
102      */
103     protected final EventLoop eventLoop() {
104         return params.factory().eventLoopSupplier().get();
105     }
106 
107     /**
108      * Returns the {@link SessionProtocol} of the {@link #delegate()}.
109      */
110     protected final SessionProtocol sessionProtocol() {
111         return sessionProtocol;
112     }
113 
114     /**
115      * Returns the {@link Endpoint} of the {@link #delegate()}.
116      */
117     protected final Endpoint endpoint() {
118         return endpoint;
119     }
120 
121     /**
122      * Executes the specified {@link Request} via {@link #delegate()}.
123      *
124      * @param method the method of the {@link Request}
125      * @param path the path part of the {@link Request} URI
126      * @param query the query part of the {@link Request} URI
127      * @param fragment the fragment part of the {@link Request} URI
128      * @param req the {@link Request}
129      * @param fallback the fallback response {@link Function} to use when
130      *                 {@link Client#execute(ClientRequestContext, Request)} of {@link #delegate()} throws
131      *                 an exception instead of returning an error response
132      */
133     protected final O execute(HttpMethod method, String path, @Nullable String query, @Nullable String fragment,
134                               I req, Function<Throwable, O> fallback) {
135         return execute(eventLoop(), method, path, query, fragment, req, fallback);
136     }
137 
138     /**
139      * Executes the specified {@link Request} via {@link #delegate()}.
140      *
141      * @param eventLoop the {@link EventLoop} to execute the {@link Request}
142      * @param method the method of the {@link Request}
143      * @param path the path part of the {@link Request} URI
144      * @param query the query part of the {@link Request} URI
145      * @param fragment the fragment part of the {@link Request} URI
146      * @param req the {@link Request}
147      * @param fallback the fallback response {@link Function} to use when
148      *                 {@link Client#execute(ClientRequestContext, Request)} of {@link #delegate()} throws
149      */
150     protected final O execute(EventLoop eventLoop,
151                               HttpMethod method, String path, @Nullable String query, @Nullable String fragment,
152                               I req, Function<Throwable, O> fallback) {
153 
154         final ClientRequestContext ctx = new DefaultClientRequestContext(
155                 eventLoop, sessionProtocol, endpoint, method, path, query, fragment, options(), req);
156 
157         try (SafeCloseable ignored = RequestContext.push(ctx)) {
158             runThreadLocalHeaderManipulator(ctx);
159             return delegate().execute(ctx, req);
160         } catch (Throwable cause) {
161             ctx.logBuilder().endResponse(cause);
162             return fallback.apply(cause);
163         }
164     }
165 
166     private static void runThreadLocalHeaderManipulator(ClientRequestContext ctx) {
167         final Function<HttpHeaders, HttpHeaders> manipulator = THREAD_LOCAL_HEADER_MANIPULATOR.get();
168         if (manipulator == null) {
169             return;
170         }
171 
172         final Attribute<HttpHeaders> attr = ctx.attr(ClientRequestContext.HTTP_HEADERS);
173         final HttpHeaders headers = attr.get();
174         attr.set(manipulator.apply(headers != null ? headers : new DefaultHttpHeaders()));
175     }
176 }
