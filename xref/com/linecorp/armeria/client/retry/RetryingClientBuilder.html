17  package com.linecorp.armeria.client.retry;
18  
19  import static com.google.common.base.Preconditions.checkArgument;
20  import static java.util.Objects.requireNonNull;
21  
22  import java.util.function.Function;
23  import java.util.function.Supplier;
24  
25  import com.google.common.base.MoreObjects;
26  import com.google.common.base.MoreObjects.ToStringHelper;
27  
28  import com.linecorp.armeria.client.Client;
29  import com.linecorp.armeria.client.ClientOptions;
30  import com.linecorp.armeria.common.Flags;
31  import com.linecorp.armeria.common.Request;
32  import com.linecorp.armeria.common.Response;
33  
34  /**
35   * Builds a new {@link RetryingClient} or its decorator function.
36   *
37   * @param <T> the type of {@link RetryingClientBuilder}
38   * @param <U> the type of the {@link Client} that this builder builds or decorates
39   * @param <I> the type of outgoing {@link Request} of the {@link Client}
40   * @param <O> the type of incoming {@link Response} of the {@link Client}
41   */
42  public abstract class RetryingClientBuilder<
43          T extends RetryingClientBuilder<T, U, I, O>, U extends RetryingClient<I, O>,
44          I extends Request, O extends Response> {
45  
46      final RetryStrategy<I, O> retryStrategy;
47      // TODO(minwoox) remove hardcoded paramters and add them to Flags with a new parser
48      Supplier<? extends Backoff> backoffSupplier = () -> Backoff
49              .exponential(Flags.defaultExponentialBackoffInitialDelayMillis(),
50                           ClientOptions.DEFAULT.defaultResponseTimeoutMillis())
51              .withJitter(0, 100)
52              .withMaxAttempts(Flags.defaultBackoffMaxAttempts());
53      int defaultMaxAttempts = Flags.defaultBackoffMaxAttempts();
54  
55      /**
56       * Creates a new builder with the specified retry strategy.
57       */
58      protected RetryingClientBuilder(RetryStrategy<I, O> retryStrategy) {
59          this.retryStrategy = requireNonNull(retryStrategy, "retryStrategy");
60      }
61  
62      @SuppressWarnings("unchecked")
63      final T self() {
64          return (T) this;
65      }
66  
67      /**
68       * Sets the {@link Supplier} that provides a {@link Backoff}. The {@link Backoff} will be used to
69       * calculate next delay to retry when the {@link RetryStrategy#shouldRetry(Request, Response)}
70       * returns {@code true}.
71       *
72       * @return {@link T} to support method chaining.
73       */
74      public T backoffSupplier(Supplier<? extends Backoff> backoffSupplier) {
75          this.backoffSupplier = requireNonNull(backoffSupplier, "backoffSupplier");
76          return self();
77      }
78  
79      /**
80       * Sets the {@code defaultMaxAttempts}. When a client sets the {@link Backoff} and does not invoke the
81       * {@link Backoff#withMaxAttempts(int)}, the client could retry infinitely in certain circumstance.
82       * This would prevent that situation. The value will be set by the
83       * {@link Flags#DEFAULT_BACKOFF_MAX_ATTEMPTS}, if the client dose not specify.
84       *
85       * @return {@link T} to support method chaining.
86       */
87      public T defaultMaxAttempts(int defaultMaxAttempts) {
88          checkArgument(defaultMaxAttempts > 0,
89                        "defaultMaxAttempts: %s (expected: > 0)", defaultMaxAttempts);
90          this.defaultMaxAttempts = defaultMaxAttempts;
91          return self();
92      }
93  
94      /**
95       * Builds a {@link U} with settings specified by previous method calls.
96       *
97       * @return {@link U} the decorating {@link RetryingClient} instance.
98       */
99      abstract U build(Client<I, O> delegate);
100 
101     /**
102      * Creates a new decorator that decorates a {@link Client} with a new {@link RetryingClient}.
103      */
104     abstract Function<Client<I, O>, U> newDecorator();
105 
106     @Override
107     public String toString() {
108         return toStringHelper().toString();
109     }
110 
111     ToStringHelper toStringHelper() {
112         return MoreObjects.toStringHelper(this)
113                           .add("retryStrategy", retryStrategy)
114                           .add("backoff", backoffSupplier.get())
115                           .add("defaultMaxAttempts", defaultMaxAttempts);
116     }
117 }
