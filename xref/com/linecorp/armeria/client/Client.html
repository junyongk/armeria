17  package com.linecorp.armeria.client;
18  
19  import com.linecorp.armeria.common.HttpRequest;
20  import com.linecorp.armeria.common.HttpResponse;
21  import com.linecorp.armeria.common.Request;
22  import com.linecorp.armeria.common.Response;
23  import com.linecorp.armeria.common.RpcRequest;
24  import com.linecorp.armeria.common.RpcResponse;
25  
26  /**
27   * Sends a {@link Request} to a remote {@link Endpoint}.
28   *
29   * <p>Note that this interface is not a user's entry point for sending a {@link Request}. It is rather
30   * a generic request processor interface implemented by a {@link DecoratingClient}, which intercepts
31   * a {@link Request}. A user is supposed to make his or her {@link Request} via the object returned by
32   * a {@link ClientBuilder} or {@link Clients}, which usually does not implement this interface.
33   *
34   * @param <I> the type of outgoing {@link Request}. Must be {@link HttpRequest} or {@link RpcRequest}.
35   * @param <O> the type of incoming {@link Response}. Must be {@link HttpResponse} or {@link RpcResponse}.
36   *
37   * @see UserClient
38   */
39  @FunctionalInterface
40  public interface Client<I extends Request, O extends Response> {
41      /**
42       * Sends a {@link Request} to a remote {@link Endpoint}, as specified in
43       * {@link ClientRequestContext#endpoint()}.
44       *
45       * @return the {@link Response} to the specified {@link Request}
46       */
47      O execute(ClientRequestContext ctx, I req) throws Exception;
48  }
