19  import static java.util.Objects.requireNonNull;
20  
21  import java.util.Queue;
22  import java.util.concurrent.ConcurrentLinkedQueue;
23  import java.util.concurrent.TimeUnit;
24  import java.util.concurrent.atomic.AtomicInteger;
25  import java.util.concurrent.atomic.AtomicReference;

27  import com.linecorp.armeria.client.Client;
28  import com.linecorp.armeria.client.ClientRequestContext;
29  import com.linecorp.armeria.client.ResponseTimeoutException;
30  import com.linecorp.armeria.client.SimpleDecoratingClient;
31  import com.linecorp.armeria.common.Request;
32  import com.linecorp.armeria.common.RequestContext;
33  import com.linecorp.armeria.common.Response;
34  import com.linecorp.armeria.common.util.SafeCloseable;

36  import io.netty.util.concurrent.ScheduledFuture;

38  /**
39   * An abstract {@link Client} decorator that limits the concurrent number of active requests.
40   *
41   * <p>{@link #numActiveRequests()} increases when {@link Client#execute(ClientRequestContext, Request)} is
42   * invoked and decreases when the {@link Response} returned by the
43   * {@link Client#execute(ClientRequestContext, Request)} is closed. When {@link #numActiveRequests()} reaches
44   * at the configured {@code maxConcurrency} the {@link Request}s are deferred until the currently active
45   * {@link Request}s are completed.
46   *
47   * @param <I> the {@link Request} type
48   * @param <O> the {@link Response} type
49   */
50  public abstract class ConcurrencyLimitingClient<I extends Request, O extends Response>
51          extends SimpleDecoratingClient<I, O> {
52  
53      private static final long DEFAULT_TIMEOUT_MILLIS = 10000L;
54  
55      private final int maxConcurrency;
56      private final long timeoutMillis;
57      private final AtomicInteger numActiveRequests = new AtomicInteger();
58      private final Queue<PendingTask> pendingRequests = new ConcurrentLinkedQueue<>();
59  
60      /**
61       * Creates a new instance that decorates the specified {@code delegate} to limit the concurrent number of
62       * active requests to {@code maxConcurrency}, with the default timeout of {@value #DEFAULT_TIMEOUT_MILLIS}
63       * milliseconds.
64       *
65       * @param delegate the delegate {@link Client}
66       * @param maxConcurrency the maximum number of concurrent active requests. {@code 0} to disable the limit.
67       */
68      protected ConcurrencyLimitingClient(Client<I, O> delegate, int maxConcurrency) {
69          this(delegate, maxConcurrency, DEFAULT_TIMEOUT_MILLIS, TimeUnit.MILLISECONDS);
70      }
71  
72      /**
73       * Creates a new instance that decorates the specified {@code delegate} to limit the concurrent number of
74       * active requests to {@code maxConcurrency}.
75       *
76       * @param delegate the delegate {@link Client}
77       * @param maxConcurrency the maximum number of concurrent active requests. {@code 0} to disable the limit.
78       * @param timeout the amount of time until this decorator fails the request if the request was not
79       *                delegated to the {@code delegate} before then
80       */
81      protected ConcurrencyLimitingClient(Client<I, O> delegate,
82                                          int maxConcurrency, long timeout, TimeUnit unit) {
83          super(delegate);
84  
85          validateAll(maxConcurrency, timeout, unit);
86  
87          this.maxConcurrency = maxConcurrency;
88          timeoutMillis = unit.toMillis(timeout);
89      }
90  
91      static void validateAll(int maxConcurrency, long timeout, TimeUnit unit) {
92          validateMaxConcurrency(maxConcurrency);
93          if (timeout < 0) {
94              throw new IllegalArgumentException("timeout: " + timeout + " (expected: >= 0)");
95          }
96          requireNonNull(unit, "unit");
97      }
98  
99      static void validateMaxConcurrency(int maxConcurrency) {
100         if (maxConcurrency < 0) {
101             throw new IllegalArgumentException("maxConcurrency: " + maxConcurrency + " (expected: >= 0)");
102         }
103     }
104 
105     /**
106      * Returns the number of the {@link Request}s that are being executed.
107      */
108     public int numActiveRequests() {
109         return numActiveRequests.get();
110     }
111 
112     @Override
113     public O execute(ClientRequestContext ctx, I req) throws Exception {
114         return maxConcurrency == 0 ? unlimitedExecute(ctx, req)
115                                    : limitedExecute(ctx, req);
116     }
117 
118     private O limitedExecute(ClientRequestContext ctx, I req) throws Exception {
119         final Deferred<O> deferred = defer(ctx, req);
120         final PendingTask currentTask = new PendingTask(ctx, req, deferred);
121 
122         pendingRequests.add(currentTask);
123         drain();
124 
125         if (!currentTask.isRun() && timeoutMillis != 0) {
126             // Current request was not delegated. Schedule a timeout.
127             final ScheduledFuture<?> timeoutFuture = ctx.eventLoop().schedule(
128                     () -> deferred.close(ResponseTimeoutException.get()),
129                     timeoutMillis, TimeUnit.MILLISECONDS);
130             currentTask.set(timeoutFuture);
131         }
132 
133         return deferred.response();
134     }
135 
136     private O unlimitedExecute(ClientRequestContext ctx, I req) throws Exception {
137         numActiveRequests.incrementAndGet();
138         boolean success = false;
139         try {
140             final O res = delegate().execute(ctx, req);
141             res.closeFuture().whenComplete((unused, cause) -> numActiveRequests.decrementAndGet());
142             success = true;
143             return res;
144         } finally {
145             if (!success) {
146                 numActiveRequests.decrementAndGet();
147             }
148         }
149     }
150 
151     void drain() {
152         while (!pendingRequests.isEmpty()) {
153             final int currentActiveRequests = numActiveRequests.get();
154             if (currentActiveRequests >= maxConcurrency) {
155                 break;
156             }
157 
158             if (numActiveRequests.compareAndSet(currentActiveRequests, currentActiveRequests + 1)) {
159                 final PendingTask task = pendingRequests.poll();
160                 if (task == null) {
161                     numActiveRequests.decrementAndGet();
162                     if (!pendingRequests.isEmpty()) {
163                         // Another request might have been added to the queue while numActiveRequests reached
164                         // at its limit.
165                         continue;
166                     } else {
167                         break;
168                     }
169                 }
170 
171                 task.run();
172             }
173         }
174     }
175 
176     /**
177      * Defers the specified {@link Request}.
178      *
179      * @return a new {@link Deferred} which provides the interface for updating the result of
180      *         {@link Request} execution later.
181      */
182     protected abstract Deferred<O> defer(ClientRequestContext ctx, I req) throws Exception;
183 
184     /**
185      * Provides the interface for updating the result of a {@link Request} execution when its {@link Response}
186      * is ready.
187      *
188      * @param <O> the {@link Response} type
189      */
190     public interface Deferred<O extends Response> {
191         /**
192          * Returns the {@link Response} which will delegate to the {@link Response} set by
193          * {@link #delegate(Response)}.
194          */
195         O response();
196 
197         /**
198          * Delegates the {@link #response() response} to the specified {@link Response}.
199          */
200         void delegate(O response);
201 
202         /**
203          * Closes the {@link #response()} without delegating.
204          */
205         void close(Throwable cause);
206     }
207 
208     private final class PendingTask extends AtomicReference<ScheduledFuture<?>> implements Runnable {
209 
210         private static final long serialVersionUID = -7092037489640350376L;
211 
212         private final ClientRequestContext ctx;
213         private final I req;
214         private final Deferred<O> deferred;
215         private boolean isRun;
216 
217         PendingTask(ClientRequestContext ctx, I req, Deferred<O> deferred) {
218             this.ctx = ctx;
219             this.req = req;
220             this.deferred = deferred;
221         }
222 
223         boolean isRun() {
224             return isRun;
225         }
226 
227         @Override
228         public void run() {
229             isRun = true;
230 
231             ScheduledFuture<?> timeoutFuture = get();
232             if (timeoutFuture != null) {
233                 if (timeoutFuture.isDone() || !timeoutFuture.cancel(false)) {
234                     // Timeout task ran already or is determined to run.
235                     numActiveRequests.decrementAndGet();
236                     return;
237                 }
238             }
239 
240             try (SafeCloseable ignored = RequestContext.push(ctx)) {
241                 try {
242                     final O actualRes = delegate().execute(ctx, req);
243                     actualRes.closeFuture().whenCompleteAsync((unused, cause) -> {
244                         numActiveRequests.decrementAndGet();
245                         drain();
246                     }, ctx.eventLoop());
247                     deferred.delegate(actualRes);
248                 } catch (Throwable t) {
249                     numActiveRequests.decrementAndGet();
250                     deferred.close(t);
251                 }
252             }
253         }
254     }
