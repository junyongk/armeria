19  import static com.google.common.base.Preconditions.checkState;

24  import java.util.concurrent.Executor;

44  /**
45   * A helper class for creating a new {@link Retrofit} instance with {@link ArmeriaCallFactory}.
46   * For example,
47   *
48   * <pre>{@code
49   * Retrofit retrofit = new ArmeriaRetrofitBuilder()
50   *     .baseUrl("http://localhost:8080/")
51   *     .build();
52   *
53   * MyApi api = retrofit.create(MyApi.class);
54   * Response<User> user = api.getUser().execute();
55   * }</pre>
56   *
57   * <p>{@link ArmeriaRetrofitBuilder} even supports {@link EndpointGroup}, so you can create {@link Retrofit}
58   * like below,
59   *
60   * <pre>{@code
61   * EndpointGroupRegistry.register("foo",
62   *                                new StaticEndpointGroup(Endpoint.of("127.0.0.1", 8080)),
63   *                                ROUND_ROBIN);
64   *
65   * Retrofit retrofit = new ArmeriaRetrofitBuilder()
66   *     .baseUrl("http://group:foo/")
67   *     .build();
68   * }</pre>
69   */
70  public final class ArmeriaRetrofitBuilder {
71  
72      private static final BiFunction<String, ? super ClientOptionsBuilder, ClientOptionsBuilder>
73              DEFAULT_CONFIGURATOR = (url, optionsBuilder) -> optionsBuilder;
74      private static final String SLASH = "/";
75  
76      private final Retrofit.Builder retrofitBuilder;
77      private final ClientFactory clientFactory;
78      private String baseUrl;
79      private BiFunction<String, ? super ClientOptionsBuilder, ClientOptionsBuilder> configurator =
80              DEFAULT_CONFIGURATOR;
81  
82      /**
83       * Creates a {@link ArmeriaRetrofitBuilder} with the default {@link ClientFactory}.
84       */
85      public ArmeriaRetrofitBuilder() {
86          this(ClientFactory.DEFAULT);
87      }
88  
89      /**
90       * Creates a {@link ArmeriaRetrofitBuilder} with the specified {@link ClientFactory}.
91       */
92      public ArmeriaRetrofitBuilder(ClientFactory clientFactory) {
93          this.clientFactory = requireNonNull(clientFactory, "clientFactory");
94          retrofitBuilder = new Retrofit.Builder();
95      }
96  
97      /**
98       * Sets the API base URL.
99       *
100      * @see Builder#baseUrl(String)
101      */
102     public ArmeriaRetrofitBuilder baseUrl(String baseUrl) {
103         requireNonNull(baseUrl, "baseUrl");
104         URI uri = URI.create(baseUrl);
105         checkArgument(SessionProtocol.find(uri.getScheme()).isPresent(),
106                       "baseUrl must have an HTTP scheme: %s", baseUrl);
107         String path = uri.getPath();
108         if (!path.isEmpty() && !SLASH.equals(path.substring(path.length() - 1))) {
109             throw new IllegalArgumentException("baseUrl must end with /: " + baseUrl);
110         }
111         this.baseUrl = baseUrl;
112         return this;
113     }
114 
115     /**
116      * Sets the {@link BiFunction} that is applied to the underlying {@link HttpClient}.
117      *
118      * @param configurator a {@link BiFunction} whose first argument is the the URI of the server endpoint and
119      *                     whose second argument is the {@link ClientOptionsBuilder} with default options of
120      *                     the new derived client
121      */
122     public ArmeriaRetrofitBuilder withClientOptions(
123             BiFunction<String, ? super ClientOptionsBuilder, ClientOptionsBuilder> configurator) {
124         this.configurator = requireNonNull(configurator, "configurator");
125         return this;
126     }
127 
128     /**
129      * Adds the specified converter factory for serialization and deserialization of objects.
130      * @see Retrofit.Builder#addCallAdapterFactory(CallAdapter.Factory)
131      */
132     public ArmeriaRetrofitBuilder addConverterFactory(Converter.Factory factory) {
133         retrofitBuilder.addConverterFactory(requireNonNull(factory, "factory"));
134         return this;
135     }
136 
137     /**
138      * Adds the specified call adapter factory for supporting service method return types other than {@link
139      * Call}.
140      * @see Retrofit.Builder#addCallAdapterFactory(CallAdapter.Factory)
141      */
142     public ArmeriaRetrofitBuilder addCallAdapterFactory(CallAdapter.Factory factory) {
143         retrofitBuilder.addCallAdapterFactory(requireNonNull(factory, "factory"));
144         return this;
145     }
146 
147     /**
148      * Sets the {@link Executor} on which {@link Callback} methods are invoked when returning {@link Call} from
149      * your service method.
150      *
151      * <p>Note: {@code executor} is not used for {@link #addCallAdapterFactory custom method return types}.
152      *
153      * @see Retrofit.Builder#callbackExecutor(Executor)
154      */
155     public ArmeriaRetrofitBuilder callbackExecutor(Executor executor) {
156         retrofitBuilder.callbackExecutor(requireNonNull(executor, "executor"));
157         return this;
158     }
159 
160     /**
161      * When calling {@link Retrofit#create} on the resulting {@link Retrofit} instance, eagerly validate
162      * the configuration of all methods in the supplied interface.
163      *
164      * @see Retrofit.Builder#validateEagerly(boolean)
165      */
166     public ArmeriaRetrofitBuilder validateEagerly(boolean validateEagerly) {
167         retrofitBuilder.validateEagerly(validateEagerly);
168         return this;
169     }
170 
171     /**
172      * Creates a new {@link Retrofit} instance using the configured values.
173      */
174     public Retrofit build() {
175         checkState(baseUrl != null, "baseUrl not set");
176         final URI uri = URI.create(baseUrl);
177         final Scheme scheme = Scheme.of(SerializationFormat.NONE, SessionProtocol.of(uri.getScheme()));
178         final String fullUri = scheme.uriText() + "://" + uri.getAuthority();
179         final HttpClient baseHttpClient =
180                 Clients.newClient(clientFactory, fullUri, HttpClient.class,
181                                   configurator.apply(fullUri, new ClientOptionsBuilder()).build());
182         return retrofitBuilder.baseUrl(convertToOkHttpUrl(baseHttpClient, uri.getPath(), GROUP_PREFIX))
183                               .callFactory(new ArmeriaCallFactory(baseHttpClient, clientFactory, configurator))
184                               .build();
185     }
186 
187     private static HttpUrl convertToOkHttpUrl(HttpClient baseHttpClient, String basePath, String groupPrefix) {
188         final URI uri = baseHttpClient.uri();
189         final SessionProtocol sessionProtocol = Scheme.tryParse(uri.getScheme())
190                                                       .map(Scheme::sessionProtocol)
191                                                       .orElseGet(() -> SessionProtocol.of(uri.getScheme()));
192         final String authority = uri.getAuthority();
193         String protocol = sessionProtocol.isTls() ? "https" : "http";
194         if (authority.startsWith("group:")) {
195             return HttpUrl.parse(protocol + "://" + authority.replace("group:", groupPrefix) + basePath);
196         } else {
197             return HttpUrl.parse(protocol + "://" + authority + basePath);
198         }
199     }
