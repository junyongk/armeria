16  package com.linecorp.armeria.client;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.net.URI;
21  
22  /**
23   * Creates a new client that connects to the specified {@link URI} using the builder pattern. Use the factory
24   * methods in {@link Clients} if you do not have many options to override.
25   *
26   * <h3>How are decorators and HTTP headers configured?</h3>
27   *
28   * <p>Unlike other options, when a user calls {@link #option(ClientOption, Object)} or {@code options()} with
29   * a {@link ClientOption#DECORATION} or a {@link ClientOption#HTTP_HEADERS}, this builder will not simply
30   * replace the old option but <em>merge</em> the specified option into the previous option value. For example:
31   * <pre>{@code
32   * ClientOptionsBuilder b = new ClientOptionsBuilder();
33   * b.option(ClientOption.HTTP_HEADERS, headersA);
34   * b.option(ClientOption.HTTP_HEADERS, headersB);
35   * b.option(ClientOption.DECORATION, decorationA);
36   * b.option(ClientOption.DECORATION, decorationB);
37   *
38   * ClientOptions opts = b.build();
39   * HttpHeaders httpHeaders = opts.httpHeaders();
40   * ClientDecoration decorations = opts.decoration();
41   * }</pre>
42   * {@code httpHeaders} will contain all HTTP headers of {@code headersA} and {@code headersB}.
43   * If {@code headersA} and {@code headersB} have the headers with the same name, the duplicate header in
44   * {@code headerB} will replace the one with the same name in {@code headerA}.
45   * Similarly, {@code decorations} will contain all decorators of {@code decorationA} and {@code decorationB},
46   * but there will be no replacement but only addition.
47   */
48  public final class ClientBuilder extends AbstractClientOptionsBuilder<ClientBuilder> {
49  
50      private final URI uri;
51      private ClientFactory factory = ClientFactory.DEFAULT;
52  
53      /**
54       * Creates a new {@link ClientBuilder} that builds the client that connects to the specified {@code uri}.
55       */
56      public ClientBuilder(String uri) {
57          this(URI.create(requireNonNull(uri, "uri")));
58      }
59  
60      /**
61       * Creates a new {@link ClientBuilder} that builds the client that connects to the specified {@link URI}.
62       */
63      public ClientBuilder(URI uri) {
64          this.uri = requireNonNull(uri, "uri");
65      }
66  
67      /**
68       * Sets the {@link ClientFactory} of the client. The default is {@link ClientFactory#DEFAULT}.
69       */
70      public ClientBuilder factory(ClientFactory factory) {
71          this.factory = requireNonNull(factory, "factory");
72          return this;
73      }
74  
75      /**
76       * Creates a new client which implements the specified {@code clientType}.
77       *
78       * @throws IllegalArgumentException if the scheme of the {@code uri} specified in
79       *                                  {@link #ClientBuilder(String)} or the specified {@code clientType} is
80       *                                  unsupported for the scheme
81       */
82      @SuppressWarnings("unchecked")
83      public <T> T build(Class<T> clientType) {
84          requireNonNull(clientType, "clientType");
85          return factory.newClient(uri, clientType, buildOptions());
86      }
87  }
