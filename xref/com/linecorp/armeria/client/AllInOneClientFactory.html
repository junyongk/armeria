17  package com.linecorp.armeria.client;
18  
19  import java.net.URI;
20  import java.util.ArrayList;
21  import java.util.Iterator;
22  import java.util.List;
23  import java.util.Map;
24  import java.util.Optional;
25  import java.util.ServiceLoader;
26  import java.util.Set;
27  import java.util.stream.Collectors;
28  
29  import javax.annotation.Nullable;
30  
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  
34  import com.google.common.collect.ImmutableMap;
35  import com.google.common.collect.Streams;
36  
37  import com.linecorp.armeria.common.Scheme;
38  
39  /**
40   * A {@link ClientFactory} which combines all discovered {@link ClientFactory} implementations.
41   *
42   * <h3>How are the {@link ClientFactory}s discovered?</h3>
43   *
44   * <p>{@link AllInOneClientFactory} looks up the {@link ClientFactoryProvider}s available in the current JVM
45   * using Java SPI (Service Provider Interface). The {@link ClientFactoryProvider} implementations will create
46   * the {@link ClientFactory} implementations.
47   */
48  public class AllInOneClientFactory extends NonDecoratingClientFactory {
49  
50      private static final Logger logger = LoggerFactory.getLogger(AllInOneClientFactory.class);
51  
52      static {
53          if (AllInOneClientFactory.class.getClassLoader() == ClassLoader.getSystemClassLoader()) {
54              Runtime.getRuntime().addShutdownHook(new Thread(ClientFactory::closeDefault));
55          }
56      }
57  
58      private final Map<Scheme, ClientFactory> clientFactories;
59  
60      /**
61       * Creates a new instance with the default {@link SessionOptions}.
62       */
63      public AllInOneClientFactory() {
64          this(false);
65      }
66  
67      /**
68       * Creates a new instance with the default {@link SessionOptions}.
69       *
70       * @param useDaemonThreads whether to create I/O event loop threads as daemon threads
71       */
72      public AllInOneClientFactory(boolean useDaemonThreads) {
73          this(SessionOptions.DEFAULT, useDaemonThreads);
74      }
75  
76      /**
77       * Creates a new instance with the specified {@link SessionOptions}.
78       */
79      public AllInOneClientFactory(SessionOptions options) {
80          this(options, false);
81      }
82  
83      /**
84       * Creates a new instance with the specified {@link SessionOptions}.
85       *
86       * @param useDaemonThreads whether to create I/O event loop threads as daemon threads
87       */
88      public AllInOneClientFactory(SessionOptions options, boolean useDaemonThreads) {
89          super(options, useDaemonThreads);
90  
91          // TODO(trustin): Allow specifying different options for different session protocols.
92          //                We have only one session protocol at the moment, so this is OK so far.
93  
94          // Do not let the delegates create or manage event loops by specifying the EventLoop explicitly.
95          // See NonDecoratingClientFactory.closeEventLoopGroup for more information.
96          final SessionOptions delegateOptions =
97                  SessionOptions.of(options, SessionOption.EVENT_LOOP_GROUP.newValue(eventLoopGroup()));
98  
99          final List<ClientFactoryProvider> providers =
100                 Streams.stream(ServiceLoader.load(ClientFactoryProvider.class,
101                                                   AllInOneClientFactory.class.getClassLoader()))
102                        .collect(Collectors.toCollection(ArrayList::new));
103 
104         if (providers.isEmpty()) {
105             throw new IllegalStateException("could not find any providers");
106         }
107 
108         final List<ClientFactory> availableClientFactories = new ArrayList<>();
109         do {
110             boolean added = false;
111             for (Iterator<ClientFactoryProvider> i = providers.iterator(); i.hasNext();) {
112                 final ClientFactoryProvider p = i.next();
113 
114                 // Find the ClientFactory instances who meet the type requirements specified in
115                 // ClientFactoryProvider.dependencies().
116                 final Map<Class<?>, ClientFactory> requiredClientFactories =
117                         findRequiredClientFactories(p, availableClientFactories);
118 
119                 if (requiredClientFactories == null) {
120                     // Dependencies are not met yet. Try instantiating others first.
121                     continue;
122                 }
123 
124                 // Dependencies are met.
125                 // Remove the provider from the list so that we instantiate a ClientFactory only once.
126                 i.remove();
127 
128                 // Instantiate a ClientFactory and add it to the list of available factories as well.
129                 availableClientFactories.add(p.newFactory(delegateOptions, requiredClientFactories));
130                 added = true;
131             }
132 
133             // If no factories were instantiated, it means none of the remaining providers could find
134             // its dependencies.
135             if (!added) {
136                 throw new IllegalStateException("failed to find the dependencies for: " + providers);
137             }
138         } while (!providers.isEmpty());
139 
140         final ImmutableMap.Builder<Scheme, ClientFactory> builder = ImmutableMap.builder();
141         for (ClientFactory f : availableClientFactories) {
142             f.supportedSchemes().forEach(s -> builder.put(s, f));
143         }
144 
145         clientFactories = builder.build();
146     }
147 
148     /**
149      * Builds the dependency map for the specified {@link ClientFactoryProvider}.
150      *
151      * @return {@code null} if dependencies could not be fully met
152      */
153     @Nullable
154     private static Map<Class<?>, ClientFactory> findRequiredClientFactories(
155             ClientFactoryProvider provider, Iterable<ClientFactory> availableClientFactories) {
156 
157         @SuppressWarnings({ "unchecked", "rawtypes" })
158         final Set<Class<ClientFactory>> dependencyTypes =
159                 (Set<Class<ClientFactory>>) (Set) provider.dependencies();
160 
161         if (dependencyTypes.isEmpty()) {
162             return ImmutableMap.of();
163         }
164 
165         final ImmutableMap.Builder<Class<?>, ClientFactory> builder = ImmutableMap.builder();
166         for (Class<ClientFactory> dependencyType : dependencyTypes) {
167             boolean found = false;
168             for (ClientFactory f : availableClientFactories) {
169                 if (dependencyType.isInstance(f)) {
170                     builder.put(dependencyType, f);
171                     found = true;
172                     break;
173                 }
174             }
175 
176             if (!found) {
177                 // Could not find all dependencies.
178                 return null;
179             }
180         }
181 
182         return builder.build();
183     }
184 
185     @Override
186     public Set<Scheme> supportedSchemes() {
187         return clientFactories.keySet();
188     }
189 
190     @Override
191     public <T> T newClient(URI uri, Class<T> clientType, ClientOptions options) {
192         final Scheme scheme = validateScheme(uri);
193         return clientFactories.get(scheme).newClient(uri, clientType, options);
194     }
195 
196     @Override
197     public <T> Optional<ClientBuilderParams> clientBuilderParams(T client) {
198         for (ClientFactory factory : clientFactories.values()) {
199             Optional<ClientBuilderParams> params = factory.clientBuilderParams(client);
200             if (params.isPresent()) {
201                 return params;
202             }
203         }
204         return Optional.empty();
205     }
206 
207     @Override
208     public void close() {
209         // The global default should never be closed.
210         if (this == ClientFactory.DEFAULT) {
211             logger.debug("Refusing to close the default {}; must be closed via closeDefault()",
212                          ClientFactory.class.getSimpleName());
213             return;
214         }
215 
216         doClose();
217     }
218 
219     void doClose() {
220         try {
221             clientFactories.values().forEach(ClientFactory::close);
222         } finally {
223             super.close();
224         }
225     }
226 }
